  const admin = await getPlatformAdmin(adminId);

    throw new Error('Admin account disabled');
  }
  
  // Check role permissions
  if (action === 'delete_permanent' && admin.role !== 'root_admin') {
    throw new ForbiddenError('Only root_admin can permanently delete');
  }
  
  if (action === 'database_access' && admin.role !== 'root_admin') {
    throw new ForbiddenError('Only root_admin can access database directly');
  }
  
  if (severity === 'delete' && admin.role === 'sysops') {
    throw new ForbiddenError('Sysops cannot delete records');
  }
  
  // Require active support session for data access
  if (action !== 'view_list') {
    const activeSession = await getSupportSession(adminId);
    if (!activeSession) {
      throw new Error('Must start support session before accessing org data');
    }
  }
  
  return true;
}

async function rollbackToSnapshot(workspaceId, snapshotId, adminId) {
  const snapshot = await db.org_snapshots.findOne({ id: snapshotId });
  
  if (!snapshot.can_rollback) {
    throw new Error('Snapshot cannot be rolled back (too old or corrupted)');
  }
  
  // Confirmation required (high-risk action)
  if (!confirmed) {
    return { requiresConfirmation: true };
  }
  
  // Create backup of CURRENT state before rollback (in case rollback needs to be undone)
  const currentStateBackup = await createOrgSnapshot(workspaceId, adminId);
  
  // Begin transaction (all or nothing)
  await db.transaction(async (trx) => {
    const data = snapshot.snapshot_data;
    
    // Restore ALL data
    await trx.employees.deleteMany({ workspaceId });
    await trx.employees.insertMany(data.employees);
    
    await trx.clients.deleteMany({ workspaceId });
    await trx.clients.insertMany(data.clients);
    
    await trx.schedules.deleteMany({ workspaceId });
    await trx.schedules.insertMany(data.schedules);
    
    await trx.timeEntries.deleteMany({ workspaceId });
    await trx.timeEntries.insertMany(data.timeEntries);
    
    // ... restore all other data
    
    // Restore Trinity memory
    await restoreTrinityMemory(workspaceId, data.trinityMemory);
    
    // Update snapshot record
    await trx.org_snapshots.update(snapshotId, {
      rolled_back_at: new Date()
    });
  });
  
  // Log rollback
  await logSupportAction(
    adminId, 
    'rollback_to_snapshot', 
    workspaceId, 
    'org_snapshot', 
    snapshotId, 
    'Rolled back due to fix failure or data corruption'
  );
  
  // Notify org owner
  await emailService.send({
    to: workspace.ownerEmail,
    subject: 'System Restored to Previous State',
    body: `Your CoAIleague system was restored to a previous state by our support team due to an issue during maintenance. All data has been safely restored. If you have questions, please contact support.`
  });
  
  return { success: true, restoredTo: snapshot.created_at };
}
