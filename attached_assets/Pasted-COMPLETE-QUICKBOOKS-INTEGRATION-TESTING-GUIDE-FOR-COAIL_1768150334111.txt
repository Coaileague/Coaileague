  async fullSync() {
    console.log('🚀 Starting QuickBooks Full Sync...');
    
    const syncOrder = [
      'accounts',    // Chart of accounts first
      'customers',   // Then clients
      'employees',   // Then employees
      'vendors',     // Then contractors
      'items',       // Then services
      'invoices',    // Then invoices
      'timeactivities', // Then time entries
      'bills'        // Finally bills
    ];
    
    const results = {};
    
    for (const entity of syncOrder) {
      try {
        results[entity] = await this.syncEntity(entity);
        console.log(`✅ ${entity}: ${results[entity].success} synced, ${results[entity].failed} failed`);
      } catch (error) {
        console.error(`❌ Failed to sync ${entity}:`, error);
        
        // Log to sync table
        await this.logSync({
          sync_type: 'full',
          entity_type: entity,
          status: 'failed',
          error_message: error.message
        });
        
        // Don't stop - continue with other entities
      }
    }
    
    return results;
  },

  // ENTITY-SPECIFIC SYNC METHODS
  async syncEntity(entityType) {
    const config = {
      customers: {
        endpoint: '/v3/company/{companyId}/query?query=select * from Customer',
        table: 'clients',
        mapFunction: this.mapCustomerToClient
      },
      employees: {
        endpoint: '/v3/company/{companyId}/query?query=select * from Employee',
        table: 'employees',
        mapFunction: this.mapEmployeeToEmployee
      },
      vendors: {
        endpoint: '/v3/company/{companyId}/query?query=select * from Vendor',
        table: 'contractors',
        mapFunction: this.mapVendorToContractor
      },
      items: {
        endpoint: '/v3/company/{companyId}/query?query=select * from Item where Type=\'Service\'',
        table: 'services',
        mapFunction: this.mapItemToService
      },
      invoices: {
        endpoint: '/v3/company/{companyId}/query?query=select * from Invoice',
        table: 'invoices',
        mapFunction: this.mapInvoiceToInvoice
      },
      timeactivities: {
        endpoint: '/v3/company/{companyId}/query?query=select * from TimeActivity',
        table: 'time_entries',
        mapFunction: this.mapTimeActivityToTimeEntry
      },
      bills: {
        endpoint: '/v3/company/{companyId}/query?query=select * from Bill',
        table: 'bills',
        mapFunction: this.mapBillToBill
      },
      accounts: {
        endpoint: '/v3/company/{companyId}/query?query=select * from Account',
        table: 'accounts',
        mapFunction: this.mapAccountToAccount
      }
    };
    
    const { endpoint, table, mapFunction } = config[entityType];
    const data = await this.queryQuickBooks(endpoint);
    
    let success = 0;
    let failed = 0;
    
    for (const record of data) {
      try {
        const mapped = mapFunction(record);
        await this.upsertRecord(table, mapped);
        success++;
      } catch (error) {
        failed++;
        console.error(`Failed to sync ${entityType} ${record.Id}:`, error);
      }
    }
    
    return { success, failed };
  },
