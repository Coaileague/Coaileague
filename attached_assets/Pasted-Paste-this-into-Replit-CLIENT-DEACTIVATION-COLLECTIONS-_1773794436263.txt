```sql
CREATE TYPE client_deactivation_reason AS ENUM (
  'non_payment',
  'legal_issue',
  'contract_terminated',
  'contract_non_renewal',
  'lawsuit',
  'unable_to_staff',
  'does_not_meet_billing_requirements',
  'does_not_meet_hourly_requirements',
  'other',
  'no_reason_provided'

ALTER TABLE clients ALTER COLUMN deactivation_reason TYPE client_deactivation_reason USING deactivation_reason::client_deactivation_reason;
```

**Add `collections_status` enum:**
```sql
CREATE TYPE collections_status AS ENUM (
  'none',
  'pending_decision',
  'active',
  'resolved',
  'written_off'
);
ALTER TABLE clients ALTER COLUMN collections_status TYPE collections_status USING collections_status::collections_status;
```

**Create `client_collections_log` table:**
```sql
CREATE TABLE IF NOT EXISTS client_collections_log (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  workspace_id varchar NOT NULL,
  client_id varchar NOT NULL,
  attempt_number integer NOT NULL,
  attempt_type varchar(50) NOT NULL,
  sent_at timestamptz DEFAULT now(),
  sent_to_email varchar(255),
  subject varchar(500),
  body_summary text,
  document_url text,
  outstanding_amount decimal(12,2),
  response_received boolean DEFAULT false,
  response_notes text,
  created_by varchar(255) DEFAULT 'trinity-system'
);
CREATE INDEX idx_collections_log_workspace ON client_collections_log(workspace_id);
CREATE INDEX idx_collections_log_client ON client_collections_log(client_id);
CREATE INDEX idx_collections_log_sent_at ON client_collections_log(sent_at);
```
