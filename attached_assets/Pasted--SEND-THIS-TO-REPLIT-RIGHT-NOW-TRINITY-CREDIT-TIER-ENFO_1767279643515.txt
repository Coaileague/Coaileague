C) Trinity Credit Validation Middleware:
typescript// File: server/services/trinity/trinityCreditEnforcement.ts

class TrinityCreditEnforcement {
  
  /**
   * Trinity checks if user has enough credits BEFORE allowing action
   * This runs for EVERY AI-powered operation
   */
  async validateCreditsBeforeAction(
    workspaceId: string,
    actionType: string,
    userId: string
  ): Promise<CreditValidationResult> {
    
    // 1. Get workspace credit balance
    const workspace = await getWorkspace(workspaceId);
    const currentBalance = workspace.ai_credits_balance;
    const tier = workspace.tier;
    
    // 2. Get credit cost for this action
    const creditsRequired = CREDIT_COSTS[actionType] || 0;
    
    // 3. Enterprise tier = unlimited credits
    if (tier === 'enterprise') {
      return {
        approved: true,
        creditsRequired,
        creditsRemaining: Infinity,
        message: 'Enterprise tier - unlimited credits'
      };
    }
    
    // 4. Check if action requires credits
    if (creditsRequired === 0) {
      // Free action - no credit check needed
      return {
        approved: true,
        creditsRequired: 0,
        creditsRemaining: currentBalance,
        message: 'Action does not require AI credits'
      };
    }
    
    // 5. Gemini analyzes: Is this action worth the credits?
    const worthinessAnalysis = await gemini.analyze(`
Action Requested: ${actionType}
Credits Required: ${creditsRequired}
User Current Balance: ${currentBalance}
User Tier: ${tier}

Context:
- User has used ${workspace.ai_credits_used_this_month} credits this month
- Monthly allowance: ${TIER_CREDITS[tier]}

Should Trinity approve this action?
Consider:
1. Does user have enough credits?
2. Is this action critical vs nice-to-have?
3. Would blocking this harm user experience?
4. Should we suggest upgrade instead?

Respond in JSON:
{
  "approved": boolean,
  "reasoning": string,
  "suggestTopUp": boolean,
  "suggestUpgrade": boolean,
  "alternativeAction": string (if any)
}
    `);
    
    const analysis = JSON.parse(worthinessAnalysis.response.text());
    
    // 6. Check sufficient balance
    if (currentBalance < creditsRequired) {
      
      // Log blocked action
      await db.credit_usage_log.create({
        workspace_id: workspaceId,
        action_type: actionType,
        credits_required: creditsRequired,
        credits_before: currentBalance,
        credits_after: currentBalance, // No change
        approved: false,
        blocked_reason: 'Insufficient credits',
        timestamp: new Date()
      });
      
      // Trinity logs enforcement
      await trinity.log({
        action: 'credit.enforcement.blocked',
        workspaceId,
        actionType,
        creditsRequired,
        currentBalance,
        geminiAnalysis: analysis
      });
      
      return {
        approved: false,
        creditsRequired,
        creditsRemaining: currentBalance,
        creditsNeeded: creditsRequired - currentBalance,
        message: `Insufficient credits. This action requires ${creditsRequired} credits, but you have ${currentBalance}.`,
        suggestTopUp: analysis.suggestTopUp,
        suggestUpgrade: analysis.suggestUpgrade,
        upgradeTo: tier === 'starter' ? 'professional' : 'enterprise'
      };
    }
    
    // 7. Sufficient credits - APPROVED
    return {
      approved: true,
      creditsRequired,
      creditsRemaining: currentBalance,
      message: `Action approved. ${creditsRequired} credits will be deducted.`
    };
  }
  
  /**
   * Trinity deducts credits AFTER successful action
   */
  async deductCredits(
    workspaceId: string,
    actionType: string,
    creditsUsed: number
  ): Promise<void> {
    
    const workspace = await getWorkspace(workspaceId);
    
    // Update balance
    await db.workspaces.update(workspaceId, {
      ai_credits_balance: workspace.ai_credits_balance - creditsUsed,
      ai_credits_used_this_month: workspace.ai_credits_used_this_month + creditsUsed
    });
    
    // Log usage
    await db.credit_usage_log.create({
      workspace_id: workspaceId,
      action_type: actionType,
      credits_required: creditsUsed,
      credits_before: workspace.ai_credits_balance,
      credits_after: workspace.ai_credits_balance - creditsUsed,
      approved: true,
      timestamp: new Date()
    });
    
    // Trinity logs deduction
    await trinity.log({
      action: 'credit.deducted',
      workspaceId,
      actionType,
      creditsUsed,
      remainingBalance: workspace.ai_credits_balance - creditsUsed
    });
    
    // Check if low balance - warn user
    const remainingBalance = workspace.ai_credits_balance - creditsUsed;
    if (remainingBalance < 100 && remainingBalance > 0) {
      await this.sendLowCreditWarning(workspaceId, remainingBalance);
    }
  }
  
  /**
   * Send warning when credits are low
   */
  async sendLowCreditWarning(workspaceId: string, balance: number): Promise<void> {
    const workspace = await getWorkspace(workspaceId);
    
    await emailService.send({
      to: workspace.ownerEmail,
      subject: '⚠️ Low AI Credit Balance',
      body: `Your AI credit balance is low: ${balance} credits remaining.

Trinity AI operations may be limited soon. Consider:
- Top up credits: ${process.env.APP_URL}/billing/credits
- Upgrade tier: ${process.env.APP_URL}/billing/upgrade

Your current tier: ${workspace.tier}
Monthly allowance: ${TIER_CREDITS[workspace.tier]} credits

- Trinity AI`
    });
  }
}

// Trinity tier enforcement
class TrinityTierEnforcement {
  
  async validateTierAccess(
    workspaceId: string,
    feature: string,
    actionDetails?: any
  ): Promise<TierAccessResult> {
    
    const workspace = await getWorkspace(workspaceId);
    const tier = workspace.tier;
    const limits = TIER_LIMITS[tier];
    
    // 1. Check if feature is restricted for this tier
    if (limits.restrictedFeatures.includes(feature)) {
      return {
        allowed: false,
        reason: `Feature '${feature}' requires ${this.getRequiredTier(feature)} tier`,
        currentTier: tier,
        requiredTier: this.getRequiredTier(feature),
        suggestUpgrade: true
      };
    }
    
    // 2. Check employee count limit
    if (actionDetails?.addingEmployee) {
      const currentEmployeeCount = await getEmployeeCount(workspaceId);
      if (limits.maxEmployees !== -1 && currentEmployeeCount >= limits.maxEmployees) {
        return {
          allowed: false,
          reason: `Your ${tier} plan allows up to ${limits.maxEmployees} employees. You have ${currentEmployeeCount}.`,
          currentTier: tier,
          suggestUpgrade: true,
          upgradeMessage: `Upgrade to ${tier === 'starter' ? 'Professional' : 'Enterprise'} for ${tier === 'starter' ? '50' : 'unlimited'} employees`
        };
      }
    }
    
    // 3. Check client limit
    if (actionDetails?.addingClient) {
      const currentClientCount = await getClientCount(workspaceId);
      if (limits.maxClients !== -1 && currentClientCount >= limits.maxClients) {
        return {
          allowed: false,
          reason: `Your ${tier} plan allows up to ${limits.maxClients} clients. You have ${currentClientCount}.`,
          currentTier: tier,
          suggestUpgrade: true
        };
      }
    }
    
    // 4. APPROVED
    return {
      allowed: true,
      currentTier: tier
    };
  }
  
  getRequiredTier(feature: string): string {
    if (TIER_LIMITS.starter.features.includes(feature)) return 'starter';
    if (TIER_LIMITS.professional.features.includes(feature)) return 'professional';
    return 'enterprise';
  }
}
