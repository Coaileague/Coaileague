def run_agent_loop(user_prompt: str, history: list, tools: list):
    # 1. Configuration: Load the System Instruction and tools
    config = types.GenerateContentConfig(
        system_instruction=YOUR_FULL_SYSTEM_INSTRUCTION,
        tools=tools # The list of custom functions
    )
    
    # 2. Call the Gemini API
    response = client.models.generate_content(
        model="gemini-2.5-pro", # Use Pro for complex reasoning
        contents=history + [{"role": "user", "parts": [{"text": user_prompt}]}],
        config=config
    )
    
    # 3. Check for Tool Calls (The Agent's Decision)
    if response.function_calls:
        # A. Execute the tool (e.g., generate_schedule) in Python
        tool_call = response.function_calls[0]
        tool_result = execute_custom_tool(tool_call.name, **dict(tool_call.args))
        
        # B. Send the tool result back to Gemini for final interpretation/formatting
        final_response = client.models.generate_content(
            model="gemini-2.5-pro",
            contents=history + [
                {"role": "user", "parts": [{"text": user_prompt}]},
                {"role": "model", "parts": [response.candidates[0].content.parts[0]]},
                {"role": "tool", "parts": [tool_result]}
            ],
            config=config
        )
        return final_response.text
    
    return response.text
