const isSecureRoute = (pathname) => {
  const protectedPrefixes = ['/dashboard', '/schedules', '/clients', '/guards', '/reports', '/billing', '/settings', '/jobs'];
  const publicRoutes = ['/', '/login', '/signup', '/pricing', '/features', '/about', '/contact', '/demo'];
  
  // Exact match for public routes
  if (publicRoutes.includes(pathname)) return false;
  
  // Check if starts with protected prefix
  return protectedPrefixes.some(prefix => pathname.startsWith(prefix));
};

const useOrgStatus = () => {
  const { user } = useAuth();
  const [orgStatus, setOrgStatus] = useState(null);
  
  useEffect(() => {
    if (!user?.organizationId) return;
    
    // Check if org has active subscription
    const checkOrgStatus = async () => {
      const status = await fetch(`/api/organizations/${user.organizationId}/status`);
      const data = await status.json();
      setOrgStatus(data);
    };
    
    checkOrgStatus();
  }, [user]);
  
  return {
    isPaid: orgStatus?.subscriptionStatus === 'active',
    isTrialActive: orgStatus?.trialEndsAt > new Date(),
    hasTrinityAccess: orgStatus?.features?.includes('trinity_ai')
  };
};

export default async function handler(req, res) {
  // 1. Verify session/token
  const session = await getSession(req, res);
  if (!session?.user) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  
  // 2. Verify org subscription
  const org = await db.organizations.findUnique({
    where: { id: session.user.organizationId }
  });
  
  if (!org || org.subscriptionStatus !== 'active') {
    return res.status(403).json({ 
      error: 'Trinity AI requires an active subscription',
      upgradeUrl: '/pricing'
    });
  }
  
  // 3. Verify Trinity is in org's plan
  if (!org.features.includes('trinity_ai')) {
    return res.status(403).json({ 
      error: 'Trinity AI not included in your plan',
      upgradeUrl: '/pricing?feature=trinity'
    });
  }
  
  // 4. Rate limiting per org
  const rateLimitKey = `trinity_calls:${org.id}`;
  const callCount = await redis.incr(rateLimitKey);
  if (callCount === 1) {
    await redis.expire(rateLimitKey, 3600); // 1 hour window
  }
  if (callCount > org.trinityCallLimit) {
    return res.status(429).json({ 
      error: 'Trinity call limit reached',
      resetAt: new Date(Date.now() + 3600000)
    });
  }
  
  // 5. Log usage for billing
  await db.trinityUsage.create({
    data: {
      organizationId: org.id,
      userId: session.user.id,
      timestamp: new Date(),
      tokensUsed: 0 // Will update after response
    }
  });
  
  // NOW process Trinity request...
  const trinityResponse = await callGeminiAPI(req.body.message);
  
  // Update token usage
  await db.trinityUsage.update({
    where: { id: usageLog.id },
    data: { tokensUsed: trinityResponse.tokensUsed }
  });
  
  return res.json(trinityResponse);
}
