THE FIX: UNIFIED WEBSOCKET
typescript// ONE WebSocket connection for everything
// /client/src/providers/WebSocketProvider.tsx

const WebSocketContext = createContext<WebSocketBus | null>(null);

class WebSocketBus {
  private socket: WebSocket | null = null;
  private handlers: Map<string, Set<(data: any) => void>> = new Map();
  
  connect(userId: string, workspaceId: string) {
    if (this.socket?.readyState === WebSocket.OPEN) return;
    
    this.socket = new WebSocket(`wss://${host}/ws?userId=${userId}&workspaceId=${workspaceId}`);
    
    this.socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      
      // Route message to appropriate handlers based on type
      const typeHandlers = this.handlers.get(data.type);
      if (typeHandlers) {
        typeHandlers.forEach(handler => handler(data));
      }
    };
  }
  
  // Subscribe to specific message types
  subscribe(type: string, handler: (data: any) => void) {
    if (!this.handlers.has(type)) {
      this.handlers.set(type, new Set());
    }
    this.handlers.get(type)!.add(handler);
    
    // Return unsubscribe function
    return () => this.handlers.get(type)?.delete(handler);
  }
  
  send(type: string, payload: any) {
    if (this.socket?.readyState === WebSocket.OPEN) {
      this.socket.send(JSON.stringify({ type, ...payload }));
    }
  }
}

// Each feature hooks into the ONE socket
function useChatMessages(roomId: string) {
  const ws = useContext(WebSocketContext);
  const [messages, setMessages] = useState([]);
  
  useEffect(() => {
    return ws.subscribe('chat_message', (data) => {
      if (data.roomId === roomId) {
        setMessages(prev => [...prev, data.message]);
      }
    });
  }, [roomId]);
  
  return messages;
}

function useNotifications() {
  const ws = useContext(WebSocketContext);
  const [notifications, setNotifications] = useState([]);
  
  useEffect(() => {
    return ws.subscribe('notification', (data) => {
      setNotifications(prev => [...prev, data]);
    });
  }, []);
  
  return notifications;
}
