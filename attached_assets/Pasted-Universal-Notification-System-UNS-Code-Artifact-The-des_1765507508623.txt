The top section uses a horizontal scrollable view or a segmented control for the tabs.

JavaScript

// React Native / TypeScript (Conceptual)

import React, { useState } from 'react';
import { View, Text, StyleSheet, ScrollView, TouchableOpacity } from 'react-native';

const Tabs = ['For You (3)', 'System Alerts (1)', 'Admin Review (1)', 'Updates'];

const NotificationTabs = ({ activeTab, setActiveTab }) => (
    <View style={styles.tabContainer}>
        <ScrollView horizontal showsHorizontalScrollIndicator={false}>
            {Tabs.map((tab) => (

                    key={tab}
                    onPress={() => setActiveTab(tab.split(' ')[0])}
                    style={[
                        styles.tabButton,
                        activeTab.startsWith(tab.split(' ')[0]) && styles.activeTab,

                >
                    <Text style={[styles.tabText, activeTab.startsWith(tab.split(' ')[0]) && styles.activeText]}>
                        {tab}
                    </Text>
                </TouchableOpacity>
            ))}
        </ScrollView>
    </View>
);

2. The Notification Card Component (Responsive Design)
This component uses flexbox (or CSS Grid principles) to achieve the clean, sectioned layout and prioritize the CRITICAL status color.

JavaScript

// NotificationCard.jsx (Conceptual Component)

const NotificationCard = ({ notification }) => {
    // Determine color based on priority
    const priorityColor = getPriorityColor(notification.priority); 
    const isCritical = notification.priority === 'CRITICAL';
    
    return (
        <View style={[styles.card, { borderLeftColor: priorityColor }]}>
            
            {/* 1. CRITICAL BANNER (Top Row) */}
            {isCritical && (
                <View style={[styles.criticalBanner, { backgroundColor: priorityColor }]}>
                    <Text style={styles.criticalText}>{notification.priority}</Text>
                </View>
            )}

            <View style={styles.contentGrid}>
                
                {/* 2. MAIN MESSAGE (Left) */}
                <View style={styles.messageArea}>
                    <Text style={styles.title}>{notification.title}</Text>
                    <Text style={styles.body}>{notification.message_body}</Text>
                    <Text style={styles.time}>{notification.service_source} - {notification.timestamp}</Text>
                </View>

                {/* 3. ACTION BUTTONS (Right) */}
                <View style={styles.actionsArea}>
                    {notification.actions.map((action, index) => (
                        <TouchableOpacity 
                            key={index} 
                            style={[styles.actionButton, isCritical && styles.criticalActionButton]}
                            onPress={() => handleAction(action)}
                        >
                            <Text style={styles.actionButtonText}>{action.label}</Text>
                        </TouchableOpacity>
                    ))}
                </View>
            </View>
            
            {/* 4. STATUS TAG (Bottom Left - e.g., ACTION REQUIRED) */}
            {notification.status_tag && (
                <View style={styles.statusTag}>
                    <Text style={styles.statusTagText}>{notification.status_tag}</Text>
                </View>
            )}
        </View>
    );
};

III. Key Stylesheet Directives for Replit
Tell Replit to focus on these principles for scalability and mobile-first design:

Mobile First Flexbox/Grid: Use display: flex and flex-direction: column by default for the mobile view, ensuring elements stack cleanly. Use justify-content: space-between to separate the message and action areas.

Color Variables: Define CSS variables for the priority colors (--color-critical: #E74C3C;, --color-action: #4A90E2;, etc.) to ensure the LLM-as-Judge status automatically dictates the visual urgency.

Adaptive Action Buttons: The actionsArea must switch from a horizontal stack on larger screens to a vertical column stack on small mobile screens to prevent button overflow.

Priority Border: Use the borderLeftColor property (as shown above) to instantly convey priority visually without taking up too much space.
