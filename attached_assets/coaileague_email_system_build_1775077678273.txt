TABLE 1: platform_email_addresses
  id UUID PRIMARY KEY DEFAULT gen_random_uuid()
  workspace_id UUID REFERENCES workspaces(id)
  user_id UUID REFERENCES users(id)
  client_id UUID REFERENCES clients(id)
  address VARCHAR(320) NOT NULL UNIQUE
  local_part VARCHAR(255) NOT NULL
  subdomain VARCHAR(100)
  display_name VARCHAR(255)
  address_type VARCHAR(50) NOT NULL
    -- 'user_personal' officer personal inbox billed per seat
    -- 'user_client' client inbox billed per seat
    -- 'workspace_system' calloffs etc free auto-handled
    -- 'platform_team' trinity@ support@ etc platform reserved
    -- 'protected_support' name@support.coaileague.com protected
  is_active BOOLEAN DEFAULT false
  is_protected BOOLEAN DEFAULT false
  is_outbound_only BOOLEAN DEFAULT false
  activated_at TIMESTAMPTZ
  activated_by UUID REFERENCES users(id)
  deactivated_at TIMESTAMPTZ
  billing_seat_id VARCHAR(255)
  fair_use_monthly_limit INTEGER DEFAULT 500
  emails_sent_this_period INTEGER DEFAULT 0
  emails_received_this_period INTEGER DEFAULT 0
  auto_trinity_process BOOLEAN DEFAULT false
  trinity_calltype VARCHAR(100)
  created_at TIMESTAMPTZ DEFAULT NOW()
  updated_at TIMESTAMPTZ DEFAULT NOW()

CREATE INDEX idx_email_addresses_address ON platform_email_addresses(address);
CREATE INDEX idx_email_addresses_workspace ON platform_email_addresses(workspace_id);
CREATE INDEX idx_email_addresses_user ON platform_email_addresses(user_id);

TABLE 2: email_routing
  id UUID PRIMARY KEY DEFAULT gen_random_uuid()
  address VARCHAR(320) NOT NULL UNIQUE
  email_address_id UUID REFERENCES platform_email_addresses(id)
  route_type VARCHAR(50) NOT NULL
    -- 'user_inbox' deliver to user mailbox
    -- 'workspace_inbox' deliver to workspace shared inbox
    -- 'trinity_process' Trinity processes automatically
    -- 'support_queue' platform support queue
    -- 'discard' accept but discard spam protection
  target_workspace_id UUID REFERENCES workspaces(id)
  target_user_id UUID REFERENCES users(id)
  target_inbox_type VARCHAR(100)
  auto_process BOOLEAN DEFAULT false
  process_as VARCHAR(100)
    -- 'calloff' 'incident' 'support_ticket' 'application' 'client_message' 'general'
  is_active BOOLEAN DEFAULT true
  created_at TIMESTAMPTZ DEFAULT NOW()

CREATE INDEX idx_email_routing_address ON email_routing(address);

TABLE 3: platform_emails
  id UUID PRIMARY KEY DEFAULT gen_random_uuid()
  workspace_id UUID REFERENCES workspaces(id)
  resend_email_id VARCHAR(255) UNIQUE
  message_id VARCHAR(500)
  in_reply_to VARCHAR(500)
  references_header TEXT
  direction VARCHAR(20) NOT NULL -- 'inbound' or 'outbound'
  from_address VARCHAR(320) NOT NULL
  from_name VARCHAR(255)
  to_addresses TEXT[] NOT NULL
  cc_addresses TEXT[] DEFAULT '{}'
  bcc_addresses TEXT[] DEFAULT '{}'
  reply_to VARCHAR(320)
  subject TEXT
  body_html TEXT
  body_text TEXT
  snippet TEXT -- first 200 chars for list view
  owner_user_id UUID REFERENCES users(id)
  owner_client_id UUID REFERENCES clients(id)
  folder VARCHAR(50) DEFAULT 'inbox'
    -- 'inbox' 'sent' 'drafts' 'starred' 'archive' 'trash' 'spam'
  is_read BOOLEAN DEFAULT false
  is_starred BOOLEAN DEFAULT false
  is_archived BOOLEAN DEFAULT false
  is_deleted BOOLEAN DEFAULT false
  trinity_processed BOOLEAN DEFAULT false
  trinity_processed_at TIMESTAMPTZ
  trinity_category VARCHAR(100)
  trinity_summary TEXT
  trinity_action_taken VARCHAR(100)
  trinity_action_record_id UUID
  trinity_draft_reply TEXT
  trinity_priority VARCHAR(20) DEFAULT 'normal'
  received_at TIMESTAMPTZ
  sent_at TIMESTAMPTZ
  created_at TIMESTAMPTZ DEFAULT NOW()

CREATE INDEX idx_platform_emails_workspace ON platform_emails(workspace_id, folder, created_at DESC);
CREATE INDEX idx_platform_emails_owner ON platform_emails(owner_user_id, folder, is_read);
CREATE INDEX idx_platform_emails_thread ON platform_emails(message_id, in_reply_to);

TABLE 4: email_attachments
  id UUID PRIMARY KEY DEFAULT gen_random_uuid()
  email_id UUID NOT NULL REFERENCES platform_emails(id) ON DELETE CASCADE
  filename VARCHAR(500) NOT NULL
  content_type VARCHAR(200)
  size_bytes INTEGER
  storage_url TEXT
  sha256_hash VARCHAR(64)
  is_inline BOOLEAN DEFAULT false
  content_id VARCHAR(255)
  created_at TIMESTAMPTZ DEFAULT NOW()

CREATE INDEX idx_email_attachments_email ON email_attachments(email_id);

TABLE 5: email_drafts
  id UUID PRIMARY KEY DEFAULT gen_random_uuid()
  workspace_id UUID NOT NULL REFERENCES workspaces(id)
  user_id UUID NOT NULL REFERENCES users(id)
  reply_to_email_id UUID REFERENCES platform_emails(id)
  from_address VARCHAR(320)
  to_addresses TEXT[] DEFAULT '{}'
  cc_addresses TEXT[] DEFAULT '{}'
  subject TEXT
  body_html TEXT
  body_text TEXT
  last_saved_at TIMESTAMPTZ DEFAULT NOW()
  created_at TIMESTAMPTZ DEFAULT NOW()

FUNCTION: provisionWorkspaceAddresses(workspaceId, emailSlug)
  Updates workspaces.email_domain = emailSlug + '.coaileague.com'
  Creates 6 system addresses all is_active=true auto_trinity_process=true:
    staffing-{slug}@coaileague.com  trinity_calltype=staffing_inquiry

    support-{slug}@coaileague.com   trinity_calltype=support_ticket

    billing-{slug}@coaileague.com   trinity_calltype=billing_inquiry
  Creates email_routing entries for all 6 with route_type=trinity_process

FUNCTION: reserveUserEmailAddress(workspaceId, userId, firstName, lastName, emailSlug)
  Generates localPart = firstName.toLowerCase() + '.' + lastName.toLowerCase()
  Generates address = localPart + '@' + emailSlug + '.coaileague.com'
  Inserts into platform_email_addresses with is_active=false
  Updates users.platform_email = address
  Returns: address string

FUNCTION: reserveClientEmailAddress(workspaceId, clientId, clientName, emailSlug)
  Generates localPart from clientName lowercase alphanumeric max 20 chars
  Generates address = localPart + '@' + emailSlug + '.coaileague.com'
  Inserts into platform_email_addresses with is_active=false
  Updates clients.platform_email = address
  Returns: address string

FUNCTION: activateEmailAddress(emailAddressId, activatedByUserId, stripeItemId)
  BEGIN transaction
  UPDATE platform_email_addresses SET is_active=true, activated_at=NOW(),
    activated_by=activatedByUserId, billing_seat_id=stripeItemId WHERE id=emailAddressId
  INSERT into email_routing with route_type=user_inbox
  INSERT into universal_audit_log action=email_activated
  COMMIT
  Throws and rolls back on any error

FUNCTION: deactivateEmailAddress(emailAddressId, deactivatedByUserId)
  UPDATE platform_email_addresses SET is_active=false, deactivated_at=NOW()
  UPDATE email_routing SET is_active=false WHERE email_address_id=emailAddressId

SECTION 6 - INBOUND WEBHOOK HANDLER
================================================================================

Extend existing file: server/routes/inbound/emailWebhookRoute.ts
POST /api/inbound/email

Processing order (must follow exactly):
1. Verify Resend HMAC signature from resend-signature header. Return 401 if invalid.
2. Deduplicate: check platform_emails for resend_email_id. Return 200 if exists.
3. Parse payload: from, to, cc, subject, html, text, message_id, in_reply_to, attachments
4. Look up email_routing by to address (first match wins if multiple to addresses)
5. If no route found: return 200 (accept silently, prevents bounce to sender)
6. INSERT into platform_emails with all fields
7. INSERT into email_attachments for each attachment
8. If route.auto_process=true: call processTrinityEmail() fire-and-forget (do not await)
9. If route.route_type=user_inbox: broadcastToWorkspace new_email event + NDS notification
10. Return 200 to Resend immediately

Trinity email processor function processTrinityEmail({emailId, workspaceId, subject, body, from, processAs, callType}):
  Get workspace tier: SELECT tier FROM workspaces WHERE id=workspaceId
  Call withGemini('gemini-2.5-flash', {workspaceId, tier, callType}, async () => {
    Send prompt asking to classify email and return JSON:
    {
      "intent": processAs type,
      "summary": one line summary,
      "priority": urgent/high/normal/low,
      "extractedData": {},
      "draftReply": suggested reply text
    }
  })
  Based on processAs type take action:
    'calloff': INSERT into calloff_records (workspace_id, source='email', source_email_id, raw_content, status='pending')
    'incident': INSERT into incident_reports (workspace_id, source='email', source_email_id, description, status='open')
    'support_ticket': INSERT into support_tickets (workspace_id, source='email', description, priority)
    'billing_inquiry': store in inbox with high priority NDS to org_owner
    'general': store in inbox with summary
  UPDATE platform_emails SET trinity_processed=true, trinity_category, trinity_summary,
    trinity_action_taken, trinity_action_record_id, trinity_draft_reply, trinity_priority
  Send NDS notification trinity_email_processed to org_owner
