```typescript
  async think(userMessage: string): Promise<TrinityResponse> {
    // 1. Build context
    const systemPrompt = this.buildSystemPrompt();
    const conversationContext = this.buildConversationContext();
    
    // 2. Initial reasoning
    const initialResponse = await this.callModel({
      system: systemPrompt,
      messages: [...conversationContext, { role: 'user', content: userMessage }],
      tools: this.tools.map(t => ({ name: t.name, description: t.description, parameters: t.parameters })),
    });
    
    // 3. Execute tool calls if needed
    if (initialResponse.toolCalls?.length > 0) {
      const toolResults = await this.executeTools(initialResponse.toolCalls);
      
      // 4. Generate final response with tool results
      const finalResponse = await this.callModel({
        system: systemPrompt,
        messages: [
          ...conversationContext,
          { role: 'user', content: userMessage },
          { role: 'assistant', content: initialResponse.content, toolCalls: initialResponse.toolCalls },
          { role: 'tool', content: JSON.stringify(toolResults) },
        ],
      });
      
      return this.buildResponse(finalResponse, toolResults);
    }
    
    return this.buildResponse(initialResponse);
  }
  
  private async executeTools(toolCalls: ToolCall[]): Promise<ToolResult[]> {
    const results: ToolResult[] = [];
    
    for (const call of toolCalls) {
      const tool = this.tools.find(t => t.name === call.name);
      if (tool) {
        try {
          const result = await tool.execute(call.parameters, this.context);
          results.push({ toolName: call.name, success: true, data: result });
        } catch (error) {
          results.push({ toolName: call.name, success: false, error: error.message });
        }
      }
    }
    
    return results;
  }
  
  // Logic loop for complex problems
  async solveComplex(problem: string, maxIterations: number = 5): Promise<TrnitySolution> {
    let iteration = 0;
    let solution: TrnitySolution | null = null;
    
    while (iteration < maxIterations && !solution?.isComplete) {
      iteration++;
      
      // Step 1: Analyze current state
      const analysis = await this.analyze(problem, solution);
      
      // Step 2: Propose next action
      const proposal = await this.propose(analysis);
      
      // Step 3: Validate proposal
      const validation = await this.validate(proposal);
      
      if (!validation.valid) {
        // Iterate with feedback
        solution = { ...solution, feedback: validation.issues };
        continue;
      }
      
      // Step 4: Execute
      const result = await this.execute(proposal);
      
      // Step 5: Evaluate
      solution = await this.evaluate(result, problem);
    }
    
    return solution;
```

```typescript
            errors: validation.errors,
            confidence: 0,
            processingTime: Date.now() - startTime,
          };
        }
      }
      
      return {
        success: true,
        data: result,
        confidence: this.calculateConfidence(result),
        processingTime: Date.now() - startTime,
      };
    } catch (error) {
      return {
        success: false,
        errors: [error.message],
        confidence: 0,
        processingTime: Date.now() - startTime,
      };
    }
  }
  
  async processBatch(tasks: WorkhorseTask[]): Promise<WorkhorseResult[]> {
    // Process in parallel with rate limiting
    const results: WorkhorseResult[] = [];
    const batchSize = 5;
    
    for (let i = 0; i < tasks.length; i += batchSize) {
      const batch = tasks.slice(i, i + batchSize);
      const batchResults = await Promise.all(batch.map(t => this.process(t)));
      results.push(...batchResults);
      
      // Rate limit delay
      if (i + batchSize < tasks.length) {
        await new Promise(resolve => setTimeout(resolve, 1000));
      }
    }
    
    return results;
  }
}
```

## 8.6 AI Orchestration Flow

```typescript
// /server/services/ai/orchestration.ts

/**
 * MASTER ORCHESTRATION
 * 
 * Routes requests to appropriate AI systems
 * and manages the overall flow.
 */

interface AIRequest {
  type: 'chat' | 'analysis' | 'generation' | 'decision';
  content: string;
  context: AIContext;
  requiresJudgment?: boolean;
  confidenceThreshold?: number;
}

interface AIResponse {
  content: string;
  model: string;
  confidence: number;
  toolsUsed: string[];
  tokens: { input: number; output: number };
  cost: number;
  wasJudged: boolean;
  judgment?: Judgment;
}

class AIOrchestrator {
  private trinity: TrinityBrain;
  private judge: ClaudeJudge;
  private workhorse: GPTWorkhorse;
  private subAgents: SubAgentOrchestrator;
  
  async process(request: AIRequest): Promise<AIResponse> {
    const startTime = Date.now();
    
    // 1. Route to appropriate primary handler
    let primaryResponse: any;
    
    switch (request.type) {
      case 'chat':
        primaryResponse = await this.trinity.think(request.content);
        break;
      case 'analysis':
        primaryResponse = await this.workhorse.process({
          type: 'document_analysis',
          input: request.content,
        });
        break;
      case 'generation':
        primaryResponse = await this.workhorse.process({
          type: 'report_generation',
          input: request.content,
        });
        break;
      case 'decision':
        // Complex decisions go through Trinity with mandatory judgment
        primaryResponse = await this.trinity.solveComplex(request.content);
        request.requiresJudgment = true;
        break;
    }
    
    // 2. Check if judgment needed
    const needsJudgment = 
      request.requiresJudgment ||
      primaryResponse.confidence < (request.confidenceThreshold || 0.7);
    
    let judgment: Judgment | undefined;
    let finalResponse = primaryResponse;
    
    if (needsJudgment) {
      judgment = await this.judge.judge({
        originalRequest: request.content,
        aiResponse: primaryResponse.content,
        model: primaryResponse.model,
        confidence: primaryResponse.confidence,
        context: request.context,
      });
      
      // 3. Handle judgment result
      if (judgment.decision === 'REJECT') {
        // Regenerate with Trinity
        finalResponse = await this.trinity.think(
          `${request.content}\n\nPrevious attempt was rejected. Issues: ${judgment.reasoning}`
        );
        
        // Re-judge
        judgment = await this.judge.judge({
          originalRequest: request.content,
          aiResponse: finalResponse.content,
          model: finalResponse.model,
          confidence: finalResponse.confidence,
          context: request.context,
        });
      } else if (judgment.decision === 'REVISE') {
        // Have Claude revise
        finalResponse.content = await this.judge.revise(
          {
            originalRequest: request.content,
            aiResponse: primaryResponse.content,
            model: primaryResponse.model,
            confidence: primaryResponse.confidence,
            context: request.context,
          },
          judgment
        );
      } else if (judgment.decision === 'ESCALATE') {
        // Flag for human review
        await this.escalateToHuman(request, primaryResponse, judgment);
        finalResponse.content = 
          "I've flagged this request for human review. A team member will follow up shortly.";
      }
    }
    
    // 4. Calculate costs and log
    const cost = this.calculateCost(primaryResponse, judgment);
    await this.logUsage(request.context, cost);
    
    return {
      content: finalResponse.content,
      model: finalResponse.model,
      confidence: finalResponse.confidence,
      toolsUsed: finalResponse.toolsUsed || [],
      tokens: finalResponse.tokens,
      cost,
      wasJudged: needsJudgment,
      judgment,
    };
  }
  
  private calculateCost(response: any, judgment?: Judgment): number {
    // Cost per 1K tokens (approximate)
    const COSTS = {
      'gemini-pro': { input: 0.00025, output: 0.0005 },
      'claude-sonnet': { input: 0.003, output: 0.015 },
      'gpt-4-turbo': { input: 0.01, output: 0.03 },
    };
    
    let total = 0;
    
    // Primary model cost
    const primaryCost = COSTS[response.model] || COSTS['gemini-pro'];
    total += (response.tokens.input / 1000) * primaryCost.input;
    total += (response.tokens.output / 1000) * primaryCost.output;
    
    // Judge cost (if used)
    if (judgment) {
      total += 0.01; // Approximate Claude judge cost
    }
    
    return total;
  }
}
```

```typescript
// Ensure critical operations use transactions

// GOOD - Uses transaction
async assignShift(shiftId: string, employeeId: string) {
  return await db.transaction(async (tx) => {
    // Check shift is unassigned
    const shift = await tx.query.shifts.findFirst({
      where: and(
        eq(shifts.id, shiftId),
        isNull(shifts.assignedEmployeeId)
      )
    });
    
    if (!shift) {
      throw new Error('Shift already assigned');
    }
    
    // Assign atomically
    await tx.update(shifts)
      .set({ assignedEmployeeId: employeeId, status: 'assigned' })
      .where(eq(shifts.id, shiftId));
    
    return shift;
  });
}

// BAD - Race condition possible
async assignShiftBAD(shiftId: string, employeeId: string) {
  const shift = await db.query.shifts.findFirst({ where: eq(shifts.id, shiftId) });
  
  // RACE: Another request could assign between check and update!
  if (shift.assignedEmployeeId) {
    throw new Error('Shift already assigned');
  }
  
  await db.update(shifts)
    .set({ assignedEmployeeId: employeeId })
    .where(eq(shifts.id, shiftId));
```
