async function handleTrinityChat(req, res) {
  const { message, workspaceId, userId, mode } = req.body;
  
  // Get conversation history
  const conversation = await getTrinityConversation(workspaceId, userId);
  const history = await getTrinityMessages(conversation.id, limit=20);
  
  // Get workspace context (for business insights)
  const workspaceData = await getWorkspaceContext(workspaceId);
  
  // Build system prompt based on mode
  const systemPrompt = buildTrinityPrompt(mode, workspaceData);
  
  // Call Gemini API
  const response = await gemini.generateContent({
    systemInstruction: systemPrompt,
    contents: formatConversationHistory(history),
    generationConfig: {
      temperature: 0.9,
      maxOutputTokens: 2048
    }
  });
  
  // Save messages
  await saveTrinityMessage(conversation.id, 'user', message, mode);
  await saveTrinityMessage(conversation.id, 'assistant', response.text, mode);
  
  return res.json({ response: response.text });
}
