  // POST-MIGRATION VALIDATION (The "Zero Data Loss" Check)
  async postMigrationValidation(syncResults) {
    console.log('🔍 Running Post-Migration Validation...\n');

    const validators = [
      {
        name: 'Record Count Match',
        test: async () => {
          // Compare QB counts vs Database counts
          const entities = ['customers', 'employees', 'vendors', 'invoices', 'items'];
          const discrepancies = [];

          for (const entity of entities) {
            const qbCount = await QuickBooksService.getCount(entity); // Helper to get QB API count
            const dbCount = await db.query(`SELECT COUNT(*) FROM ${this.getTableFor(entity)}`);
            
            if (parseInt(dbCount) !== qbCount) {
              discrepancies.push(`${entity}: QB(${qbCount}) vs DB(${dbCount})`);
            }
          }

          if (discrepancies.length > 0) {
            throw new Error(`Count Mismatch: ${discrepancies.join(', ')}`);
          }
          return true;
        }
      },
      {
        name: 'Financial Totals Reconciliation',
        test: async () => {
          // Verify total Accounts Receivable matches
          const qbAR = await QuickBooksService.getReceivablesTotal();
          const dbAR = await db.query('SELECT SUM(balance) FROM clients');
          
          const diff = Math.abs(qbAR - dbAR);
          if (diff > 1.00) { // Allow small float variance
            throw new Error(`Accounts Receivable Mismatch: QB($${qbAR}) vs DB($${dbAR})`);
          }
          return true;
        }
      },
      {
        name: 'Orphan Check',
        test: async () => {
          // Ensure all invoices are linked to valid clients
          const orphans = await db.query(`
            SELECT count(*) FROM invoices i 
            LEFT JOIN clients c ON i.client_id = c.id 
            WHERE c.id IS NULL
          `);
          
          if (orphans > 0) {
            throw new Error(`Found ${orphans} orphaned invoices (no client link)`);
          }
          return true;
        }
      }
    ];

    for (const check of validators) {
      try {
        await check.test();
        console.log(`✅ ${check.name}`);
      } catch (error) {
        console.error(`❌ ${check.name}: ${error.message}`);
        // Log critical failure to sync log
        await this.logFailure('validation', check.name, error.message);
      }
    }
  },

  // DATA INTEGRITY & RELATIONSHIP MAPPING
  async checkReferentialIntegrity() {
    console.log('🔗 Checking Referential Integrity...\n');
    
    // 1. Verify Employee SSN Encryption
    const unencryptedSSNs = await db.query(`
      SELECT count(*) FROM employees 
      WHERE ssn IS NOT NULL AND length(ssn) < 20
    `); // Encrypted strings are typically long
    
    if (unencryptedSSNs > 0) {
      console.error('❌ SECURITY ALERT: Found unencrypted SSNs in database');
    } else {
      console.log('✅ Employee SSN Encryption Verified');
    }

    // 2. Verify Service Codes
    const invalidServiceItems = await db.query(`
      SELECT count(*) FROM invoice_line_items li
      LEFT JOIN services s ON li.service_id = s.id
      WHERE s.id IS NULL
    `);

    if (invalidServiceItems > 0) {
      console.error(`❌ Found ${invalidServiceItems} invoice lines with invalid service codes`);
    } else {
      console.log('✅ Service Item Mapping Verified');
    }
  },

  // UTILITIES
  getTableFor(entity) {
    const map = {
      'customers': 'clients',
      'vendors': 'contractors',
      'items': 'services',
      'invoices': 'invoices',
      'employees': 'employees'
    };
    return map[entity];
  },

  async logFailure(type, checkName, message) {
    await db.query(`
      INSERT INTO qb_sync_log (sync_type, operation, status, error_message, started_at)
      VALUES ('validation', $1, 'failed', $2, NOW())
    `, [checkName, message]);
  }
