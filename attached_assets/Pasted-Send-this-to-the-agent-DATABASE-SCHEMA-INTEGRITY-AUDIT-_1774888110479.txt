```sql
-- Get complete list of tables in production DB
SELECT 
  t.table_name,
  COUNT(c.column_name) as column_count,
  pg_size_pretty(pg_total_relation_size(
    quote_ident(t.table_schema) || '.' || quote_ident(t.table_name)
  )) as table_size,
  (SELECT COUNT(*) FROM information_schema.table_constraints tc
   WHERE tc.table_name = t.table_name 
   AND tc.constraint_type = 'FOREIGN KEY') as fk_count,
  (SELECT COUNT(*) FROM pg_indexes pi2 
   WHERE pi2.tablename = t.table_name) as index_count
FROM information_schema.tables t
JOIN information_schema.columns c 
  ON c.table_name = t.table_name 
  AND c.table_schema = t.table_schema
WHERE t.table_schema = 'public'
AND t.table_type = 'BASE TABLE'
GROUP BY t.table_name, t.table_schema
ORDER BY t.table_name;
```

CREATE OR REPLACE FUNCTION trg_auto_clockin_reference()
RETURNS TRIGGER AS $$
DECLARE
  today_str TEXT;
  next_num INTEGER;
BEGIN
  IF NEW.clockin_reference IS NULL THEN
    today_str := TO_CHAR(NOW(), 'YYYYMMDD');
    SELECT COALESCE(MAX(
      CAST(SUBSTRING(clockin_reference FROM LENGTH('CLK-' || today_str || '-') + 1) AS INTEGER)
    ), 0) + 1
    INTO next_num
    FROM time_entries
    WHERE clockin_reference LIKE 'CLK-' || today_str || '-%';
    NEW.clockin_reference := 'CLK-' || today_str || '-' || LPAD(next_num::TEXT, 5, '0');
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_clockin_reference ON time_entries;
CREATE TRIGGER trg_clockin_reference
BEFORE INSERT ON time_entries
FOR EACH ROW EXECUTE FUNCTION trg_auto_clockin_reference();

CREATE OR REPLACE FUNCTION sync_invoice_subtotal()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE invoices SET
    subtotal = (SELECT COALESCE(SUM(amount), 0) FROM invoice_line_items WHERE invoice_id = NEW.invoice_id),
    updated_at = NOW()
  WHERE id = NEW.invoice_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_sync_invoice_subtotal ON invoice_line_items;
CREATE TRIGGER trg_sync_invoice_subtotal
AFTER INSERT OR UPDATE OR DELETE ON invoice_line_items
FOR EACH ROW EXECUTE FUNCTION sync_invoice_subtotal();

-- Verify document_number trigger
CREATE OR REPLACE FUNCTION trg_auto_document_number()
RETURNS TRIGGER AS $$
DECLARE today_str TEXT; next_num INTEGER;
BEGIN
  IF NEW.document_number IS NULL THEN
    today_str := TO_CHAR(NOW(), 'YYYYMMDD');
    SELECT COALESCE(MAX(CAST(SUBSTRING(document_number FROM LENGTH('DOC-' || today_str || '-') + 1) AS INTEGER)), 0) + 1
    INTO next_num FROM document_vault WHERE document_number LIKE 'DOC-' || today_str || '-%';
    NEW.document_number := 'DOC-' || today_str || '-' || LPAD(next_num::TEXT, 5, '0');
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_document_number ON document_vault;
CREATE TRIGGER trg_document_number BEFORE INSERT ON document_vault
FOR EACH ROW EXECUTE FUNCTION trg_auto_document_number();

-- Workspace org_id trigger
CREATE OR REPLACE FUNCTION trg_auto_org_id()
RETURNS TRIGGER AS $$
DECLARE slug TEXT; next_num INTEGER;
BEGIN
  IF NEW.org_id IS NULL THEN
    NEW.org_id := generate_org_id(COALESCE(NEW.state_code, 'TX'));
  END IF;
  IF NEW.email_slug IS NULL AND NEW.company_name IS NOT NULL THEN
    slug := LOWER(REGEXP_REPLACE(NEW.company_name, '[^A-Za-z0-9]', '', 'g'));
    slug := LEFT(slug, 20);
    WHILE EXISTS (SELECT 1 FROM workspaces WHERE email_slug = slug AND id != COALESCE(NEW.id, gen_random_uuid())) LOOP
      slug := LEFT(slug, 18) || FLOOR(RANDOM() * 99)::TEXT;
    END LOOP;
    NEW.email_slug := slug;
    NEW.trinity_email := 'trinity.' || slug || '@mail.coaileague.com';
    NEW.careers_email := 'careers.' || slug || '@mail.coaileague.com';
    NEW.verify_email := 'verify.' || slug || '@mail.coaileague.com';
    NEW.support_email := 'support.' || slug || '@mail.coaileague.com';
    NEW.calloffs_email := 'calloffs.' || slug || '@mail.coaileague.com';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_workspace_org_id ON workspaces;
CREATE TRIGGER trg_workspace_org_id
BEFORE INSERT ON workspaces
FOR EACH ROW EXECUTE FUNCTION trg_auto_org_id();

-- Client number trigger
CREATE OR REPLACE FUNCTION trg_auto_client_number()
RETURNS TRIGGER AS $$
DECLARE org_short TEXT; next_num INTEGER;
BEGIN
  IF NEW.client_number IS NULL THEN
    SELECT UPPER(LEFT(REGEXP_REPLACE(company_name, '[^A-Za-z]', '', 'g'), 3))
    INTO org_short FROM workspaces WHERE id = NEW.workspace_id;
    IF org_short IS NULL OR LENGTH(org_short) < 3 THEN org_short := 'CLT'; END IF;
    SELECT COALESCE(MAX(CAST(SUBSTRING(client_number FROM LENGTH('CLT-' || org_short || '-') + 1) AS INTEGER)), 0) + 1
    INTO next_num FROM clients WHERE workspace_id = NEW.workspace_id;
    NEW.client_number := 'CLT-' || org_short || '-' || LPAD(next_num::TEXT, 5, '0');
  END IF;
  RETURN NEW;
END;
