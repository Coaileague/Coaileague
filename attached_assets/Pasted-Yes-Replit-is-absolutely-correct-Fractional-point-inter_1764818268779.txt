// Add this helper method to the class
drawPolygon(ctx, x, y, radius, points, innerRatio, color, time) {
    ctx.fillStyle = color;
    // Glow effect
    if (!ctx.fillStyle.includes('rgba')) {
        ctx.shadowBlur = 15; ctx.shadowColor = color;
    }

    ctx.beginPath();
    
    // LOGIC: We round 'points' to ensure the loop runs, 
    // but the LERP in the update loop ensures the transition feels smooth.
    const p = Math.max(3, Math.round(points)); 
    const step = Math.PI / p;
    const rotation = -(time * 0.05);

    for(let i=0; i<2*p; i++) {
        // Alternating radius creates the star/shape effect
        const r = (i % 2 === 0) ? radius : radius * innerRatio;
        const a = i * step + rotation;
        ctx.lineTo(x + Math.cos(a) * r, y + Math.sin(a) * r);
    }
    ctx.closePath();
    ctx.fill();
    
    // Reset shadow
    ctx.shadowBlur = 0;
}
