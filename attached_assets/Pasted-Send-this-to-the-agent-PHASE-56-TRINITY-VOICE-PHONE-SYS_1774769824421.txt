-- Organization ID generator
CREATE OR REPLACE FUNCTION generate_org_id(p_state_code TEXT DEFAULT 'TX')
RETURNS TEXT AS $$
DECLARE
  next_num INTEGER;
  new_id TEXT;
BEGIN
  SELECT COALESCE(MAX(
    CAST(SUBSTRING(org_id FROM LENGTH('ORG-' || p_state_code || '-') + 1) AS INTEGER)
  ), 0) + 1
  INTO next_num
  FROM workspaces
  WHERE org_id LIKE 'ORG-' || p_state_code || '-%';
  
  new_id := 'ORG-' || p_state_code || '-' || LPAD(next_num::TEXT, 5, '0');
  RETURN new_id;
END;
$$ LANGUAGE plpgsql;

-- Client number generator (per workspace)
CREATE OR REPLACE FUNCTION generate_client_number(p_workspace_id UUID)
RETURNS TEXT AS $$
DECLARE
  org_short TEXT;
  next_num INTEGER;
BEGIN
  SELECT UPPER(LEFT(REGEXP_REPLACE(company_name, '[^A-Za-z]', '', 'g'), 3))
  INTO org_short
  FROM workspaces WHERE id = p_workspace_id;
  
  IF org_short IS NULL OR LENGTH(org_short) < 3 THEN
    org_short := 'ORG';
  END IF;
  
  SELECT COALESCE(MAX(
    CAST(SUBSTRING(client_number FROM LENGTH('CLT-' || org_short || '-') + 1) AS INTEGER)
  ), 0) + 1
  INTO next_num
  FROM clients
  WHERE workspace_id = p_workspace_id;
  
  RETURN 'CLT-' || org_short || '-' || LPAD(next_num::TEXT, 5, '0');
END;
$$ LANGUAGE plpgsql;

-- Employee number generator (per workspace)
CREATE OR REPLACE FUNCTION generate_employee_number(p_workspace_id UUID)
RETURNS TEXT AS $$
DECLARE
  org_short TEXT;
  next_num INTEGER;
BEGIN
  SELECT UPPER(LEFT(REGEXP_REPLACE(company_name, '[^A-Za-z]', '', 'g'), 3))
  INTO org_short
  FROM workspaces WHERE id = p_workspace_id;
  
  IF org_short IS NULL OR LENGTH(org_short) < 3 THEN
    org_short := 'EMP';
  END IF;
  
  SELECT COALESCE(MAX(
    CAST(SUBSTRING(employee_number FROM LENGTH('EMP-' || org_short || '-') + 1) AS INTEGER)
  ), 0) + 1
  INTO next_num
  FROM employees
  WHERE workspace_id = p_workspace_id
  AND employee_number LIKE 'EMP-' || org_short || '-%';
  
  RETURN 'EMP-' || org_short || '-' || LPAD(next_num::TEXT, 5, '0');
END;
$$ LANGUAGE plpgsql;

RETURNS TEXT AS $$

  today_str TEXT;
  next_num INTEGER;

  INTO next_num
  FROM document_vault

  RETURN 'DOC-' || today_str || '-' || LPAD(next_num::TEXT, 5, '0');
END;
$$ LANGUAGE plpgsql;

RETURNS TEXT AS $$
DECLARE
  period_str TEXT;
  next_num INTEGER;
BEGIN
  period_str := TO_CHAR(NOW(), 'YYYYMM');

  WHERE workspace_id = p_workspace_id

Build `server/services/universalIdResolver.ts`:

```typescript
// Resolves any ID prefix to the correct entity
// Used by Trinity, voice system, search, and documents

export async function resolveEntityById(
  id: string, 
  workspaceId: string
): Promise<{ entityType: string; entity: any; deepLink: string } | null> {
  
  const prefix = id.split('-')[0];
  
  switch(prefix) {
    case 'ORG':
      // Platform staff only
      const org = await db.query(
        'SELECT * FROM workspaces WHERE org_id = $1', [id]
      );
      return org.rows[0] ? {
        entityType: 'workspace',
        entity: org.rows[0],
        deepLink: '/admin/workspaces/' + org.rows[0].id
      } : null;
      
    case 'CLT':
      const client = await db.query(
        'SELECT * FROM clients WHERE client_number = $1 AND workspace_id = $2',
        [id, workspaceId]
      );
      return client.rows[0] ? {
        entityType: 'client',
        entity: client.rows[0],
        deepLink: '/clients/' + client.rows[0].id
      } : null;
      
    case 'EMP':
      const employee = await db.query(
        'SELECT * FROM employees WHERE employee_number = $1 AND workspace_id = $2',
        [id, workspaceId]
      );
      return employee.rows[0] ? {
        entityType: 'employee',
        entity: employee.rows[0],
        deepLink: '/officers/' + employee.rows[0].id
      } : null;
      
    case 'DOC':
      const doc = await db.query(
        'SELECT * FROM document_vault WHERE document_number = $1 AND workspace_id = $2',
        [id, workspaceId]
      );
      return doc.rows[0] ? {
        entityType: 'document',
        entity: doc.rows[0],
        deepLink: '/documents/' + doc.rows[0].id
      } : null;
      
    case 'INV':
      const invoice = await db.query(
        'SELECT * FROM invoices WHERE invoice_number = $1 AND workspace_id = $2',
        [id, workspaceId]
      );
      return invoice.rows[0] ? {
        entityType: 'invoice',
        entity: invoice.rows[0],
        deepLink: '/invoices/' + invoice.rows[0].id
      } : null;
      
    case 'CLK':
      const clockin = await db.query(
        'SELECT * FROM time_entries WHERE clockin_reference = $1 AND workspace_id = $2',
        [id, workspaceId]
      );
      return clockin.rows[0] ? {
        entityType: 'time_entry',
        entity: clockin.rows[0],
        deepLink: '/time-attendance/' + clockin.rows[0].id
      } : null;
      
    default:
      return null;
  }
}
```

→ Store in employees.clockin_pin_hash
→ Store in clockin_pin_history
→ Return: { success: true, message: 'PIN set successfully' }

// Verify PIN (used by voice system)
POST /api/clockin-pin/verify
{ employeeNumber: 'EMP-ACM-00034', pin: '847291' }
→ Find employee by employee_number
→ Compare pin against bcrypt hash
→ Rate limit: 5 attempts per 15 minutes per employee
→ On fail 5 times: lock for 30 minutes, NDS to supervisor
→ Return: { verified: true/false, employeeId, workspaceId }
// Note: never return employee data on failed verify

// Reset PIN (manager action)
POST /api/clockin-pin/reset/:employeeId
requireAuth, requireRole('org_owner', 'org_manager')
→ Clear existing PIN hash
→ Send NDS to officer to set new PIN
→ Officer must set new PIN before next voice clock-in

    }
    
    const workspace = await db.query(
      'SELECT * FROM workspaces WHERE id = $1',
      [phoneNumber.rows[0].workspace_id]
    );
    
    // 2. Check credit balance before answering
    const credits = await this.checkCreditBalance(workspace.rows[0].id);
    if (credits.balance_cents < 50) { // Less than $0.50
      return this.buildTwiml(response => {
        response.say({ voice: 'Polly.Joanna' },
          `Thank you for calling ${workspace.rows[0].company_name}. 
           We are unable to process your call at this time. 
           Please contact us directly.`);
        response.hangup();
      });
    }
    
    // 3. Create call session
    const session = await db.query(`
      INSERT INTO voice_call_sessions 
      (workspace_id, phone_number_id, call_sid, caller_phone, call_status)
      VALUES ($1, $2, $3, $4, 'in_progress')
      RETURNING id
    `, [workspace.rows[0].id, phoneNumber.rows[0].id, CallSid, From]);
    
    // 4. Check caller identity from phone number
    const callerIdentity = await this.identifyCaller(From, workspace.rows[0].id);
    
    // 5. Present main menu
    return this.buildMainMenu(workspace.rows[0], session.rows[0].id, callerIdentity);
  }
  
  // Build TwiML main menu
  async buildMainMenu(workspace: any, sessionId: string, callerIdentity: any): Promise<string> {
    const twiml = new VoiceResponse();
    const companyName = workspace.company_name;
    
    // Recording announcement
    twiml.say({ voice: 'Polly.Joanna', language: 'en-US' },

  async buildMainIVR(sessionId: string, lang: string, workspace: any): Promise<string> {
    const twiml = new VoiceResponse();
    const companyName = workspace.company_name;
    
    const gather = twiml.gather({

                Para empleo y carreras, oprima 6.
                Para hablar con alguien, oprima 0.`
    } : {
      greeting: `Thank you for calling ${companyName}.`,
      options: `For new security services, press 1.
                For existing client support, press 2.
                For employment verification, press 3.
                For staff support, press 4.
                For after-hours emergencies, press 5.
                For careers and employment, press 6.

  // Identify caller from phone number
  async identifyCaller(phone: string, workspaceId: string) {
    const employee = await db.query(
      'SELECT id, first_name, last_name, employee_number FROM employees WHERE phone = $1 AND workspace_id = $2 AND status = $3',
      [phone, workspaceId, 'active']
    );
    if (employee.rows[0]) return { type: 'employee', data: employee.rows[0] };
    
    const client = await db.query(
      'SELECT id, company_name, client_number FROM clients WHERE contact_phone = $1 AND workspace_id = $2',
      [phone, workspaceId]
    );
    if (client.rows[0]) return { type: 'client', data: client.rows[0] };
    
    const lead = await db.query(
      'SELECT id, contact_name, company_name FROM sales_leads WHERE contact_phone = $1 AND workspace_id = $2',
      [phone, workspaceId]
    );
    if (lead.rows[0]) return { type: 'prospect', data: lead.rows[0] };
    
    return { type: 'unknown', data: null };
  }
  
  buildTwiml(builder: (response: any) => void): string {
    const twiml = new VoiceResponse();
    builder(twiml);
    return twiml.toString();
  }
  
  // Deduct call cost from org credit balance
  async deductCallCost(workspaceId: string, durationSeconds: number, ratePerMinuteCents: number) {
    const cost = Math.ceil(durationSeconds / 60) * ratePerMinuteCents;
    
    await db.query(`
      UPDATE voice_credit_accounts 
      SET balance_cents = balance_cents - $1,
          total_spent_cents = total_spent_cents + $1,
          updated_at = NOW()
      WHERE workspace_id = $2 AND balance_cents >= $1
      RETURNING balance_cents
    `, [cost, workspaceId]);
    
    await db.query(`
      INSERT INTO voice_credit_transactions
      (workspace_id, transaction_type, amount_cents, balance_after_cents, description)
      SELECT $1, 'call_deduction', $2, balance_cents, $3
      FROM voice_credit_accounts WHERE workspace_id = $1
    `, [workspaceId, -cost, `Call ${durationSeconds}s at ${ratePerMinuteCents}¢/min`]);
    
    // Alert if balance low
    const balance = await this.checkCreditBalance(workspaceId);
    if (balance.balance_cents < 500) { // Less than $5
      await notificationDeliveryService.send({
        workspaceId,
        type: 'voice_credit_low',
        data: { balance: balance.balance_cents / 100 }
      });
    }
  }
  
  async checkCreditBalance(workspaceId: string) {
    const result = await db.query(
      'SELECT * FROM voice_credit_accounts WHERE workspace_id = $1',
      [workspaceId]
    );
    return result.rows[0] || { balance_cents: 0 };
  }

// Verify all requests are genuinely from Twilio
function validateTwilioSignature(req: Request, res: Response, next: NextFunction) {
  const authToken = process.env.TWILIO_AUTH_TOKEN!;
  const signature = req.headers['x-twilio-signature'] as string;

}

export async function handleVoiceClockInVerify(req: Request, sessionId: string): Promise<string> {
  const { SpeechResult, Digits } = req.body;
  const input = SpeechResult || Digits;
  
  // Parse employee number from speech or DTMF
  // Speech: "EMP ACM 00034" → "EMP-ACM-00034"
  const employeeNumber = parseEmployeeNumber(input);
  
  // Look up employee
  const employee = await db.query(
    'SELECT * FROM employees WHERE employee_number = $1',
    [employeeNumber]
  );
  
  if (!employee.rows[0]) {
    const twiml = new VoiceResponse();
    twiml.say({ voice: 'Polly.Joanna' },

    );
    twiml.redirect(`/api/voice/ext/staff/clockin?sessionId=${sessionId}`);
    return twiml.toString();
  }
  
  // Store employee in session
  await db.query(
    'UPDATE voice_call_sessions SET caller_identified_as_employee_id = $1 WHERE id = $2',
    [employee.rows[0].id, sessionId]
  );
  
  // Check if employee has PIN set
  if (!employee.rows[0].clockin_pin_hash) {
    const twiml = new VoiceResponse();
    twiml.say({ voice: 'Polly.Joanna' },
      `Hi ${employee.rows[0].first_name}. You have not set up your voice PIN yet. 
       Please log into the platform to set your 6-digit clock-in PIN before using voice clock-in.`
    );
    twiml.hangup();
    return twiml.toString();
  }
  
  // Request PIN
  const twiml = new VoiceResponse();
  const gather = twiml.gather({
    numDigits: '6',
    action: `/api/voice/ext/staff/clockin/pin?sessionId=${sessionId}&employeeId=${employee.rows[0].id}`,

  });
  
  gather.say({ voice: 'Polly.Joanna' },

export async function handleVoiceClockInPin(req: Request, sessionId: string, employeeId: string): Promise<string> {
  const { Digits } = req.body;
  
  // Verify PIN
  const verified = await clockinPinService.verifyPin(employeeId, Digits);
  
  if (!verified) {
    const twiml = new VoiceResponse();
    twiml.say({ voice: 'Polly.Joanna' },
      'Incorrect PIN. Please try again.'
    );
    twiml.redirect(`/api/voice/ext/staff/clockin/pin?sessionId=${sessionId}&employeeId=${employeeId}`);
    return twiml.toString();
  }
  
  // PIN verified — check if can clock in
  const employee = await db.query(
    'SELECT e.*, w.company_name FROM employees e JOIN workspaces w ON w.id = e.workspace_id WHERE e.id = $1',
    [employeeId]
  );
  const emp = employee.rows[0];
  
  // License check
  if (emp.license_expiry && new Date(emp.license_expiry) < new Date()) {
    const twiml = new VoiceResponse();
    twiml.say({ voice: 'Polly.Joanna' },
      `I cannot complete your clock-in. Your security license expired on ${formatDate(emp.license_expiry)}. 
       Please contact your supervisor immediately.`
    );
    twiml.hangup();
    return twiml.toString();
  }
  
  // Find active shift
  const shift = await db.query(`
    SELECT s.*, p.name as post_name 
    FROM shifts s
    JOIN posts p ON p.id = s.post_id
    WHERE s.assigned_officer_id = $1
    AND s.shift_date = CURRENT_DATE
    AND s.start_time <= NOW()::TIME + INTERVAL '15 minutes'
    AND s.end_time > NOW()::TIME
    AND s.status = 'published'
  `, [employeeId]);
  
  const twiml = new VoiceResponse();
  
  if (shift.rows[0]) {
    // Create time entry
    const timeEntry = await db.query(`
      INSERT INTO time_entries 
      (workspace_id, employee_id, shift_id, post_id, clock_in, 
       clock_in_method, clockin_reference, status)
      VALUES ($1, $2, $3, $4, NOW(), 'voice_phone', generate_clockin_reference(), 'active')
      RETURNING *
    `, [emp.workspace_id, employeeId, shift.rows[0].id, shift.rows[0].post_id]);
    
    // Notify supervisor
    await notificationDeliveryService.send({
      workspaceId: emp.workspace_id,
      type: 'officer_voice_clockin',
      data: {
        officerName: `${emp.first_name} ${emp.last_name}`,
        employeeNumber: emp.employee_number,
        postName: shift.rows[0].post_name,
        time: new Date().toLocaleTimeString(),
        reference: timeEntry.rows[0].clockin_reference
      }
    });
    
    // Broadcast to dashboard
    broadcastToWorkspace(emp.workspace_id, {
      type: 'officer_clocked_in',
      data: { employeeId, postName: shift.rows[0].post_name }
    });
    
    twiml.say({ voice: 'Polly.Joanna' },
      `Confirmed. You are clocked in for your shift at ${shift.rows[0].post_name}. 
       Your clock-in reference is ${timeEntry.rows[0].clockin_reference}. 
       Your supervisor has been notified. Have a safe shift ${emp.first_name}.`
    );
  } else {
    // No scheduled shift found
    twiml.say({ voice: 'Polly.Joanna' },
      `I do not see a scheduled shift for you right now. 
       If you are working an emergency or unscheduled shift please press 1.
       Otherwise press 2 to return to the main menu.`
    );
    // Handle response...
  }
  
  twiml.hangup();
  return twiml.toString();
}

**EMAIL INTERVIEW PIPELINE:**

Trinity processes incoming applications via the existing inbound email system. Each org gets:
`staffing-[orgshort]@mail.coaileague.com` — existing
`careers-[orgshort]@mail.coaileague.com` — new for applications

When application email arrives:
1. Trinity extracts: name, contact info, position interest, any resume text
2. Trinity scores initial qualification: 0-100
3. If score ≥ 60: sends Email Round 1 (3-5 questions adaptive to position)
4. Candidate replies: Trinity scores responses, adjusts Round 2 questions
5. If score ≥ 75 after Round 2: invites to chat interview
6. All tracked in interview_sessions table
7. Dashboard shows ranked candidates
