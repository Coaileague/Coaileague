            {/* Modal Header */}
            <div className="sticky top-0 bg-gradient-to-r from-blue-600 to-purple-600 text-white p-4 rounded-t-2xl">
              <div className="flex items-center justify-between">
                <h2 className="text-lg font-bold">Create Shift</h2>
                <button onClick={() => setShowShiftModal(false)} className="p-2 hover:bg-white hover:bg-opacity-20 rounded-lg">
                  <X className="w-6 h-6" />
                </button>
              </div>
            </div>

            {/* Modal Body */}
            <div className="p-4 space-y-4">
              {/* Open Shift Toggle */}
              <div className="bg-orange-50 border-2 border-orange-200 rounded-lg p-3">
                <div className="flex items-center justify-between">
                  <div className="flex items-center space-x-2">
                    <AlertCircle className="w-5 h-5 text-orange-600" />
                    <div>
                      <div className="font-semibold text-gray-900 text-sm">Open Shift</div>
                      <div className="text-xs text-gray-600">Let AI fill later</div>
                    </div>
                  </div>
                  <button
                    onClick={() => setShiftForm(prev => ({ ...prev, isOpenShift: !prev.isOpenShift }))}
                    className={`relative inline-flex h-6 w-11 items-center rounded-full transition-colors ${
                      shiftForm.isOpenShift ? 'bg-orange-600' : 'bg-gray-300'
                    }`}
                  >
                    <span className={`inline-block h-4 w-4 transform rounded-full bg-white transition-transform ${
                      shiftForm.isOpenShift ? 'translate-x-6' : 'translate-x-1'
                    }`} />
                  </button>
                </div>
              </div>

              {/* Employee Selection */}
              {!shiftForm.isOpenShift && (
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-2">Employee</label>
                  <select
                    value={shiftForm.employeeId || ''}
                    onChange={(e) => setShiftForm(prev => ({ ...prev, employeeId: parseInt(e.target.value) }))}
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                  >
                    <option value="">Select employee...</option>
                    {employees.map(emp => (
                      <option key={emp.id} value={emp.id}>
                        {emp.name} - {emp.position} (Score: {emp.score})
                      </option>
                    ))}
                  </select>
                </div>
              )}

              {/* Position & Client */}
              <div className="grid grid-cols-2 gap-3">
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-2">Position</label>
                  <input
                    type="text"
                    value={shiftForm.position}
                    onChange={(e) => setShiftForm(prev => ({ ...prev, position: e.target.value }))}
                    placeholder="Server, Cook..."
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                  />
                </div>
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-2">Client/Area</label>
                  <input
                    type="text"
                    value={shiftForm.client}
                    onChange={(e) => setShiftForm(prev => ({ ...prev, client: e.target.value }))}
                    placeholder="Main Dining..."
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                  />
                </div>
              </div>

              {/* Time */}
              <div className="grid grid-cols-2 gap-3">
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-2">Clock In</label>
                  <input
                    type="time"
                    value={shiftForm.clockIn}
                    onChange={(e) => setShiftForm(prev => ({ ...prev, clockIn: e.target.value }))}
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                  />
                </div>
                <div>
                  <label className="block text-sm font-medium text-gray-700 mb-2">Clock Out</label>
                  <input
                    type="time"
                    value={shiftForm.clockOut}
                    onChange={(e) => setShiftForm(prev => ({ ...prev, clockOut: e.target.value }))}
                    className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                  />
                </div>
              </div>

              {/* Location */}
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Location</label>
                <input
                  type="text"
                  value={shiftForm.location}
                  onChange={(e) => setShiftForm(prev => ({ ...prev, location: e.target.value }))}
                  placeholder="Specific location or address"
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                />
              </div>

              {/* Notes */}
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Notes</label>
                <textarea
                  value={shiftForm.notes}
                  onChange={(e) => setShiftForm(prev => ({ ...prev, notes: e.target.value }))}
                  placeholder="Add any special instructions..."
                  rows={2}
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                />
              </div>

              {/* Post Orders */}
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-2">Post Orders</label>
                <div className="space-y-2">
                  {availablePostOrders.map(order => (
                    <div
                      key={order.id}
                      onClick={() => togglePostOrder(order.id)}
                      className={`border-2 rounded-lg p-3 ${
                        shiftForm.postOrders.includes(order.id)
                          ? 'border-blue-500 bg-blue-50'
                          : 'border-gray-200'
                      }`}
                    >
                      <div className="flex items-start justify-between mb-1">
                        <div className="flex items-start space-x-2">
                          <div className={`mt-0.5 w-5 h-5 rounded border-2 flex items-center justify-center ${
                            shiftForm.postOrders.includes(order.id)
                              ? 'border-blue-500 bg-blue-500'
                              : 'border-gray-300'
                          }`}>
                            {shiftForm.postOrders.includes(order.id) && (
                              <CheckCircle className="w-4 h-4 text-white" />
                            )}
                          </div>
                          <div className="flex-1">
                            <div className="font-semibold text-gray-900 text-sm">{order.title}</div>
                            <div className="text-xs text-gray-600 mt-0.5">{order.description}</div>
                          </div>
                        </div>
                      </div>
                      
                      <div className="flex flex-wrap gap-1 mt-2 ml-7">
                        {order.requiresAcknowledgment && (
                          <span className="text-xs bg-yellow-100 text-yellow-800 px-2 py-0.5 rounded">
                            ✓ Acknowledge
                          </span>
                        )}
                        {order.requiresSignature && (
                          <span className="text-xs bg-purple-100 text-purple-800 px-2 py-0.5 rounded">
                            ✍️ Sign
                          </span>
                        )}
                        {order.requiresPhotos && (
                          <span className="text-xs bg-blue-100 text-blue-800 px-2 py-0.5 rounded">
                            📷 {order.photoFrequency}
                          </span>
                        )}
                      </div>
                    </div>
                  ))}
                </div>
              </div>
            </div>

            {/* Modal Footer */}
            <div className="sticky bottom-0 bg-white border-t border-gray-200 p-4">
              <button
                onClick={handleSaveShift}
                disabled={!shiftForm.position || (!shiftForm.isOpenShift && !shiftForm.employeeId)}
                className="w-full bg-gradient-to-r from-blue-600 to-purple-600 text-white py-3 rounded-lg font-medium disabled:opacity-50 disabled:cursor-not-allowed"
              >
                Create Shift
              </button>
            </div>
          </div>
        </div>
      )}

      {/* AI Panel Drawer */}
      {showAIPanel && (
        <div className="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-end">
          <div className="bg-white rounded-t-2xl w-full max-h-[80vh] overflow-y-auto">
            <div className="sticky top-0 bg-gradient-to-r from-purple-600 to-blue-600 text-white p-4 rounded-t-2xl">
              <div className="flex items-center justify-between">
                <div>
                  <h2 className="text-lg font-bold">AI Recommendations</h2>
                  <p className="text-sm opacity-90">{pendingAIChanges.length} pending changes</p>
                </div>
                <button onClick={() => setShowAIPanel(false)} className="p-2 hover:bg-white hover:bg-opacity-20 rounded-lg">
                  <X className="w-6 h-6" />
                </button>
              </div>
            </div>

            <div className="p-4 space-y-3">
              {/* AI Status */}
              <div className="bg-gradient-to-r from-purple-50 to-blue-50 border border-purple-200 rounded-lg p-3">
                <div className="flex items-center justify-between mb-2">
                  <span className="text-sm font-medium text-gray-900">AI Confidence</span>
                  <span className="text-lg font-bold text-purple-600">{aiConfidence}%</span>
                </div>
                <div className="w-full h-2 bg-gray-200 rounded-full overflow-hidden">
                  <div 
                    className="h-full bg-gradient-to-r from-purple-600 to-blue-600"
                    style={{ width: `${aiConfidence}%` }}
                  />
                </div>
                <div className="mt-2 text-xs text-gray-600">
                  Test Runs: {testRunCount}/20 • {20 - testRunCount} more for full automation
                </div>
              </div>

              {/* AI Changes */}
              {pendingAIChanges.map(change => (
                <div key={change.id} className="border border-gray-200 rounded-lg p-3 bg-white">
                  <div className="flex items-start justify-between mb-2">
                    <div className="flex items-center space-x-2">
                      <Bot className="w-5 h-5 text-purple-600" />
                      <span className="font-semibold text-gray-900 text-sm">
                        {change.type === 'new_shift' ? 'New Shift' : 'Shift Swap'}
                      </span>
                    </div>
                    <span className="text-xs font-bold text-purple-600">{change.confidence}%</span>
                  </div>

                  {change.type === 'new_shift' ? (
                    <div className="space-y-1 mb-2 text-sm">
                      <div className="flex items-center space-x-2">
                        <User className="w-4 h-4 text-gray-400" />
                        <span>{change.employee.name}</span>
                      </div>
                      <div className="flex items-center space-x-2">
                        <Clock className="w-4 h-4 text-gray-400" />
                        <span>{formatTime(change.startHour)} - {formatTime(change.startHour + change.duration)}</span>
                      </div>
                    </div>
                  ) : (
                    <div className="space-y-1 mb-2 text-sm">
                      <div>From: {change.fromEmployee.name}</div>
                      <div>To: {change.toEmployee.name}</div>
                    </div>
                  )}

                  <div className="bg-blue-50 border border-blue-200 rounded-lg p-2 mb-3">
                    <div className="text-xs font-medium text-blue-900 mb-1">Reasoning:</div>
                    <div className="text-xs text-blue-800">{change.reason}</div>
                    <div className="flex items-center justify-between mt-2">
                      <span className="text-xs text-blue-600">Profit Impact:</span>
                      <span className="text-xs font-bold text-green-600">{change.profitImpact}</span>
                    </div>
                  </div>

                  <div className="grid grid-cols-2 gap-2">
                    <button
                      onClick={() => handleApproveAIChange(change.id)}
                      className="bg-green-600 text-white py-2 rounded-lg text-sm font-medium flex items-center justify-center space-x-1"
                    >
                      <CheckCircle className="w-4 h-4" />
                      <span>Approve</span>
                    </button>
                    <button
                      onClick={() => setPendingAIChanges(pendingAIChanges.filter(c => c.id !== change.id))}
                      className="bg-gray-200 text-gray-700 py-2 rounded-lg text-sm font-medium"
                    >
                      Reject
                    </button>
                  </div>
                </div>
              ))}

              {pendingAIChanges.length === 0 && (
                <div className="text-center py-8">
                  <CheckCircle className="w-12 h-12 text-green-500 mx-auto mb-3" />
                  <p className="text-gray-600 font-medium">All caught up!</p>
                  <p className="text-sm text-gray-500">No pending AI recommendations</p>
                </div>
              )}

              <button className="w-full bg-gradient-to-r from-purple-600 to-blue-600 text-white py-3 rounded-lg font-medium flex items-center justify-center space-x-2">
                <Play className="w-4 h-4" />
                <span>Generate AI Schedule</span>
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Employee List Drawer */}
      {showEmployeeList && (
        <div className="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-end">
          <div className="bg-white rounded-t-2xl w-full max-h-[80vh] overflow-y-auto">
            <div className="sticky top-0 bg-white border-b border-gray-200 p-4 rounded-t-2xl">
              <div className="flex items-center justify-between">
                <h2 className="text-lg font-bold text-gray-900">Team</h2>
                <button onClick={() => setShowEmployeeList(false)} className="p-2 hover:bg-gray-100 rounded-lg">
                  <X className="w-6 h-6" />
                </button>
              </div>
            </div>

            <div className="p-4 space-y-2">
              {employees.map(employee => (
                <div key={employee.id} className="border border-gray-200 rounded-lg p-3">
                  <div className="flex items-center space-x-3 mb-2">
                    <div 
                      className="w-12 h-12 rounded-full flex items-center justify-center text-white font-bold"
                      style={{ backgroundColor: employee.color }}
                    >
                      {employee.name.split(' ').map(n => n[0]).join('')}
                    </div>
                    <div className="flex-1">
                      <div className="font-bold text-gray-900">{employee.name}</div>
                      <div className="text-sm text-gray-600">{employee.position}</div>
                    </div>
                    <div className="text-right">
                      <div className="text-lg font-bold text-green-600">{employee.score}</div>
                      <div className="text-xs text-gray-500">${employee.hourlyRate}/hr</div>
                    </div>
                  </div>
                  <div className="flex flex-wrap gap-1">
                    {employee.certifications.map(cert => (
                      <span key={cert} className="text-xs bg-gray-100 px-2 py-1 rounded">
                        {cert}
                      </span>
                    ))}
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}

      {/* Notifications Drawer */}
      {showNotifications && (
        <div className="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-end">
          <div className="bg-white rounded-t-2xl w-full max-h-[80vh] overflow-y-auto">
            <div className="sticky top-0 bg-white border-b border-gray-200 p-4 rounded-t-2xl">
              <div className="flex items-center justify-between">
                <h2 className="text-lg font-bold text-gray-900">Notifications</h2>
                <button onClick={() => setShowNotifications(false)} className="p-2 hover:bg-gray-100 rounded-lg">
                  <X className="w-6 h-6" />
                </button>
              </div>
            </div>

            <div className="divide-y divide-gray-100">
              {notifications.map(notif => (
                <div key={notif.id} className={`p-4 ${!notif.read ? 'bg-blue-50' : 'bg-white'}`}>
                  <div className="flex items-start space-x-3">
                    <div className={`p-2 rounded-full ${
                      notif.type === 'shift_change' ? 'bg-blue-100' :
                      notif.type === 'ai_suggestion' ? 'bg-purple-100' :
                      'bg-green-100'
                    }`}>
                      {notif.type === 'shift_change' ? <Calendar className="w-4 h-4 text-blue-600" /> :
                       notif.type === 'ai_suggestion' ? <Bot className="w-4 h-4 text-purple-600" /> :
                       <CheckCircle className="w-4 h-4 text-green-600" />}
                    </div>
                    <div className="flex-1">
                      <p className="text-sm text-gray-900 font-medium">{notif.message}</p>
                      <p className="text-xs text-gray-500 mt-1">{notif.time}</p>
                    </div>
                    {!notif.read && (
                      <div className="w-2 h-2 bg-blue-600 rounded-full"></div>
                    )}
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default MobileSchedulingSystem;
