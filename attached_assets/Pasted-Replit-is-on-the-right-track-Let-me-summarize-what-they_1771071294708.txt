const globalSocket = {
  socket: null,
  subscribers: new Map(), // roomId -> callback
  
  connect() {
    if (this.socket?.readyState === WebSocket.OPEN) return;
    
    this.socket = new WebSocket(wsUrl);
    
    // SINGLE onopen handler
    this.socket.onopen = () => {
      console.log('[ChatSocket] Connected');
      // Process any pending operations
      this.processPendingQueue();
    };
    
    // SINGLE onmessage handler that routes to subscribers
    this.socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      const handler = this.subscribers.get(data.roomId);
      if (handler) handler(data);
    };
  },
  
  joinRoom(roomId, onMessage) {
    this.subscribers.set(roomId, onMessage);
    
    if (this.socket?.readyState === WebSocket.OPEN) {
      this.socket.send(JSON.stringify({ type: 'join_conversation', conversationId: roomId }));
    } else {
      this.pendingJoins.push(roomId);
    }
  }
};
