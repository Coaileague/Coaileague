    <script>
        class GeminiAgent {
            constructor(containerId, canvasId) {
                this.container = document.getElementById(containerId);
                this.canvas = document.getElementById(canvasId);
                this.ctx = this.canvas.getContext('2d', { alpha: false });
                
                this.config = {
                    colors: {
                        IDLE: '#38bdf8',      // Sky Blue
                        SEARCHING: '#10b981', // Emerald
                        THINKING: '#a855f7',  // Purple
                        ANALYZING: '#6366f1', // Indigo
                        CODING: '#34d399',    // Green
                        LISTENING: '#fbbf24', // Amber
                        UPLOADING: '#06b6d4', // Cyan
                        SUCCESS: '#f472b6',   // Pink
                        ERROR: '#ef4444'      // Red
                    }
                };

                this.state = {
                    mode: 'IDLE',
                    w: 0, h: 0,
                    scale: 1, 
                    time: 0,
                    touchX: null, touchY: null, isTouching: false,
                    shake: 0,
                    mutation: 0
                };

                // TWINS - With current color properties for lerping
                this.twins = [
                    { id: 0, x: 0, y: 0, trail: [], color: '#38bdf8', targetColor: '#38bdf8', angle: 0 }, 
                    { id: 1, x: 0, y: 0, trail: [], color: '#a855f7', targetColor: '#a855f7', angle: Math.PI } 
                ];
                
                this.particles = [];
                this.shockwaves = [];
                this.init();
                this.setupTouch();
            }

            setupTouch() {
                const handleTouch = (e) => {
                    const rect = this.canvas.getBoundingClientRect();
                    const touch = e.touches ? e.touches[0] : e;
                    this.state.touchX = (touch.clientX - rect.left) * (this.canvas.width / rect.width);
                    this.state.touchY = (touch.clientY - rect.top) * (this.canvas.height / rect.height);
                    this.state.isTouching = true;
                };
                const endTouch = () => { this.state.isTouching = false; };

                this.canvas.addEventListener('touchstart', handleTouch, {passive: true});
                this.canvas.addEventListener('touchmove', handleTouch, {passive: true});
                this.canvas.addEventListener('touchend', endTouch);
                this.canvas.addEventListener('mousedown', handleTouch);
                this.canvas.addEventListener('mousemove', (e) => { if (e.buttons === 1) handleTouch(e); });
                this.canvas.addEventListener('mouseup', endTouch);
            }

            init() {
                this.resize();
                const observer = new ResizeObserver(() => this.resize());
                observer.observe(this.container);
                this.setMode('IDLE');
                this.animate();
            }

            resize() {
                const rect = this.container.getBoundingClientRect();
                this.state.w = rect.width;
                this.state.h = rect.height;
                const dpr = Math.min(window.devicePixelRatio, 2);
                this.canvas.width = this.state.w * dpr;
                this.canvas.height = this.state.h * dpr;
                this.ctx.scale(dpr, dpr);
                this.state.scale = this.state.w; 
            }

            setMode(mode) {
                // 1. DECONSTRUCT EFFECT (Explode old state)
                // Spawn particles at current twin positions colored like the OLD mode
                this.twins.forEach(t => {
                    this.spawnExplosionAt(t.x, t.y, t.color, 8); 
                });

                this.state.mode = mode;
                const color = this.config.colors[mode];
                
                // 2. TRIGGER MUTATION
                this.state.mutation = 1.0; 
                this.spawnShockwave(color);
                this.triggerFlash();
                
                // UI Updates
                const badge = document.getElementById('statusBadge');
                badge.style.color = color;
                badge.style.borderColor = `rgba(${this.hexToRgb(color)}, 0.3)`;
                badge.style.background = `rgba(${this.hexToRgb(color)}, 0.1)`;
                document.getElementById('statusText').innerText = `CO-AI ${mode}`;
                
                const select = document.getElementById('modeSelect');
                select.style.borderColor = color;
                select.style.boxShadow = `0 0 0 2px rgba(${this.hexToRgb(color)}, 0.2)`;

                // Special Case Cleanups
                if (mode === 'SUCCESS') this.spawnExplosionAt(0, 0, color, 30);
                if (mode === 'ERROR') this.state.shake = 25;
                if (mode === 'CODING' || mode === 'UPLOADING') this.particles = [];
                
                // 3. TARGET COLOR SETTING (For smooth lerp)
                if(mode === 'IDLE') { 
                    this.twins[0].targetColor = '#38bdf8'; 
                    this.twins[1].targetColor = '#a855f7'; 
                } else if(mode === 'ERROR') { 
                    this.twins[0].targetColor = '#ef4444'; 
                    this.twins[1].targetColor = '#ef4444'; 
                } else { 
                    this.twins[0].targetColor = color; 
                    this.twins[1].targetColor = '#ffffff'; 
                }
            }

            triggerFlash() {
                const flash = document.getElementById('mutationFlash');
                flash.classList.remove('mutation-active');
                void flash.offsetWidth; 
                flash.classList.add('mutation-active');
            }

            // Linear Interpolation for Colors
            lerpColor(a, b, amount) { 
                const ar = parseInt(a.substring(1).substring(0,2), 16), ag = parseInt(a.substring(1).substring(2,4), 16), ab = parseInt(a.substring(1).substring(4,6), 16);
                const br = parseInt(b.substring(1).substring(0,2), 16), bg = parseInt(b.substring(1).substring(2,4), 16), bb = parseInt(b.substring(1).substring(4,6), 16);
                const rr = Math.round(ar + amount * (br - ar)), rg = Math.round(ag + amount * (bg - ag)), rb = Math.round(ab + amount * (bb - ab));
                return '#' + ((1 << 24) + (rr << 16) + (rg << 8) + rb).toString(16).slice(1);
            }

            update() {
                this.state.time += 1;
                const s = this.state.scale;
                const cx = this.state.w / 2;
                const cy = this.state.h / 2;
                
                // Physics Decay
                if(this.state.shake > 0) { this.state.shake *= 0.9; if(this.state.shake < 0.5) this.state.shake = 0; }
                if(this.state.mutation > 0) { this.state.mutation *= 0.94; if(this.state.mutation < 0.01) this.state.mutation = 0; }

                this.twins.forEach((t, i) => {
                    // COLOR MORPHING
                    t.color = this.lerpColor(t.color, t.targetColor, 0.05);

                    let tx = 0, ty = 0;

                    // --- MUTATION JITTER (The "Rewriting" Phase) ---
                    // Before settling into the new pattern, they shake and scatter
                    let mutationOffsetX = (Math.random() - 0.5) * 50 * this.state.mutation;
                    let mutationOffsetY = (Math.random() - 0.5) * 50 * this.state.mutation;

                    if (this.state.isTouching) {
                        const mx = (this.state.touchX / window.devicePixelRatio) - cx;
                        const my = (this.state.touchY / window.devicePixelRatio) - cy;
                        tx = mx + Math.cos(this.state.time * 0.1 + i*Math.PI) * (30*s*0.002);
                        ty = my + Math.sin(this.state.time * 0.1 + i*Math.PI) * (30*s*0.002);
                    } 
                    else if (this.state.mode === 'IDLE') {
                        // COMPLEX SPIROGRAPH FOR IDLE
                        // Smoother, deeper breathing motion for "Cool Down"
                        const t1 = this.state.time * 0.01;
                        const t2 = this.state.time * 0.03;
                        // Twin 1 and 2 follow different harmonics
                        if (i === 0) {
                            tx = Math.cos(t1) * (60*s*0.002) + Math.sin(t2) * (20*s*0.002);
                            ty = Math.sin(t1) * (40*s*0.002) + Math.cos(t2) * (20*s*0.002);
                        } else {
                            tx = Math.sin(t1) * (60*s*0.002) - Math.cos(t2) * (20*s*0.002);
                            ty = Math.cos(t1) * (40*s*0.002) - Math.sin(t2) * (20*s*0.002);
                        }
                    }
                    else if (this.state.mode === 'SEARCHING') {
                        if (i === 0) { tx = 0; ty = 0; } 
                        else {
                            const angle = this.state.time * 0.05;
                            const rad = 100 * s * 0.002;
                            tx = Math.cos(angle) * rad; ty = Math.sin(angle) * rad;
                            if (this.state.time % 40 === 0) this.spawnParticle(tx, ty, t.color);
                        }
                    }
                    else if (this.state.mode === 'ANALYZING') {
                        const angle = i === 0 ? -Math.PI/4 : Math.PI*0.75;
                        const dist = 60 * s * 0.002;
                        tx = Math.cos(angle) * dist; ty = Math.sin(angle) * dist;
                        tx += Math.sin(this.state.time * 0.1 + i) * 5;
                    }
                    else if (this.state.mode === 'THINKING') {
                        t.angle += 0.2; // Faster for performance look
                        const radius = (50 + Math.sin(this.state.time * 0.2)*10) * s * 0.002;
                        tx = Math.cos(t.angle + (i*Math.PI)) * radius; ty = Math.sin(t.angle + (i*Math.PI)) * radius;
                    } 
                    else if (this.state.mode === 'CODING') {
                        const step = 40 * s * 0.002;
                        const speed = this.state.time * 0.08 + (i * 10);
                        // Hard snapping movement for digital feel
                        tx = Math.round(Math.cos(speed) * 3.5) * step; 
                        ty = Math.round(Math.sin(speed) * 3.5) * step;
                    }
                    else if (this.state.mode === 'UPLOADING') {
                        const angle = this.state.time * 0.3 + (i * Math.PI);
                        const radius = 30 * s * 0.002;
                        tx = Math.cos(angle) * radius; 
                        ty = (Math.sin(this.state.time * 0.05) * 60) * s * 0.002;
                        if (this.state.time % 4 === 0) this.spawnParticle(tx, ty, t.color, 0, 3);
                    } 
                    else if (this.state.mode === 'LISTENING') {
                        const audio = Math.sin(this.state.time * 0.2 + i) * Math.sin(this.state.time * 0.5);
                        tx = (i === 0 ? -30 : 30) * s * 0.002; ty = audio * 70 * s * 0.002;
                    }
                    else if (this.state.mode === 'SUCCESS') { tx = 0; ty = 0; }
                    else if (this.state.mode === 'ERROR') { tx = (Math.random()-0.5) * 20; ty = (Math.random()-0.5) * 20; }

                    // Apply Mutation Scatter
                    tx += mutationOffsetX;
                    ty += mutationOffsetY;

                    // Physics Lerp
                    const lerpSpeed = this.state.mutation > 0 ? 0.05 : 0.12; // Slower during mutation for floaty feel
                    t.x += (tx - t.x) * lerpSpeed;
                    t.y += (ty - t.y) * lerpSpeed;

                    t.trail.push({x: t.x, y: t.y, life: 1.0});
                    if (t.trail.length > 20) t.trail.shift();
                    t.trail.forEach(p => p.life -= 0.05);
                });
            }

            draw() {
                const { w, h, scale: s } = this.state;
                const cx = w / 2;
                const cy = h / 2;

                this.ctx.save();
                this.ctx.fillStyle = '#020617';
                this.ctx.fillRect(0, 0, w, h);
                
                // RGB SHIFT (Chromatic Aberration) based on mutation
                if (this.state.mutation > 0.01) {
                    this.ctx.globalCompositeOperation = 'screen';
                    const shift = this.state.mutation * 8;
                    
                    this.ctx.save();
                    this.ctx.translate(cx - shift, cy);
                    this.ctx.fillStyle = 'rgba(255, 0, 0, 0.5)';
                    this.drawContent(w, h, s, true); // Red Channel
                    this.ctx.restore();

                    this.ctx.save();
                    this.ctx.translate(cx + shift, cy);
                    this.ctx.fillStyle = 'rgba(0, 0, 255, 0.5)';
                    this.drawContent(w, h, s, true); // Blue Channel
                    this.ctx.restore();
                    
                    this.ctx.globalCompositeOperation = 'source-over';
                }

                // Main Draw
                let shakeX = 0, shakeY = 0;
                if(this.state.shake > 0) {
                    shakeX = (Math.random()-0.5)*this.state.shake;
                    shakeY = (Math.random()-0.5)*this.state.shake;
                }
                this.ctx.translate(cx + shakeX, cy + shakeY);
                this.drawContent(w, h, s, false);

                // Digital Glitch Lines during mutation
                if (this.state.mutation > 0.1) {
                    this.ctx.fillStyle = `rgba(255, 255, 255, ${this.state.mutation * 0.1})`;
                    const glitchY = (Math.random() - 0.5) * h;
                    const glitchH = Math.random() * 20;
                    this.ctx.fillRect(-w/2, glitchY, w, glitchH);
                }

                this.ctx.restore();
            }

            drawContent(w, h, s, isGhost) {
                if(!isGhost) this.drawGrid(w, h, s);
                this.drawShockwaves();
                this.drawParticles();
                
                this.twins.forEach(t => {
                    // Trail
                    this.ctx.beginPath();
                    for(let i=0; i<t.trail.length-1; i++) {
                        const p1 = t.trail[i];
                        const p2 = t.trail[i+1];
                        this.ctx.strokeStyle = isGhost ? this.ctx.fillStyle : t.color;
                        this.ctx.globalAlpha = p1.life * 0.5;
                        this.ctx.lineWidth = (p1.life * 10 * s * 0.002);
                        this.ctx.moveTo(p1.x, p1.y);
                        this.ctx.lineTo(p2.x, p2.y);
                        this.ctx.stroke();
                    }
                    this.ctx.globalAlpha = 1.0;
                    
                    // Star Shape
                    // Scale pulses with mutation
                    const pulse = 1 + (this.state.mutation * 0.5 * Math.sin(this.state.time));
                    this.drawStar(t.x, t.y, 15 * s * 0.002 * pulse, 4 * s * 0.002, isGhost ? this.ctx.fillStyle : t.color);
                });

                // Analysis Lines
                if(this.state.mode === 'ANALYZING') {
                    this.ctx.strokeStyle = isGhost ? this.ctx.fillStyle : this.twins[0].color;
                    this.ctx.lineWidth = 1;
                    this.ctx.globalAlpha = 0.3;
                    const pts = 6;
                    for(let i=0; i<pts; i++) {
                        const a = (Math.PI*2/pts)*i + this.state.time*0.01;
                        const r = 100 * s * 0.002;
                        this.ctx.beginPath(); this.ctx.moveTo(this.twins[0].x, this.twins[0].y);
                        this.ctx.lineTo(Math.cos(a)*r, Math.sin(a)*r); this.ctx.stroke();
                        this.ctx.beginPath(); this.ctx.moveTo(this.twins[1].x, this.twins[1].y);
                        this.ctx.lineTo(Math.cos(a)*r, Math.sin(a)*r); this.ctx.stroke();
                    }
                    this.ctx.globalAlpha = 1.0;
                }

                // Connector
                if(this.state.mode !== 'ERROR' && this.state.mode !== 'SEARCHING') {
                    const t1 = this.twins[0]; const t2 = this.twins[1];
                    const dist = Math.hypot(t1.x - t2.x, t1.y - t2.y);
                    if(dist < 150 * s * 0.002) {
                        this.ctx.strokeStyle = isGhost ? this.ctx.fillStyle : `rgba(255,255,255,${1 - dist/(150*s*0.002)})`;
                        this.ctx.lineWidth = 1;
                        this.ctx.beginPath(); this.ctx.moveTo(t1.x, t1.y); this.ctx.lineTo(t2.x, t2.y); this.ctx.stroke();
                    }
                }
            }

            drawStar(x, y, outerR, innerR, color) {
                this.ctx.fillStyle = color;
                if(!this.ctx.fillStyle.includes('rgba')) { // Don't shadow if ghost render
                    this.ctx.shadowBlur = 20; this.ctx.shadowColor = color;
                }
                this.ctx.beginPath();
                for(let i=0; i<4; i++) {
                    const angle = (i * Math.PI / 2) - (this.state.time * 0.05); 
                    this.ctx.lineTo(x + Math.cos(angle) * outerR, y + Math.sin(angle) * outerR);
                    this.ctx.lineTo(x + Math.cos(angle + Math.PI/4) * innerR, y + Math.sin(angle + Math.PI/4) * innerR);
                }
                this.ctx.closePath(); this.ctx.fill();
                this.ctx.shadowBlur = 0;
                this.ctx.fillStyle = '#fff';
                this.ctx.beginPath(); this.ctx.arc(x, y, innerR, 0, Math.PI*2); this.ctx.fill();
            }

            drawGrid(w, h, s) {
                this.ctx.strokeStyle = 'rgba(255,255,255,0.03)';
                this.ctx.lineWidth = 1;
                const gridSize = 50 * s * 0.002;
                const offsetY = (this.state.time * 0.5) % gridSize;
                const countX = Math.ceil(w / 2 / gridSize) + 1;
                const countY = Math.ceil(h / 2 / gridSize) + 1;
                this.ctx.beginPath();
                for(let i = -countX; i <= countX; i++) { this.ctx.moveTo(i * gridSize, -h/2); this.ctx.lineTo(i * gridSize, h/2); }
                for(let i = -countY; i <= countY; i++) { this.ctx.moveTo(-w/2, i * gridSize + offsetY); this.ctx.lineTo(w/2, i * gridSize + offsetY); }
                this.ctx.stroke();
            }

            spawnShockwave(color) { this.shockwaves.push({ r: 0, opacity: 1, color: color }); }

            drawShockwaves() {
                for(let i=this.shockwaves.length-1; i>=0; i--) {
                    let sw = this.shockwaves[i];
                    sw.r += 8; // Faster ripple
                    sw.opacity -= 0.04;
                    if(sw.opacity <= 0) { this.shockwaves.splice(i, 1); continue; }
                    this.ctx.strokeStyle = sw.color; this.ctx.globalAlpha = sw.opacity;
                    this.ctx.lineWidth = 3; this.ctx.beginPath(); this.ctx.arc(0, 0, sw.r, 0, Math.PI*2); this.ctx.stroke(); this.ctx.globalAlpha = 1.0;
                }
            }

            spawnExplosionAt(x, y, color, count) {
                for(let i=0; i<count; i++) {
                    const a = Math.random() * Math.PI * 2; const v = 2 + Math.random() * 3;
                    this.particles.push({ x: x, y: y, vx: Math.cos(a)*v, vy: Math.sin(a)*v, life: 1.0, color: color });
                }
            }

            spawnExplosion(count) { this.spawnExplosionAt(0, 0, this.config.colors.SUCCESS, count); }

            spawnParticle(x, y, color, vx=0, vy=0) {
                this.particles.push({ x, y, vx: vx || (Math.random()-0.5), vy: vy || (Math.random()-0.5), life: 1.0, color });
            }

            drawParticles() {
                for(let i=this.particles.length-1; i>=0; i--) {
                    let p = this.particles[i];
                    p.x += p.vx; p.y += p.vy; p.life -= 0.03;
                    if(p.life <= 0) { this.particles.splice(i,1); continue; }
                    this.ctx.fillStyle = p.color; this.ctx.globalAlpha = p.life;
                    this.ctx.beginPath(); this.ctx.arc(p.x, p.y, 2, 0, Math.PI*2); this.ctx.fill();
                }
                this.ctx.globalAlpha = 1.0;
            }

            hexToRgb(hex) {
                const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
                return result ? `${parseInt(result[1], 16)}, ${parseInt(result[2], 16)}, ${parseInt(result[3], 16)}` : '255,255,255';
            }

            animate() {
                this.update();
                this.draw();
                this.state.frameId = requestAnimationFrame(() => this.animate());
            }
        }

        const agent = new GeminiAgent('agentWidget', 'agentCanvas');
