The "Trinity Scorecard" UI Logic
This component should live on the Admin Dashboard under a "Recent Automations" or "Trinity Activity" tab.
// TrinityScorecard.jsx (Simplified Logic)
import React from 'react';

const TrinityScorecard = ({ decision }) => {
  const { trigger_event, final_action, candidates_evaluated, metadata } = decision;

  return (
    <div className="border rounded-lg p-6 bg-white shadow-sm">
      <div className="flex justify-between items-center mb-4">
        <h3 className="text-lg font-bold">Trinity Decision: {trigger_event}</h3>
        <span className="px-3 py-1 bg-blue-100 text-blue-700 rounded-full text-sm font-medium">
          {final_action}
        </div>
      </div>

      <div className="mb-6 p-4 bg-gray-50 rounded-md">
        <p className="text-sm text-gray-600">
          <strong>Reasoning:</strong> Trinity evaluated {candidates_evaluated.length} candidates. 
          The selection was optimized for 
          <span className="text-green-600 font-semibold"> Minimum OT Cost</span> and 
          <span className="text-green-600 font-semibold"> Proximity</span>.
        </p>
      </div>

      <table className="w-full text-left text-sm">
        <thead className="border-b">
          <tr>
            <th className="pb-2">Guard</th>
            <th className="pb-2">Match %</th>
            <th className="pb-2">Key Factor</th>
            <th className="pb-2">Result</th>
          </tr>
        </thead>
        <tbody>
          {candidates_evaluated.map((c, i) => (
            <tr key={i} className="border-b last:border-0">
              <td className="py-3 font-medium">{c.guard_name}</td>
              <td className="py-3">{(c.rank_score * 100).toFixed(0)}%</td>
              <td className="py-3 text-gray-500">
                {c.outcome === 'REJECTED' ? c.reason : "Top Qualified"}
              </td>
              <td className="py-3">
                {c.outcome === 'SELECTED' ? 
                  <span className="text-green-500">✔ Assigned</span> : 
                  <span className="text-gray-400">Skipped</span>}
              </td>
            </tr>
          ))}
        </tbody>
      </table>
      
      <div className="mt-4 text-xs text-gray-400">
        Engine: {metadata.model_used} | Latency: {metadata.processing_time_ms}ms
      </div>
    </div>
  );
};
