  const [isRecording, setIsRecording] = useState(false);
  const [transcript, setTranscript] = useState('');
  
  const handlePressIn = () => {
    startRecording();
    hapticFeedback('impact', 'medium');
    setIsRecording(true);
  };
  
  const handlePressOut = async () => {
    const audio = await stopRecording();
    hapticFeedback('impact', 'light');
    
    // Stream transcription
    streamTranscription(audio, (partial) => {
      setTranscript(partial);
    });
    
    // Send to Trinity with context
    const response = await trinity.processVoice({
      audio,
      transcript,
      pageContext: getCurrentContext()
    });
    
    setIsRecording(false);
  };
  
  return (
    <PressableButton
      onPressIn={handlePressIn}
      onPressOut={handlePressOut}
      delayLongPress={0}
    >
      <MicIcon animated={isRecording} />
      {transcript && <LiveTranscript>{transcript}</LiveTranscript>}
    </PressableButton>
  );
};

javascriptconst CollaborativeCursor = ({ trinityFocus }) => {
  const [cursorPos, setCursorPos] = useState({ x: 0, y: 0 });
  
  // Animate Trinity's "gaze"
  useEffect(() => {
    if (trinityFocus.element) {
      const el = document.querySelector(trinityFocus.element);
      const rect = el.getBoundingClientRect();
      
      // Smooth animation to element
      animateCursorTo({
        x: rect.left + rect.width / 2,
        y: rect.top + rect.height / 2
      });
      
      // Pulse highlight
      el.classList.add('trinity-focus');
      setTimeout(() => el.classList.remove('trinity-focus'), 2000);
    }
  }, [trinityFocus]);
  
  return (
    <TrinityPointer
      style={{
        left: cursorPos.x,
        top: cursorPos.y,
        transition: 'all 0.8s cubic-bezier(0.4, 0, 0.2, 1)'
      }}
    >
      ✨ {/* Trinity's cursor */}
    </TrinityPointer>
  );
};
