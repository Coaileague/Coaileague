# COAILEAGUE - SCHEMA, ROUTE & WORKFLOW INTEGRITY AUDIT

## MISSION: SOLID FOUNDATION - ZERO SILENT FAILURES

This audit finds:
1. **Missing tables** - Referenced but don't exist
2. **Missing enums** - Used in code but not defined
3. **Missing subtables** - Related tables needed but not created
4. **Workflow failures** - Silently failing due to missing DB structures
5. **Route mount issues** - Routes not mounted or conflicting
6. **Frontend-backend mismatch** - API calls to non-existent endpoints
7. **Schema-code drift** - Schema doesn't match what code expects

**SILENT FAILURES ARE THE WORST BUGS** - They don't crash, they just don't work.

## 6.3 Workflow Error Handling Fix

```typescript
// BAD - Silent failure
async function processWorkflow() {
  try {
    await step1();
    await step2(); // If this fails, step1 is orphaned
    await step3();
  } catch (error) {
    console.log(error); // Silent failure!
  }
}

// GOOD - Transaction with proper error handling
async function processWorkflow() {
  const result = await db.transaction(async (tx) => {
    const step1Result = await tx.insert(table1).values({...}).returning();
    const step2Result = await tx.insert(table2).values({...}).returning();
    const step3Result = await tx.insert(table3).values({...}).returning();
    return { step1Result, step2Result, step3Result };
  });
  
  // If any step fails, entire transaction rolls back
  return result;
}

// With explicit error handling
async function processWorkflowWithHandling() {
  try {
    return await db.transaction(async (tx) => {
      // steps...
    });
  } catch (error) {
    // Log error
    await logError('workflow_failure', error);
    // Notify admin
    await notifyAdmin('Workflow failed', error);
    // Throw to caller
    throw new WorkflowError('Workflow failed', { cause: error });
  }
}
```

// Create integrity check script: server/scripts/integrity-check.ts

async function checkIntegrity() {
  const issues: string[] = [];
  
  // 1. Check all tables exist
  const tables = await db.execute(sql`
    SELECT table_name FROM information_schema.tables 
    WHERE table_schema = 'public'
  `);
  const expectedTables = ['users', 'workspaces', 'employees', /* ... */];
  for (const table of expectedTables) {
    if (!tables.rows.find(r => r.table_name === table)) {
      issues.push(`Missing table: ${table}`);
    }
  }
  
  // 2. Check foreign key constraints
  const fks = await db.execute(sql`
    SELECT tc.table_name, kcu.column_name, ccu.table_name AS foreign_table
    FROM information_schema.table_constraints AS tc
    JOIN information_schema.key_column_usage AS kcu
      ON tc.constraint_name = kcu.constraint_name
    JOIN information_schema.constraint_column_usage AS ccu
      ON ccu.constraint_name = tc.constraint_name
    WHERE tc.constraint_type = 'FOREIGN KEY'
  `);
  
  // 3. Check for orphan records
  const orphanEmployees = await db.execute(sql`
    SELECT e.id FROM employees e
    LEFT JOIN workspaces w ON e.workspace_id = w.id
    WHERE w.id IS NULL
  `);
  if (orphanEmployees.rows.length > 0) {
    issues.push(`Orphan employees: ${orphanEmployees.rows.length}`);
  }
  
  // 4. Check enum consistency
  const invalidStatuses = await db.execute(sql`
    SELECT id, status FROM employees 
    WHERE status NOT IN ('active', 'inactive', 'pending', 'terminated')
  `);
  if (invalidStatuses.rows.length > 0) {
    issues.push(`Invalid employee statuses: ${invalidStatuses.rows.length}`);
  }
  
  return issues;
}
