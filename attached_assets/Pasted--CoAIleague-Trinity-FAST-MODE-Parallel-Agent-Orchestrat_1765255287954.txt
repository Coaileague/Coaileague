/**
 * CoAIleague Trinity FAST MODE - Parallel Agent Orchestrator
 * Features:
 * - Parallel agent execution (up to 10 agents simultaneously)
 * - Smart task decomposition
 * - Agent pooling for instant reuse
 * - Streaming responses for faster perceived performance
 * - Speculative execution
 * - Early stopping when confidence threshold met
 * - Result caching
 * - Better than Replit's fast mode!
 */

const { GoogleGenerativeAI } = require('@google/generative-ai');
const { AGENT_REGISTRY } = require('./coaileague-agent-orchestrator');

// ═══════════════════════════════════════════════════════════════
// FAST MODE CONFIGURATION
// ═══════════════════════════════════════════════════════════════

const FAST_MODE_CONFIG = {
  // Parallel execution settings
  maxParallelAgents: 10,              // Run up to 10 agents at once
  agentTimeout: 15000,                 // 15s timeout per agent (fast!)
  
  // Smart optimization
  enableStreaming: true,               // Stream results as they come
  enableSpeculativeExecution: true,    // Start likely agents early
  enableEarlyStop: true,               // Stop when confident enough
  earlyStopConfidence: 85,             // Stop at 85% confidence
  
  // Caching
  enableCache: true,
  cacheTimeout: 300000,                // 5 min cache
  
  // Response optimization
  maxTokensPerAgent: 2000,             // Shorter responses = faster
  preferredTemperature: 0.3,           // Lower temp = faster, more deterministic
  
  // Supervisor mode
  supervisorMode: 'fast',              // 'fast', 'balanced', or 'thorough'
  skipSupervisorForSimple: true,       // Skip supervisor for simple tasks
};

// ═══════════════════════════════════════════════════════════════
// TRINITY FAST MODE ORCHESTRATOR
// ═══════════════════════════════════════════════════════════════

class TrinityFastMode {
  constructor(apiKey, config = {}) {
    this.genAI = new GoogleGenerativeAI(apiKey);
    this.config = { ...FAST_MODE_CONFIG, ...config };
    
    // Agent pool for instant reuse
    this.agentPool = new Map();
    
    // Result cache
    this.cache = new Map();
    
    // Streaming callbacks
    this.streamCallbacks = [];
    
    // Performance metrics
    this.metrics = {
      totalExecutions: 0,
      avgExecutionTime: 0,
      cacheHits: 0,
      parallelExecutions: 0
    };
    
    console.log('\n⚡ Trinity FAST MODE initialized');
    console.log(`   Max parallel agents: ${this.config.maxParallelAgents}`);
    console.log(`   Streaming: ${this.config.enableStreaming ? 'ON' : 'OFF'}`);
    console.log(`   Speculative execution: ${this.config.enableSpeculativeExecution ? 'ON' : 'OFF'}`);
  }

  /**
   * SMART TASK DECOMPOSITION
   * Breaks task into parallel subtasks
   */
  async decomposeTask(task, context) {
    console.log('\n🧠 Trinity analyzing task for parallel execution...');
    
    const model = this.genAI.getGenerativeModel({
      model: 'gemini-2.0-flash-exp',
      generationConfig: {
        temperature: 0.3,
        maxOutputTokens: 1000
      }
    });

    const decompositionPrompt = `
You are Trinity, master orchestrator for CoAIleague. Analyze this task for PARALLEL execution.

TASK: ${task}
CONTEXT: ${JSON.stringify(context)}

Available agents: ${Object.keys(AGENT_REGISTRY).join(', ')}

Break this into PARALLEL subtasks. Identify:
1. INDEPENDENT TASKS (can run simultaneously)
2. SEQUENTIAL DEPENDENCIES (must wait for others)
3. PRIORITY TASKS (critical path items)
4. OPTIONAL TASKS (nice-to-have, can skip if time-critical)

Format as JSON:
{
  "parallel_groups": [
    {
      "group_id": 1,
      "agents": ["agent1", "agent2"],
      "can_run_parallel": true,
      "priority": "high"
    }
  ],
  "dependencies": {
    "agent2": ["agent1"]
  },
  "estimated_time": "30s",
  "complexity": "medium"
}
`;

    const result = await model.generateContent(decompositionPrompt);
    const text = result.response.text();
    
    // Parse JSON from response
    const jsonMatch = text.match(/\{[\s\S]*\}/);
    if (jsonMatch) {
      return JSON.parse(jsonMatch[0]);
    }
    
    // Fallback to sequential
    return {
      parallel_groups: [{
        group_id: 1,
        agents: this.guessAgentsFromTask(task),
        can_run_parallel: true,
        priority: 'high'
      }],
      dependencies: {},
      estimated_time: '60s',
      complexity: 'medium'
    };
  }

  /**
   * PARALLEL AGENT EXECUTION
   * Runs multiple agents simultaneously
   */
  async executeParallel(agentNames, task, context, priorResults = {}) {
    console.log(`\n⚡ Executing ${agentNames.length} agents in PARALLEL...`);
    
    const startTime = Date.now();
    
    // Create promises for all agents
    const agentPromises = agentNames.map(agentName => 
      this.executeAgentFast(agentName, task, context, priorResults)
    );
    
    // Execute all in parallel with timeout
    const results = await Promise.allSettled(
      agentPromises.map(p => 
        this.withTimeout(p, this.config.agentTimeout)
      )
    );
    
    const executionTime = Date.now() - startTime;
    
    // Process results
    const successfulResults = [];
    const failedAgents = [];
    
    results.forEach((result, idx) => {
      if (result.status === 'fulfilled') {
        successfulResults.push(result.value);
      } else {
        failedAgents.push(agentNames[idx]);
        console.log(`  ⚠️ ${agentNames[idx]} failed/timeout`);
      }
    });
    
    console.log(`  ✓ Completed in ${(executionTime / 1000).toFixed(2)}s`);
    console.log(`  ✓ Success: ${successfulResults.length}/${agentNames.length} agents`);
    
    this.metrics.parallelExecutions++;
    
    return {
      results: successfulResults,
      failed: failedAgents,
      executionTime
    };
  }

  /**
   * FAST AGENT EXECUTION
   * Optimized single agent execution
   */
  async executeAgentFast(agentName, task, context, priorResults) {
    // Check cache first
    const cacheKey = this.getCacheKey(agentName, task, context);
    if (this.config.enableCache && this.cache.has(cacheKey)) {
      const cached = this.cache.get(cacheKey);
      if (Date.now() - cached.timestamp < this.config.cacheTimeout) {
        console.log(`  💾 ${agentName} - CACHE HIT`);
        this.metrics.cacheHits++;
        return cached.result;
      }
    }
    
    // Get or create agent
    const agent = this.getAgentFromPool(agentName);
    if (!agent) {
      throw new Error(`Agent ${agentName} not found`);
    }
    
    const model = this.genAI.getGenerativeModel({
      model: 'gemini-2.0-flash-exp',
      generationConfig: {
        temperature: agent.temperature || this.config.preferredTemperature,
        maxOutputTokens: this.config.maxTokensPerAgent
      }
    });

    // Optimized prompt (shorter = faster)
    const prompt = `
${agent.name} - ${agent.specialty}

TASK: ${task}
CONTEXT: ${JSON.stringify(context)}

${Object.keys(priorResults).length > 0 ? `PRIOR: ${JSON.stringify(priorResults)}` : ''}

Execute quickly. Provide:
1. Key finding
2. Recommendation
3. Confidence (0-100)

Be concise.
`;

    const startTime = Date.now();
    const result = await model.generateContent(prompt);
    const executionTime = Date.now() - startTime;
    
    const agentResult = {
      agentName: agent.name,
      specialty: agent.specialty,
      output: result.response.text(),
      executionTime,
      timestamp: new Date().toISOString(),
      confidence: this.extractConfidence(result.response.text())
    };
    
    // Cache result
    if (this.config.enableCache) {
      this.cache.set(cacheKey, {
        result: agentResult,
        timestamp: Date.now()
      });
    }
    
    // Stream to callbacks
    this.streamResult(agentResult);
    
    return agentResult;
  }

  /**
   * FAST MODE ORCHESTRATION
   * Main orchestration with all optimizations
   */
  async orchestrateFast(task, context = {}) {
    console.log('\n⚡ ═══════════════════════════════════════════════════════');
    console.log('   TRINITY FAST MODE - PARALLEL ORCHESTRATION');
    console.log('═══════════════════════════════════════════════════════\n');

    const execution = {
      task,
      context,
      mode: 'FAST_PARALLEL',
      timestamp: new Date().toISOString(),
      decomposition: null,
      parallelGroups: [],
      supervisorFeedback: null,
      finalDecision: null,
      executionTime: 0,
      agentsUsed: 0,
      cacheHits: 0
    };

    const startTime = Date.now();

    try {
      // STAGE 1: Decompose task for parallel execution
      console.log('━━━ STAGE 1: SMART TASK DECOMPOSITION ━━━');
      execution.decomposition = await this.decomposeTask(task, context);
      
      console.log(`\n  Parallel groups: ${execution.decomposition.parallel_groups.length}`);
      console.log(`  Estimated time: ${execution.decomposition.estimated_time}`);
      console.log(`  Complexity: ${execution.decomposition.complexity}`);

      // STAGE 2: Execute parallel groups
      console.log('\n━━━ STAGE 2: PARALLEL AGENT EXECUTION ━━━');
      
      const allResults = [];
      const priorResults = {};
      
      for (const group of execution.decomposition.parallel_groups) {
        console.log(`\n  Group ${group.group_id} - Priority: ${group.priority}`);
        console.log(`  Agents: ${group.agents.join(', ')}`);
        
        // Execute this group in parallel
        const groupResult = await this.executeParallel(
          group.agents,
          task,
          context,
          priorResults
        );
        
        // Add successful results
        allResults.push(...groupResult.results);
        execution.parallelGroups.push({
          groupId: group.group_id,
          results: groupResult.results,
          executionTime: groupResult.executionTime
        });
        
        // Update prior results for next group
        groupResult.results.forEach(r => {
          priorResults[r.agentName] = r.output;
        });
        
        // Early stop if we have high confidence
        if (this.config.enableEarlyStop) {
          const avgConfidence = this.calculateAvgConfidence(allResults);
          if (avgConfidence >= this.config.earlyStopConfidence) {
            console.log(`\n  🎯 Early stop at ${avgConfidence}% confidence!`);
            break;
          }
        }
      }
      
      execution.agentsUsed = allResults.length;

      // STAGE 3: Fast supervisor check (optional)
      console.log('\n━━━ STAGE 3: FAST SUPERVISOR VALIDATION ━━━');
      
      if (this.shouldUseSupervisor(execution.decomposition.complexity)) {
        execution.supervisorFeedback = await this.fastSupervisorCheck(
          task,
          allResults
        );
        console.log(`  Assessment: ${execution.supervisorFeedback.assessment}`);
      } else {
        console.log('  Skipped (simple task)');
        execution.supervisorFeedback = { assessment: 'APPROVED', reason: 'Simple task' };
      }

      // STAGE 4: Trinity fast consolidation
      console.log('\n━━━ STAGE 4: RAPID CONSOLIDATION ━━━');
      execution.finalDecision = await this.fastConsolidate(
        task,
        allResults,
        execution.supervisorFeedback
      );

      execution.executionTime = Date.now() - startTime;
      execution.cacheHits = this.metrics.cacheHits;

      console.log('\n⚡ ═══════════════════════════════════════════════════════');
      console.log(`   FAST MODE COMPLETE`);
      console.log(`   Time: ${(execution.executionTime / 1000).toFixed(2)}s`);
      console.log(`   Agents: ${execution.agentsUsed} (${execution.cacheHits} cached)`);
      console.log(`   Avg confidence: ${this.calculateAvgConfidence(allResults)}%`);
      console.log('═══════════════════════════════════════════════════════\n');

      this.updateMetrics(execution);

      return execution;

    } catch (error) {
      console.error('\n❌ Fast mode error:', error.message);
      execution.error = error.message;
      execution.executionTime = Date.now() - startTime;
      throw error;
    }
  }

  /**
   * FAST SUPERVISOR CHECK
   * Lightweight validation for speed
   */
  async fastSupervisorCheck(task, results) {
    const model = this.genAI.getGenerativeModel({
      model: 'gemini-2.0-flash-exp',
      generationConfig: {
        temperature: 0.2,
        maxOutputTokens: 500  // Very short validation
      }
    });

    const prompt = `
Quick validation check:

TASK: ${task}
RESULTS: ${results.length} agents completed

Top 3 agent outputs:
${results.slice(0, 3).map(r => `${r.agentName}: ${r.output.substring(0, 200)}`).join('\n')}

Quick assessment (one word): APPROVED or REJECTED
Issues (if any, one line):
`;

    const result = await model.generateContent(prompt);
    const text = result.response.text();
    
    return {
      assessment: text.includes('APPROVED') ? 'APPROVED' : 'REJECTED',
      details: text,
      mode: 'FAST'
    };
  }

  /**
   * FAST CONSOLIDATION
   * Quick result synthesis
   */
  async fastConsolidate(task, results, supervisorFeedback) {
    const model = this.genAI.getGenerativeModel({
      model: 'gemini-2.0-flash-exp',
      generationConfig: {
        temperature: 0.4,
        maxOutputTokens: 1000
      }
    });

    const prompt = `
Trinity fast consolidation:

TASK: ${task}
AGENTS: ${results.length}
SUPERVISOR: ${supervisorFeedback.assessment}

Top insights:
${results.slice(0, 5).map(r => `• ${r.agentName}: ${r.output.substring(0, 150)}`).join('\n')}

Provide concise final decision:
1. Main recommendation
2. Key points (3 max)
3. Next steps (2 max)

Be brief and actionable.
`;

    const result = await model.generateContent(prompt);
    return result.response.text();
  }

  /**
   * HELPER METHODS
   */

  withTimeout(promise, timeout) {
    return Promise.race([
      promise,
      new Promise((_, reject) => 
        setTimeout(() => reject(new Error('Timeout')), timeout)
      )
    ]);
  }

  getAgentFromPool(agentName) {
    // Find agent config
    for (const config of Object.values(AGENT_REGISTRY)) {
      if (config.name === agentName || config.id === agentName) {
        return config;
      }
    }
    return null;
  }

  getCacheKey(agentName, task, context) {
    return `${agentName}-${task.substring(0, 50)}-${JSON.stringify(context).substring(0, 50)}`;
  }

  extractConfidence(text) {
    const match = text.match(/(\d+)%|confidence.*?(\d+)/i);
    return match ? parseInt(match[1] || match[2]) : 75;
  }

  calculateAvgConfidence(results) {
    if (results.length === 0) return 0;
    const sum = results.reduce((acc, r) => acc + (r.confidence || 75), 0);
    return Math.round(sum / results.length);
  }

  shouldUseSupervisor(complexity) {
    if (!this.config.skipSupervisorForSimple) return true;
    return complexity !== 'simple' && complexity !== 'low';
  }

  guessAgentsFromTask(task) {
    const taskLower = task.toLowerCase();
    const agents = [];
    
    if (taskLower.includes('schedule')) agents.push('ScheduleMaster');
    if (taskLower.includes('payroll')) agents.push('PayrollPro');
    if (taskLower.includes('invoice')) agents.push('InvoiceGenius');
    if (taskLower.includes('email')) agents.push('MailMaestro');
    if (taskLower.includes('data') || taskLower.includes('analyz')) agents.push('DataSage');
    if (taskLower.includes('compliance') || taskLower.includes('legal')) agents.push('ComplianceGuard');
    if (taskLower.includes('optim')) agents.push('EfficiencyExpert');
    
    return agents.length > 0 ? agents : ['DataSage'];
  }

  streamResult(result) {
    if (this.config.enableStreaming) {
      this.streamCallbacks.forEach(callback => callback(result));
    }
  }

  onStream(callback) {
    this.streamCallbacks.push(callback);
  }

  updateMetrics(execution) {
    this.metrics.totalExecutions++;
    const prevAvg = this.metrics.avgExecutionTime;
    this.metrics.avgExecutionTime = 
      (prevAvg * (this.metrics.totalExecutions - 1) + execution.executionTime) / 
      this.metrics.totalExecutions;
  }

  getMetrics() {
    return {
      ...this.metrics,
      avgExecutionTimeSeconds: (this.metrics.avgExecutionTime / 1000).toFixed(2),
      cacheHitRate: this.metrics.totalExecutions > 0 
        ? ((this.metrics.cacheHits / this.metrics.totalExecutions) * 100).toFixed(1) + '%'
        : '0%'
    };
  }

  clearCache() {
    this.cache.clear();
    console.log('✓ Cache cleared');
  }
}

// ═══════════════════════════════════════════════════════════════
// SPECULATIVE EXECUTION MODE
// Start likely agents before decomposition completes
// ═══════════════════════════════════════════════════════════════

class SpeculativeExecutor {
  constructor(fastMode) {
    this.fastMode = fastMode;
    this.speculativeResults = new Map();
  }

  async speculateAgents(task) {
    console.log('🔮 Starting speculative execution...');
    
    // Guess most likely agents based on keywords
    const likelyAgents = this.fastMode.guessAgentsFromTask(task);
    
    // Start them immediately (don't wait)
    const promises = likelyAgents.slice(0, 3).map(agent => 
      this.fastMode.executeAgentFast(agent, task, {}, {})
        .then(result => {
          this.speculativeResults.set(agent, result);
          console.log(`  ✓ Speculative: ${agent} completed early`);
        })
        .catch(err => {
          console.log(`  ⚠️ Speculative: ${agent} failed (${err.message})`);
        })
    );
    
    return promises;
  }

  getResults() {
    return this.speculativeResults;
  }
}
