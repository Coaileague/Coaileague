// BEFORE — matches everything with data-state=open including accordions:

// AFTER — only match actual modals and sheets:
const openElements = document.querySelectorAll(
  'dialog[data-state=open], [role=dialog][data-state=open], [data-modal=true][data-state=open], [data-sheet=true][data-state=open]'
