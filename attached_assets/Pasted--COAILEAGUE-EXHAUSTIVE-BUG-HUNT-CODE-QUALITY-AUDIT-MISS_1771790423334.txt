## 1.1 Dead Imports & Unused Variables

```bash
# Find ALL imports and check if they're used
# Dead imports cause bundle bloat and indicate incomplete code

# Step 1: List all imports
grep -rn "^import\|^import {" --include="*.tsx" --include="*.ts" client/src/ server/

# Step 2: For each import, check if it's used in the file
# This script finds potentially dead imports:
for file in $(find client/src server -name "*.tsx" -o -name "*.ts" 2>/dev/null); do
  # Extract imports
  imports=$(grep -oP "import \{ \K[^}]+" "$file" 2>/dev/null | tr ',' '\n' | tr -d ' ')
  for imp in $imports; do
    # Count usage (excluding the import line itself)
    count=$(grep -c "\b$imp\b" "$file" 2>/dev/null)
    if [ "$count" -le 1 ]; then
      echo "DEAD IMPORT: $imp in $file"
    fi
  done
done

# Alternative: Use TypeScript compiler
npx tsc --noEmit 2>&1 | grep -i "declared but"
```

**500 ERROR CHECKLIST:**

```
TEMPLATE FOR PROPER ERROR HANDLING:

router.get('/api/something/:id', async (req, res) => {
  try {
    // Validate input
    if (!req.params.id) {
      return res.status(400).json({ success: false, error: 'ID required' });
    }
    
    // Check auth
    if (!req.user) {
      return res.status(401).json({ success: false, error: 'Unauthorized' });
    }
    
    // Database operation with workspace isolation
    const result = await db.query.something.findFirst({
      where: and(
        eq(something.id, req.params.id),
        eq(something.workspaceId, req.user.workspaceId) // CRITICAL
      )
    });
    
    if (!result) {
      return res.status(404).json({ success: false, error: 'Not found' });
    }
    
    return res.json({ success: true, data: result });
    
  } catch (error) {
    console.error('Error in /api/something/:id:', error);
    return res.status(500).json({ 
      success: false, 
      error: 'Internal server error',
      // Don't expose error details in production
      ...(process.env.NODE_ENV === 'development' && { details: error.message })
    });
  }
});
```

```bash
  dir=$(dirname $file)
  import_path=$(echo $line | grep -oP "from '[^']+'" | tr -d "'" | sed 's/from //')

  fi
done
```
