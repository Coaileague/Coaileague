**The Difference Between a PDF and an Online Form**

```
PDF Document:
→ Static file — like a printout
→ Can be viewed and downloaded
→ Cannot be filled in or signed 
   without special software
→ Trinity can generate these fine
→ Problem: officer receives PDF, 
   prints it, signs it, scans it back
   That is 2005 workflow

Online Form (what you actually need):
→ Dynamic web page with fillable fields
→ Can be completed on a phone
→ Electronic signature built in
→ Submitted directly to Trinity
→ Trinity receives data structured
→ Automatically creates the PDF record
   after submission
→ That is the correct modern workflow
```

// client/src/components/forms/SignatureCapture.tsx
// Uses react-signature-canvas or similar
// Three modes:
// 1. Draw with mouse or finger (touchscreen)
// 2. Type name in cursive font
// 3. Upload image of signature

import SignatureCanvas from 'react-signature-canvas';

export function SignatureCapture({ onCapture, label = 'Sign here' }) {
  const sigPad = useRef(null);
  const [mode, setMode] = useState<'draw' | 'type' | 'upload'>('draw');
  const [typedName, setTypedName] = useState('');
  
  const capture = () => {
    if (mode === 'draw') {
      const dataUrl = sigPad.current.toDataURL('image/png');
      onCapture({ type: 'drawn', data: dataUrl, timestamp: new Date().toISOString() });
    }
    if (mode === 'type') {
      // Render typedName in cursive font to canvas
      onCapture({ type: 'typed', data: typedName, timestamp: new Date().toISOString() });
    }
  };
  
  return (
    <div className="border rounded-lg p-4">
      <label className="text-sm font-medium mb-2 block">{label}</label>
      
      {/* Mode selector */}
      <div className="flex gap-2 mb-3">
        <button onClick={() => setMode('draw')} 
          className={mode === 'draw' ? 'btn-active' : 'btn'}>Draw</button>
        <button onClick={() => setMode('type')}
          className={mode === 'type' ? 'btn-active' : 'btn'}>Type Name</button>
      </div>
      
      {mode === 'draw' && (
        <SignatureCanvas 
          ref={sigPad}
          canvasProps={{ className: 'border rounded w-full h-32 bg-white' }}
        />
      )}
      
      {mode === 'type' && (
        <input 
          type="text"
          value={typedName}
          onChange={e => setTypedName(e.target.value)}
          placeholder="Type your full legal name"
          className="w-full border rounded p-2 font-cursive text-2xl"
          style={{ fontFamily: 'Dancing Script, cursive' }}
        />
      )}
      
      <div className="flex gap-2 mt-2">
        <button onClick={() => sigPad.current?.clear()}>Clear</button>
        <button onClick={capture} className="btn-primary">Confirm Signature</button>
      </div>
      
      <p className="text-xs text-gray-500 mt-2">
        By signing you agree this is your legal electronic signature 
        per the ESIGN Act and UETA.
      </p>
    </div>
  );
}

**SUBAGENT 3 — Form-to-PDF Pipeline:**

```typescript
// server/services/formToPdfService.ts

export class FormToPdfService {
  
  // Called after every form submission
  async generateSignedDocument(
    submission: FormSubmission,
    form: PlatformForm,
    invitation: FormInvitation
  ): Promise<string> {
    
    // 1. Merge submission data with form template
    const documentData = this.mergeFormData(form, submission.data);
    
    // 2. Generate PDF based on form_type
    let pdfBuffer: Buffer;
    
    switch(form.form_type) {
      case 'offer_letter':
        pdfBuffer = await this.generateOfferLetterPdf(documentData, invitation.workspace_id);
        break;
      case 'onboarding_w4':
        pdfBuffer = await this.generateW4Pdf(documentData);
        break;
      case 'onboarding_i9_section1':
        pdfBuffer = await this.generateI9Pdf(documentData);
        break;
      case 'client_service_agreement':
        pdfBuffer = await this.generateServiceAgreementPdf(documentData, invitation.workspace_id);
        break;
      case 'sales_proposal_accepted':
        pdfBuffer = await this.generateContractPdf(documentData, invitation.workspace_id);
        break;
      default:
        pdfBuffer = await this.generateGenericFormPdf(documentData, form.title);
    }
    
    // 3. Embed signature into PDF
    if (submission.data.signature) {
      pdfBuffer = await this.embedSignature(pdfBuffer, submission.data.signature);
    }
    
    // 4. Add legal footer to every page
    pdfBuffer = await this.addLegalFooter(pdfBuffer, {
      signedAt: submission.submitted_at,
      ipAddress: submission.ip_address,
      signerEmail: submission.submitted_by_email,
      documentNumber: '[to be assigned]'
    });
    
    // 5. Calculate SHA-256 hash
    const hash = crypto.createHash('sha256').update(pdfBuffer).digest('hex');
    
    // 6. Store in document vault
    const stored = await documentVaultService.store({
      buffer: pdfBuffer,
      workspaceId: invitation.workspace_id,
      employeeId: invitation.context_type === 'employee' ? invitation.context_id : null,
      clientId: invitation.context_type === 'client' ? invitation.context_id : null,
      documentType: form.form_type,
      integrityHash: hash,
      metadata: {
        formId: form.id,
        submissionId: submission.id,
        signedAt: submission.submitted_at
      }
    });
    
    return stored.url;
  }
}
```

// Multi-party signing orchestrator
async function advanceSigningSequence(sequenceId: string) {
  const sequence = await getSigningSequence(sequenceId);
  const signers = sequence.signers;
  const currentSigner = signers[sequence.current_signer_index];
  
  if (currentSigner.status !== 'signed') return; // Not ready to advance
  
  const nextIndex = sequence.current_signer_index + 1;
  
  if (nextIndex >= signers.length) {
    // All parties signed — document complete
    await markDocumentFullySigned(sequence.document_id);
    await notifyAllParties(sequence, 'document_fully_executed');
    return;
  }
  
  // Activate next signer
  const nextSigner = signers[nextIndex];
  const signingToken = generateSecureToken();
  signers[nextIndex].status = 'pending';
  signers[nextIndex].token = signingToken;
  
  await updateSigningSequence(sequenceId, nextIndex, signers);
  
  // Send signing request to next party
  await notificationDeliveryService.send({
    workspaceId: sequence.workspace_id,
    type: 'document_awaiting_your_signature',
    toEmail: nextSigner.email || null,
    toUserId: nextSigner.userId || null,
    data: {
      signingUrl: `${BASE_URL}/sign/${signingToken}`,
      documentName: sequence.document_title,
      previousSignerRole: currentSigner.role
    }
  });
}
