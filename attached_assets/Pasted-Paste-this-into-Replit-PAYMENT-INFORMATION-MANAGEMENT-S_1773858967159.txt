```typescript
import crypto from 'crypto'

const ENCRYPTION_KEY = process.env.PAYMENT_ENCRYPTION_KEY  // 32 bytes hex
const ALGORITHM = 'aes-256-gcm'

export function encryptPaymentData(plaintext: string): string {
  // AES-256-GCM with random IV per encryption
  // Returns: iv:authTag:encryptedData (all hex)
  // Never store plaintext routing numbers, account numbers, or card data
  const iv = crypto.randomBytes(16)
  const cipher = crypto.createCipheriv(ALGORITHM, Buffer.from(ENCRYPTION_KEY, 'hex'), iv)
  const encrypted = Buffer.concat([cipher.update(plaintext, 'utf8'), cipher.final()])
  const authTag = cipher.getAuthTag()
  return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted.toString('hex')}`
}

export function decryptPaymentData(encryptedString: string): string {
  const [ivHex, authTagHex, encryptedHex] = encryptedString.split(':')
  const iv = Buffer.from(ivHex, 'hex')
  const authTag = Buffer.from(authTagHex, 'hex')
  const encrypted = Buffer.from(encryptedHex, 'hex')
  const decipher = crypto.createDecipheriv(ALGORITHM, Buffer.from(ENCRYPTION_KEY, 'hex'), iv)
  decipher.setAuthTag(authTag)
  return decipher.update(encrypted) + decipher.final('utf8')
}

export function maskAccountNumber(account: string): string {
  return '****' + account.slice(-4)
}

export function maskRoutingNumber(routing: string): string {
  return '****' + routing.slice(-4)
}

// Hash for deduplication checks — never reversible
export function hashPaymentIdentifier(value: string): string {
  return crypto.createHmac('sha256', process.env.PAYMENT_HASH_SECRET)
    .update(value.toLowerCase().trim())
    .digest('hex')
}
```

**SECTION 2 — RATE LIMITING ENFORCEMENT SERVICE**

Create `server/services/payments/paymentChangeLimiter.ts`:

```typescript
export interface RateLimitConfig {
  entityType: 'employee_bank' | 'org_bank' | 'subscription'
  monthlyLimit: number      // standard monthly limit
  ownerOverrideLimit: number  // how many times owner can override
  supportOverrideUnlimited: boolean
}

export const RATE_LIMIT_CONFIGS: Record<string, RateLimitConfig> = {
  employee_bank: {
    entityType: 'employee_bank',
    monthlyLimit: 1,           // employee can change once per month
    ownerOverrideLimit: 2,     // owner can override up to 2 times per month for an employee
    supportOverrideUnlimited: true
  },
  org_bank: {
    entityType: 'org_bank',
    monthlyLimit: 1,           // org bank account changes once per month
    ownerOverrideLimit: 0,     // owner CANNOT override org bank — support only
    supportOverrideUnlimited: true
  },
  subscription: {
    entityType: 'subscription',
    monthlyLimit: 1,           // subscription payment method once per month
    ownerOverrideLimit: 0,     // owner cannot override subscription changes — support only
    supportOverrideUnlimited: true
  }
}

export async function checkRateLimit(
  workspaceId: string,
  entityId: string,
  entityType: string,
  requestorRole: string,
  isOverride: boolean
): Promise<{ allowed: boolean, reason?: string, changesThisMonth: number, limit: number }> {
  
  // Platform support can always make changes
  if (requestorRole === 'platform_support' || requestorRole === 'platform_admin') {
    return { allowed: true, changesThisMonth: 0, limit: 999 }
  }
  
  const config = RATE_LIMIT_CONFIGS[entityType]
  const currentMonth = new Date().toISOString().slice(0, 7)  // YYYY-MM
  
  // Count changes this month
  const changesThisMonth = await countChangesThisMonth(workspaceId, entityId, entityType, currentMonth)
  
  // Standard limit check
  if (changesThisMonth < config.monthlyLimit) {
    return { allowed: true, changesThisMonth, limit: config.monthlyLimit }
  }
  
  // Over standard limit — check if override is valid
  if (isOverride) {
    // Only org_owner can use override
    if (!['org_owner', 'org_co_owner'].includes(requestorRole)) {
      return {
        allowed: false,
        reason: `Payment information can only be changed once per month. Override requires org owner authorization.`,
        changesThisMonth,
        limit: config.monthlyLimit
      }
    }
    
    // Check override limit
    if (config.ownerOverrideLimit === 0) {
      return {
        allowed: false,
        reason: `This payment information can only be changed once per month. Changes beyond this limit require support agent assistance. Contact support for help.`,
        changesThisMonth,
        limit: config.monthlyLimit
      }
    }
    
    // Count owner overrides this month
    const overridesThisMonth = await countOverridesThisMonth(workspaceId, entityId, entityType, currentMonth)
    
    if (overridesThisMonth >= config.ownerOverrideLimit) {
      return {
        allowed: false,
        reason: `Maximum overrides reached for this month (${config.ownerOverrideLimit} overrides allowed). Contact support for additional changes.`,
        changesThisMonth,
        limit: config.monthlyLimit
      }
    }
    
    return { allowed: true, changesThisMonth, limit: config.monthlyLimit }
  }
  
  // Over limit, no override
  return {
    allowed: false,
    reason: `Payment information has already been updated ${changesThisMonth} time(s) this month. The next change can be made on ${getFirstOfNextMonth()}. If this is urgent, contact your org owner for an override or reach out to support.`,
    changesThisMonth,
    limit: config.monthlyLimit
  }
}
```
