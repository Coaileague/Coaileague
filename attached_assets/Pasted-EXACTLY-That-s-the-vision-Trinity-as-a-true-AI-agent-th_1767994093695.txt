async function executeTrinityWorkflow(workflow) {
  for (let action of workflow.actions) {
    // Show progress in chat
    updateChatBubble(`Executing: ${action.type}...`);
    
    // Execute action
    const result = await executeAction(action);
    
    // Show result with animation
    if (result.success) {
      showSuccess(`✓ ${action.type} completed`);
      highlightDOMElement(action.targetElement); // Pulse animation
    } else {
      showError(`✗ ${action.type} failed: ${result.error}`);
      offerRollback(workflow);
    }
    
    // Wait between actions (user sees progress)
    await sleep(500);
  }
  
  // Final summary
  showCompletionSummary({
    jobCreated: true,
    guardsNotified: true,
    scheduleUpdated: true,
    nextSteps: ["Guards have 2 hours to accept", "Client will receive invoice"]
  });
}
