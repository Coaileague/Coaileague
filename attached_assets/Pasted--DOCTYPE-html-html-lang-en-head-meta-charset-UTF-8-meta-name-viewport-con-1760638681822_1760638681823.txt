    <script>
        const messagesContainer = document.getElementById('messages-container');
        const chatInput = document.getElementById('chat-input');
        const userCountSpan = document.getElementById('user-count');
        const initialUserCount = parseInt(userCountSpan.textContent);
        
        /**
         * Simulates sending a message and renders it to the chat window.
         */
        function sendMessage() {
            const messageText = chatInput.value.trim();

            if (messageText === "") {
                return;
            }

            // 1. Create message element
            const messageDiv = document.createElement('div');
            messageDiv.className = 'flex justify-end'; // Align to the right for "self" message
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'self-msg max-w-[80%]';
            
            const headerSpan = document.createElement('span');
            headerSpan.className = 'font-bold text-green-700 text-sm';
            headerSpan.textContent = 'You';

            const paragraph = document.createElement('p');
            paragraph.className = 'text-gray-800 whitespace-pre-wrap';
            paragraph.textContent = messageText;

            contentDiv.appendChild(headerSpan);
            contentDiv.appendChild(paragraph);
            messageDiv.appendChild(contentDiv);
            
            // 2. Append to container and clear input
            messagesContainer.appendChild(messageDiv);
            chatInput.value = '';
            
            // 3. Scroll to the bottom
            messagesContainer.scrollTop = messagesContainer.scrollHeight;

            // 4. Simulate an automated system response (like a bot/IRC server)
            setTimeout(simulateBotResponse, 1500);
        }

        /**
         * Simulates a system or other user response.
         */
        function simulateBotResponse() {
            const botResponses = [
                "*** Server: Message sent successfully. ***",
                "*** Server: A wild 'PixelPioneer' has joined the channel. ***",
                "*** ArcadeMaster has sent you a NUDGE! ***",
                "ArcadeMaster: LOL! I totally forgot about that! Those were the days.",
                "System: Please refrain from excessive use of emoticons. IRC rules still apply.",
                "CodeWrangler: Speaking of 2.0, my connection speed is currently 56k. BRB.",
            ];

            const response = botResponses[Math.floor(Math.random() * botResponses.length)];
            
            const isSystem = response.startsWith('***');

            const messageDiv = document.createElement('div');
            messageDiv.className = 'flex justify-start'; // Align left for others
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'max-w-[80%]';

            if (isSystem) {
                // System message style
                messageDiv.className = 'text-center';
                contentDiv.className = 'system-msg';
                contentDiv.textContent = response;
                messageDiv.appendChild(contentDiv);
            } else {
                // User message style
                contentDiv.className = 'user-msg';
                
                const parts = response.split(':');
                const username = parts[0].trim();
                const text = parts.slice(1).join(':').trim();

                const headerSpan = document.createElement('span');
                headerSpan.className = 'font-bold text-blue-700 text-sm';
                headerSpan.textContent = username;

                const paragraph = document.createElement('p');
                paragraph.className = 'text-gray-800 whitespace-pre-wrap';
                paragraph.textContent = text;
                
                contentDiv.appendChild(headerSpan);
                contentDiv.appendChild(paragraph);
                messageDiv.appendChild(contentDiv);
            }

            messagesContainer.appendChild(messageDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            
            // Randomly update user count for realism
            if (Math.random() < 0.2) {
                userCountSpan.textContent = initialUserCount + Math.floor(Math.random() * 3) - 1;
            }
        }

        // Attach event listener for Enter key
        chatInput.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault(); // Prevent new line
                sendMessage();
            }
        });
        
        // Initial scroll to bottom (after initial messages are loaded)
        window.onload = () => {
             messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }

    </script>
