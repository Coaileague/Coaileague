typescript// When QuickBooks sync fails
if (syncResult.error) {
  // Send error to Gemini for analysis
  const model = genAI.getGenerativeModel({ model: GEMINI_MODELS.PRO });
  
  const prompt = `You are a QuickBooks integration specialist.
  
  ERROR DETAILS:
  ${JSON.stringify(syncResult.error)}
  
  DATA PAYLOAD:
  ${JSON.stringify(payloadSample)}
  
  PREVIOUS ATTEMPTS: ${retryCount}
  
  Analyze this error and provide:
  1. Root cause analysis
  2. Recommended action: RETRY | FIX_DATA | ESCALATE | ABORT
  3. If RETRY: suggested delay and modified payload
  4. If FIX_DATA: specific data corrections needed
  5. If ESCALATE: reason for human intervention
  
  Respond in JSON format.`;
  
  const analysis = await model.generateContent(prompt);
  const decision = JSON.parse(analysis.response.text());
  
  // LLM Judge validates the retry decision
  const riskEval = await llmJudgeEvaluator.evaluateRisk({
    action: 'quickbooks.retry_sync',
    context: decision
  });
  
  if (decision.action === 'RETRY' && riskEval.approved) {
    // Gemini suggested retry with modified data
    await retrySyncWithModifications(decision.modifications);
  } else if (decision.action === 'ESCALATE') {
    // Gemini determined human needed
    await escalateToAdmin(decision.reasoning);
  }
}
