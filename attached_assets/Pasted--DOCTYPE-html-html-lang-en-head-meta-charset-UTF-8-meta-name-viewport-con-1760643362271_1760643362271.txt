    <script>
        const messagesContainer = document.getElementById('messages-container');
        const chatInput = document.getElementById('chat-input');
        const userCountSpan = document.getElementById('user-count');
        const usernameInput = document.getElementById('username-input');
        const initialUserCount = parseInt(userCountSpan.textContent);
        
        /**
         * Simulates sending a message and renders it to the chat window.
         */
        function sendMessage() {
            const messageText = chatInput.value.trim();
            const myUsername = usernameInput.value;

            if (messageText === "") {
                return;
            }

            // 1. Create message element
            const messageDiv = document.createElement('div');
            messageDiv.className = 'flex justify-end'; // Align to the right for "self" message
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'self-msg max-w-[80%] flex items-start space-x-2';
            
            // Avatar/Emoji for self
            const selfEmojiDiv = document.createElement('div');
            selfEmojiDiv.className = 'w-8 h-8 flex-shrink-0 bg-fuchsia-200 rounded-full text-lg text-center leading-loose order-1';
            selfEmojiDiv.textContent = '✨';

            const textContainer = document.createElement('div');
            textContainer.className = 'order-2';
            
            const headerSpan = document.createElement('span');
            headerSpan.className = 'font-bold text-green-700 text-sm block text-right';
            headerSpan.textContent = myUsername; // Use the dynamic username

            const paragraph = document.createElement('p');
            paragraph.className = 'text-gray-800 mt-1 whitespace-pre-wrap';
            paragraph.textContent = messageText;

            textContainer.appendChild(headerSpan);
            textContainer.appendChild(paragraph);
            contentDiv.appendChild(selfEmojiDiv);
            contentDiv.appendChild(textContainer);
            messageDiv.appendChild(contentDiv);
            
            // 2. Append to container and clear input
            messagesContainer.appendChild(messageDiv);
            chatInput.value = '';
            
            // 3. Scroll to the bottom
            messagesContainer.scrollTop = messagesContainer.scrollHeight;

            // 4. Simulate an automated system response
            setTimeout(simulateBotResponse, 1500);
        }

        /**
         * Simulates a system or other user response.
         */
        function simulateBotResponse() {
            const cuteBotResponses = [
                "*** System: Ooh, a new message just dropped! Let the good times roll! ***",
                "💖 CodeWitch: That's totally relatable! I'm sending good vibes your way! 🤗",
                "🎧 SynthwaveSam: BRB, got to switch my playlist to something more ✨aesthetic✨.",
                "*** DreamWeaver has sent a NUDGE! Did you feel that? ***",
                "🌈 PixelPainter: Working on a cute pixel art commission right now! Check it out later! 🎨",
                "🔮 MysticDev: My crystal ball says... you should try Tailwind CSS. It's magic!",
            ];

            const response = cuteBotResponses[Math.floor(Math.random() * cuteBotResponses.length)];
            
            const isSystem = response.startsWith('***');

            const messageDiv = document.createElement('div');
            messageDiv.className = 'flex justify-start'; // Align left for others
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'max-w-[80%]';

            if (isSystem) {
                // System message style
                messageDiv.className = 'text-center';
                contentDiv.className = 'system-msg';
                contentDiv.textContent = response;
                messageDiv.appendChild(contentDiv);
            } else {
                // User message style
                contentDiv.className = 'user-msg flex items-start space-x-2';
                
                const parts = response.split(':');
                const username = parts[0].trim();
                const text = parts.slice(1).join(':').trim();

                // Get a random emoji for the simulated user
                const userEmojis = ['😸', '🦄', '👽', '🎶', '🎀'];
                const randomEmoji = userEmojis[Math.floor(Math.random() * userEmojis.length)];
                
                // Avatar/Emoji for user
                const userEmojiDiv = document.createElement('div');
                userEmojiDiv.className = 'w-8 h-8 flex-shrink-0 bg-purple-200 rounded-full text-lg text-center leading-loose';
                userEmojiDiv.textContent = randomEmoji;
                
                const textContainer = document.createElement('div');

                const headerSpan = document.createElement('span');
                headerSpan.className = 'font-bold text-blue-700 text-sm block';
                headerSpan.textContent = username;

                const paragraph = document.createElement('p');
                paragraph.className = 'text-gray-800 mt-1 whitespace-pre-wrap';
                paragraph.textContent = text;
                
                textContainer.appendChild(headerSpan);
                textContainer.appendChild(paragraph);

                contentDiv.appendChild(userEmojiDiv);
                contentDiv.appendChild(textContainer);
                messageDiv.appendChild(contentDiv);
            }

            messagesContainer.appendChild(messageDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            
            // Randomly update user count for realism
            if (Math.random() < 0.2) {
                const newUserCount = initialUserCount + Math.floor(Math.random() * 3) - 1;
                userCountSpan.textContent = Math.max(2, newUserCount); // Ensure at least 2 users
            }
        }

        // Attach event listener for Enter key - Updated to support Shift+Enter
        chatInput.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault(); // Prevent new line
                sendMessage();
            }
        });
        
        // Initial scroll to bottom (after initial messages are loaded)
        window.onload = () => {
             messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }

    </script>
