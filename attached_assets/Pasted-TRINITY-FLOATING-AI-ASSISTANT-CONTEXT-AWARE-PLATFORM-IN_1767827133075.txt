```javascript
// Frontend sends page context to Trinity on every route change
const pageContext = {
  route: '/schedule',

    date_range: '2025-01-13 to 2025-01-19',
    employees_shown: 15,

    overtime_hours: 18
  },

    'create_shift',
    'auto_fill_schedule',
    'view_employee_availability',
    'publish_schedule'
  ]

// Send to Trinity backend on page load/change
await fetch('/api/trinity/update-context', {
  method: 'POST',
  body: JSON.stringify({
    workspace_id,
    user_id,
    context: pageContext
  })
});
```
