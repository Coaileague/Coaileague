      addThinkingStep(`💡 Trying alternative: ${alternative.description}`);

    await sleep(500);

  } else {
    addMessage(`⚠️ Couldn't fully complete goal after ${attempts} attempts. Here's what worked: ${partialResults()}`);
  }
