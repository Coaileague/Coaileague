    <script>
        // Sheet Management
        function openAddShift() {
            document.getElementById('sheetOverlay').classList.add('open');
            document.getElementById('addSheetSheet').classList.add('open');
        }

        function showAutoSchedule() {
            document.getElementById('sheetOverlay').classList.add('open');
            document.getElementById('autoScheduleSheet').classList.add('open');
        }

        function closeSheet() {
            document.getElementById('sheetOverlay').classList.remove('open');
            document.getElementById('addSheetSheet').classList.remove('open');
            document.getElementById('autoScheduleSheet').classList.remove('open');
        }

        function openQuickAdd(employeeName) {
            document.getElementById('employeeSelect').value = employeeName;
            openAddShift();
        }

        // Actions
        function saveShift() {
            closeSheet();
            showToast('✓ Shift created successfully');
        }

        function autoSchedule(type) {
            closeSheet();
            document.getElementById('loadingOverlay').classList.add('show');
            
            setTimeout(() => {
                document.getElementById('loadingOverlay').classList.remove('show');
                let message = '';
                switch(type) {
                    case 'week':
                        message = '✓ 12 shifts scheduled for this week';
                        break;
                    case 'optimize':
                        message = '✓ Schedule optimized - saved $890 in overtime';
                        break;
                    case 'month':
                        message = '✓ 156 shifts scheduled for next month';
                        break;
                    case 'gaps':
                        message = '✓ 3 coverage gaps filled';
                        break;
                }
                showToast(message);
            }, 2500);
        }

        function previousWeek() {
            showToast('← Loaded previous week');
        }

        function nextWeek() {
            showToast('→ Loaded next week');
        }

        // Toast
        function showToast(message) {
            const toast = document.getElementById('toast');
            document.getElementById('toastMessage').textContent = message;
            toast.classList.add('show');
            setTimeout(() => {
                toast.classList.remove('show');
            }, 3000);
        }

        // Day Tab Selection
        document.addEventListener('DOMContentLoaded', function() {
            const dayTabs = document.querySelectorAll('.day-tab');
            dayTabs.forEach(tab => {
                tab.addEventListener('click', function() {
                    dayTabs.forEach(t => t.classList.remove('active'));
                    this.classList.add('active');
                    showToast('Loaded ' + this.querySelector('.day-name').textContent);
                });
            });
        });

        // Prevent body scroll when sheet is open
        const sheetOverlay = document.getElementById('sheetOverlay');
        sheetOverlay.addEventListener('touchmove', (e) => {
            e.preventDefault();
        }, { passive: false });
    </script>
