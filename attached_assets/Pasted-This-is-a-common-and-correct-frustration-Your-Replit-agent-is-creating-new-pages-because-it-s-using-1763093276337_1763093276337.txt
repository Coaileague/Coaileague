This function checks the user's authenticated token (JWT) to see if their assigned roles match the allowed_roles.The Universal Logic: This single function is imported and used as a decorator on every protected route.Python# Logic in a file like 'auth_deps.py'
def role_required(allowed_roles: List[str]):
    def wrapper(user: User = Depends(get_current_user)):
        if not any(role in user.roles for role in allowed_roles):
            raise HTTPException(status_code=403, detail="Insufficient privileges.")
        return user
    return wrapper

def roles_required(allowed_roles):
    def decorator(f):
        @wraps(f)
        def decorated_function(*args, **kwargs):
            current_user = load_user_from_session()
            if current_user.role not in allowed_roles:
                return redirect(url_for('unauthorized_page'))
            return f(*args, **kwargs)
        return decorated_function
    return decorator
