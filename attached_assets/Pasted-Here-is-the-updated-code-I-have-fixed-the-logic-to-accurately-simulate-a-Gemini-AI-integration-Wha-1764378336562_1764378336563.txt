import React, { useState, useEffect, useRef } from 'react';

// --- MOCK GEMINI SERVICE ---
// In production, this would be replaced by the @google/generative-ai SDK
const MockGeminiService = {
  analyzeLead: async (lead) => {
    // Simulating network delay for realism
    await new Promise(r => setTimeout(r, 2000));

    // LOGIC: Deterministic "Smart" Mocking based on lead data
    // This simulates Gemini analyzing the text content
    let baseScore = 75;
    const risks = [];
    const strengths = [];
    
    // 1. Budget Analysis
    if (lead.budget === '<$100K') {
      baseScore -= 20;
      risks.push('Budget below typical enterprise threshold');
    } else if (lead.budget === '$1M+') {
      baseScore += 10;
      strengths.push('Strong financial backing');
    }

    // 2. Timeline Analysis
    if (lead.timeline === 'Immediate') {
      baseScore += 5;
      risks.push('Aggressive timeline requires immediate resource allocation');
    }

    // 3. Technical Analysis (Keyword matching)
    const techStack = (lead.aiStack || '').toLowerCase();
    if (techStack.includes('tensorflow') || techStack.includes('pytorch')) {
      strengths.push('Modern tech stack (PyTorch/TF)');
    } else if (techStack === '') {
      risks.push('No existing AI infrastructure defined');
    }

    // 4. Industry Fit
    if (['Healthcare AI', 'Financial Services'].includes(lead.industry)) {
      strengths.push(`High-value ${lead.industry} use case`);
    }

    // Clamp score 0-100
    const finalScore = Math.min(Math.max(baseScore + Math.floor(Math.random() * 10 - 5), 10), 99);

    return {
      score: finalScore,
      analysis: {
        summary: `Analyzed ${lead.orgName} using Gemini 2.5 Flash context window. The proposed use case regarding ${lead.useCase.substring(0, 30)}... shows ${finalScore > 70 ? 'strong' : 'moderate'} promise.`,
        strengths: strengths.length > 0 ? strengths : ['Standard application'],
        risks: risks.length > 0 ? risks : ['None identified'],
        recommendedAction: finalScore < 50 ? 'reject' : finalScore > 80 ? 'onboarding' : 'human_review'
      }
    };
  },

  chat: async (input, context) => {
    await new Promise(r => setTimeout(r, 1200));
    const lowerInput = input.toLowerCase();
    
    // Simulating Gemini's ability to read the context (the selected lead)
    if (context?.selectedLead) {
      const l = context.selectedLead;
      if (lowerInput.includes('risk') || lowerInput.includes('concern')) {
        return `**Gemini Analysis:**\nBased on the profile for **${l.orgName}**, the primary risks are:\n${l.analysis?.risks.map(r => `* ${r}`).join('\n') || '* No specific risks flagged yet.'}`;
      }
      if (lowerInput.includes('email') || lowerInput.includes('draft')) {
        return `**Draft Outreach for ${l.contact}:**\n\n"Hi ${l.contact.split(' ')[0]},\n\nI reviewed your RFP for ${l.orgName} regarding ${l.useCase}. Given your timeline of ${l.timeline}, we should discuss how our platform fits your needs...\n\n[Gemini: Draft saved to drafts folder]"`;
      }
    }

    // General Pipeline queries
    if (lowerInput.includes('status') || lowerInput.includes('summary')) {
      return `**Pipeline Report:**\nCurrently tracking **${context.totalLeads}** active RFPs.\n\n* **Attention Needed:** ${context.needsReviewCount} leads waiting in Human Review.\n* **High Priority:** ${context.highValueCount} leads with budget >$1M.`;
    }

    return `**Gemini:** I can help you analyze **${context?.selectedLead?.orgName || 'your pipeline'}**. Try asking about risks, drafting emails, or prioritizing leads.`;
  }
};

    setProcessing(true);
    
    // CALL MOCK GEMINI
    const result = await MockGeminiService.analyzeLead(lead);
    
    // Apply Logic Rules automatically based on Gemini's result
    let nextStage = 'human_review'; // Default
    if (result.analysis.recommendedAction === 'reject') nextStage = 'rejected';
    if (result.analysis.recommendedAction === 'onboarding') nextStage = 'onboarding'; // Or negotiation
    if (lead.referral) nextStage = 'human_review'; // Always review referrals manually

      ...lead, 
      score: result.score, 
      analysis: result.analysis, 
      stage: nextStage 
    };

    if (selected && selected.id === lead.id) setSelected(updatedLead);
    setProcessing(false);
  };

    if (!aiInput.trim()) return;
    const userMsg = { role: 'user', text: aiInput };
    setAiChat(prev => [...prev, userMsg]);
    setAiInput('');
    setProcessing(true);

    // Context for Gemini
    const context = {
      selectedLead: selected,
      totalLeads: leads.length,
      needsReviewCount: leads.filter(l => l.stage === 'human_review').length,
      highValueCount: leads.filter(l => l.budget === '$1M+').length
    };
    
    // CALL MOCK GEMINI
    const response = await MockGeminiService.chat(userMsg.text, context);
    
    setAiChat(prev => [...prev, { role: 'ai', text: response }]);
    setProcessing(false);
  };

    setLeads(prev => [newLead, ...prev]);

  };

  const filtered = leads.filter(l => !search || l.orgName.toLowerCase().includes(search.toLowerCase()));

            ✨ Gemini AI

            + New RFP
          </button>

      </header>

      {/* Main Layout */}
      <div style={styles.mainLayout}>

              </div>

          </div>
          
          <div style={styles.navSection}>
            <div style={styles.navTitle}>Analytics</div>
            <div style={styles.miniStat}>

    budget: '$100K-250K', timeline: 'Q3 2026', useCase: '', aiStack: '', referral: false
  });

  return (
    <form onSubmit={(e) => { e.preventDefault(); onSubmit(form); }} style={{display:'flex', flexDirection:'column', gap: 12}}>
      <input style={styles.formInput} placeholder="Org Name" value={form.orgName} onChange={e => setForm({...form, orgName: e.target.value})} required />
      <select style={styles.formSelect} value={form.budget} onChange={e => setForm({...form, budget: e.target.value})}>
        <option value="<$100K">Budget: &lt;$100K (Triggers Low Score)</option>
        <option value="$100K-250K">Budget: $100K-250K</option>
        <option value="$1M+">Budget: $1M+ (Triggers High Score)</option>
      </select>
      <input style={styles.formInput} placeholder="AI Use Case" value={form.useCase} onChange={e => setForm({...form, useCase: e.target.value})} required />
      <div style={{display:'flex', justifyContent:'flex-end', gap:10, marginTop:10}}>
        <button type="button" onClick={onCancel} style={{...styles.btn, ...styles.btnSecondary}}>Cancel</button>
        <button type="submit" style={{...styles.btn, ...styles.btnPrimary}}>Save Lead</button>
      </div>
    </form>
  );
}
