CREATE TABLE IF NOT EXISTS chat_rooms(

import React, { useState, useEffect, useRef } from "react";
import { Send, Paperclip, X, Menu, Phone, Video, MoreVertical, User, Clock, CheckCheck } from "lucide-react";

interface Message {
  id: string;
  sender_id: string;
  sender_type: "user" | "agent" | "system";
  message_type: "text" | "image" | "file" | "action";
  content: string;
  metadata?: any;
  created_at: string;
}

interface Participant {
  id: string;
  name: string;
  type: "user" | "agent";
  avatar?: string;
  status: "online" | "away" | "offline";
}

interface Macro {
  id: string;
  title: string;
  content: string;
  shortcut: string;
  category: string;
}

export default function SupportChat({ 
  roomId, 
  currentUserId, 
  isAgent = false 
}: { 
  roomId: string; 
  currentUserId: string; 
  isAgent?: boolean;
}) {
  const [messages, setMessages] = useState<Message[]>([]);
  const [input, setInput] = useState("");
  const [participants, setParticipants] = useState<Participant[]>([]);
  const [showParticipants, setShowParticipants] = useState(false);
  const [showMacros, setShowMacros] = useState(false);
  const [macros, setMacros] = useState<Macro[]>([]);
  const [typing, setTyping] = useState<string[]>([]);
  const [showActions, setShowActions] = useState(false);
  
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const wsRef = useRef<WebSocket | null>(null);

  // WebSocket connection
  useEffect(() => {
    const ws = new WebSocket(`wss://${window.location.host}/ws/chat/${roomId}`);
    wsRef.current = ws;

    ws.onmessage = (e) => {
      const data = JSON.parse(e.data);
      
      if (data.type === "message") {
        setMessages(prev => [...prev, data.message]);
      } else if (data.type === "typing") {
        setTyping(prev => {
          if (data.isTyping && !prev.includes(data.userName)) {
            return [...prev, data.userName];
          } else if (!data.isTyping) {
            return prev.filter(n => n !== data.userName);
          }
          return prev;
        });
      } else if (data.type === "participants") {
        setParticipants(data.participants);
      }
    };

    return () => ws.close();
  }, [roomId]);

  // Load initial data
  useEffect(() => {
    Promise.all([
      fetch(`/api/chat/${roomId}/messages`).then(r => r.json()),
      fetch(`/api/chat/${roomId}/participants`).then(r => r.json()),
      isAgent ? fetch(`/api/chat/macros`).then(r => r.json()) : Promise.resolve([])
    ]).then(([msgs, parts, mcrs]) => {
      setMessages(msgs);
      setParticipants(parts);
      setMacros(mcrs);
    });
  }, [roomId, isAgent]);

  // Auto-scroll to bottom
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  // Typing indicator
  useEffect(() => {
    if (input) {
      wsRef.current?.send(JSON.stringify({ type: "typing", isTyping: true }));
    }
    const timeout = setTimeout(() => {
      wsRef.current?.send(JSON.stringify({ type: "typing", isTyping: false }));
    }, 1000);
    return () => clearTimeout(timeout);
  }, [input]);

  const sendMessage = () => {
    if (!input.trim()) return;

    const message = {
      room_id: roomId,
      content: input,
      sender_type: isAgent ? "agent" : "user"
    };

    fetch(`/api/chat/${roomId}/messages`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(message)
    });

    setInput("");
  };

  const useMacro = (macro: Macro) => {
    setInput(macro.content);
    setShowMacros(false);
  };

  const getInitials = (name: string) => {
    return name.split(" ").map(n => n[0]).join("").toUpperCase().slice(0, 2);
  };

  const renderMessage = (msg: Message) => {
    const isOwn = msg.sender_id === currentUserId;
    const isSystem = msg.sender_type === "system";

    if (isSystem) {
      return (
        <div className="flex justify-center my-4">
          <div className="bg-zinc-800/50 px-4 py-2 rounded-full text-xs text-zinc-400">
            {msg.content}
          </div>
        </div>
      );
    }

    return (
      <div className={`flex gap-3 ${isOwn ? "flex-row-reverse" : ""} mb-4 px-4`}>
        <div className={`flex-shrink-0 w-10 h-10 rounded-full flex items-center justify-center text-sm font-semibold ${
          isOwn ? "bg-blue-500" : msg.sender_type === "agent" ? "bg-emerald-500" : "bg-zinc-700"
        }`}>
          {getInitials(participants.find(p => p.id === msg.sender_id)?.name || "U")}
        </div>
        <div className={`flex flex-col max-w-[75%] ${isOwn ? "items-end" : ""}`}>
          <div className={`rounded-2xl px-4 py-2 ${
            isOwn 
              ? "bg-blue-500 text-white" 
              : msg.sender_type === "agent"
              ? "bg-emerald-500/20 text-emerald-100 border border-emerald-500/30"
              : "bg-zinc-800 text-zinc-100"
          }`}>
            {msg.message_type === "image" ? (
              <img src={msg.content} alt="Uploaded" className="rounded-lg max-w-full" />
            ) : (
              <p className="text-sm leading-relaxed">{msg.content}</p>
            )}
          </div>
          <div className="flex items-center gap-2 mt-1 px-2">
            <span className="text-xs text-zinc-500">
              {new Date(msg.created_at).toLocaleTimeString([], { hour: "2-digit", minute: "2-digit" })}
            </span>
            {isOwn && (
              <CheckCheck className="w-3 h-3 text-blue-400" />
            )}
          </div>
        </div>
      </div>
    );
  };

  return (
    <div className="flex flex-col h-screen bg-zinc-950 text-zinc-100">
      {/* Header */}
      <header className="bg-zinc-900 border-b border-zinc-800 px-4 py-3 flex items-center justify-between">
        <div className="flex items-center gap-3">
          <button 
            onClick={() => setShowParticipants(!showParticipants)}
            className="p-2 hover:bg-zinc-800 rounded-lg transition-colors"
          >
            <Menu className="w-5 h-5" />
          </button>
          <div>
            <h1 className="font-semibold text-sm">Support Chat</h1>
            <p className="text-xs text-zinc-400">
              {participants.filter(p => p.status === "online").length} online
            </p>
          </div>
        </div>
        <div className="flex items-center gap-2">
          {isAgent && (
            <>
              <button className="p-2 hover:bg-zinc-800 rounded-lg transition-colors">
                <Phone className="w-5 h-5" />
              </button>
              <button className="p-2 hover:bg-zinc-800 rounded-lg transition-colors">
                <Video className="w-5 h-5" />
              </button>
            </>
          )}
          <button 
            onClick={() => setShowActions(!showActions)}
            className="p-2 hover:bg-zinc-800 rounded-lg transition-colors"
          >
            <MoreVertical className="w-5 h-5" />
          </button>
        </div>
      </header>

      {/* Messages */}
      <div className="flex-1 overflow-y-auto">
        <div className="py-4">
          {messages.map(msg => (
            <div key={msg.id}>
              {renderMessage(msg)}
            </div>
          ))}
          
          {/* Typing indicator */}
          {typing.length > 0 && (
            <div className="flex gap-3 mb-4 px-4">
              <div className="w-10 h-10 rounded-full bg-zinc-700 flex items-center justify-center">
                <User className="w-5 h-5" />
              </div>
              <div className="bg-zinc-800 rounded-2xl px-4 py-3">
                <div className="flex gap-1">
                  <div className="w-2 h-2 bg-zinc-500 rounded-full animate-bounce" style={{ animationDelay: "0ms" }} />
                  <div className="w-2 h-2 bg-zinc-500 rounded-full animate-bounce" style={{ animationDelay: "150ms" }} />
                  <div className="w-2 h-2 bg-zinc-500 rounded-full animate-bounce" style={{ animationDelay: "300ms" }} />
                </div>
              </div>
            </div>
          )}
          
          <div ref={messagesEndRef} />
        </div>
      </div>

      {/* Input */}
      <div className="bg-zinc-900 border-t border-zinc-800 p-4">
        <div className="flex items-end gap-2">
          <button className="p-2 hover:bg-zinc-800 rounded-lg transition-colors">
            <Paperclip className="w-5 h-5 text-zinc-400" />
          </button>
          
          <div className="flex-1 bg-zinc-800 rounded-2xl px-4 py-2 focus-within:ring-2 focus-within:ring-blue-500">
            <textarea
              value={input}
              onChange={(e) => setInput(e.target.value)}
              onKeyDown={(e) => {
                if (e.key === "Enter" && !e.shiftKey) {
                  e.preventDefault();
                  sendMessage();
                }
              }}
              placeholder="Type a message..."
              rows={1}
              className="w-full bg-transparent border-none outline-none resize-none text-sm"
            />
          </div>

          {isAgent && (
            <button 
              onClick={() => setShowMacros(!showMacros)}
              className="p-2 hover:bg-zinc-800 rounded-lg transition-colors"
            >
              <Clock className="w-5 h-5 text-zinc-400" />
            </button>
          )}

          <button 
            onClick={sendMessage}
            disabled={!input.trim()}
            className="p-3 bg-blue-500 hover:bg-blue-600 disabled:bg-zinc-700 disabled:cursor-not-allowed rounded-xl transition-colors"
          >
            <Send className="w-5 h-5" />
          </button>
        </div>
      </div>

      {/* Floating Participants Panel */}
      {showParticipants && (
        <div className="fixed inset-0 bg-black/50 z-50" onClick={() => setShowParticipants(false)}>

          >
            <div className="sticky top-0 bg-zinc-900 border-b border-zinc-800 px-4 py-3 flex items-center justify-between">
              <h2 className="font-semibold">Participants ({participants.length})</h2>
              <button onClick={() => setShowParticipants(false)}>
                <X className="w-5 h-5" />
              </button>
            </div>
            <div className="p-4 space-y-3">

                    <div className={`w-12 h-12 rounded-full flex items-center justify-center text-sm font-semibold ${

                    }`}>
                      {getInitials(p.name)}

    </div>
  );

import { WebSocket } from "ws";
import { Pool } from "pg";

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const rooms = new Map<string, Set<WebSocket>>();

export function handleChatWebSocket(ws: WebSocket, roomId: string, userId: string, userName: string) {
  // Add to room
  if (!rooms.has(roomId)) {
    rooms.set(roomId, new Set());
  }
  rooms.get(roomId)!.add(ws);

  // Broadcast participants update
  broadcastToRoom(roomId, {
    type: "participants",
    participants: Array.from(rooms.get(roomId)!).map(s => ({ id: userId, name: userName, status: "online" }))
  });

  ws.on("message", async (data) => {
    const msg = JSON.parse(data.toString());

    if (msg.type === "typing") {
      broadcastToRoom(roomId, {
        type: "typing",
        userName,
        isTyping: msg.isTyping
      }, ws);
    }
  });

  ws.on("close", () => {
    rooms.get(roomId)?.delete(ws);
    if (rooms.get(roomId)?.size === 0) {
      rooms.delete(roomId);
    }
    
    // Clean up typing indicator
    pool.query("DELETE FROM typing_indicators WHERE room_id = $1 AND user_id = $2", [roomId, userId]);
  });
}

function broadcastToRoom(roomId: string, message: any, exclude?: WebSocket) {
  const room = rooms.get(roomId);
  if (!room) return;

  const payload = JSON.stringify(message);
  room.forEach(client => {
    if (client !== exclude && client.readyState === WebSocket.OPEN) {
      client.send(payload);
    }
  });
}

import { Router } from "express";
import { Pool } from "pg";

const pool = new Pool({ connectionString: process.env.DATABASE_URL });
const r = Router();

// Get room messages
r.get("/api/chat/:roomId/messages", async (req, res) => {
  const result = await pool.query(
    \`SELECT * FROM chat_messages WHERE room_id = $1 ORDER BY created_at ASC LIMIT 100\`,
    [req.params.roomId]
  );
  res.json(result.rows);
});

// Send message
r.post("/api/chat/:roomId/messages", async (req, res) => {
  const { content, sender_type } = req.body;
  const sender_id = (req as any).user?.id || "anonymous";

  const result = await pool.query(
    \`INSERT INTO chat_messages(room_id, sender_id, sender_type, content)
     VALUES($1, $2, $3, $4)
     RETURNING *\`,
    [req.params.roomId, sender_id, sender_type, content]
  );

  // Update room updated_at
  await pool.query(
    \`UPDATE chat_rooms SET updated_at = now() WHERE id = $1\`,
    [req.params.roomId]
  );

  res.json(result.rows[0]);
});

// Get participants
r.get("/api/chat/:roomId/participants", async (req, res) => {
  const result = await pool.query(
    \`SELECT DISTINCT 
      cm.sender_id as id,
      COALESCE(u.name, 'User') as name,
      cm.sender_type as type,
      'online' as status
     FROM chat_messages cm
     LEFT JOIN users u ON u.id = cm.sender_id
     WHERE cm.room_id = $1\`,
    [req.params.roomId]
  );
  res.json(result.rows);
});

// Get macros (agents only)
r.get("/api/chat/macros", async (req, res) => {
  const result = await pool.query(
    \`SELECT * FROM chat_macros ORDER BY category, title\`
  );
  res.json(result.rows);
});

// Create room
r.post("/api/chat/rooms", async (req, res) => {
  const { subject, priority } = req.body;
  const user_id = (req as any).user?.id || "anonymous";

  const result = await pool.query(
    \`INSERT INTO chat_rooms(user_id, subject, priority)
     VALUES($1, $2, $3)
     RETURNING *\`,
    [user_id, subject, priority || "normal"]
  );

  res.json(result.rows[0]);
});

export default r;

```bash

```

### 2. Mount REST routes in `server.ts`

```ts
import chatRouter from './routes/api/chat';
app.use(chatRouter);
```

### 3. Set up WebSocket server

```ts
import { WebSocketServer } from 'ws';
import { handleChatWebSocket } from './routes/ws/chat';

const wss = new WebSocketServer({ server: httpServer });

wss.on('connection', (ws, req) => {
  const url = new URL(req.url!, `http://${req.headers.host}`);
  const roomId = url.pathname.split('/').pop()!;
  const userId = req.headers['x-user-id'] as string; // from auth
  const userName = req.headers['x-user-name'] as string;

  handleChatWebSocket(ws, roomId, userId, userName);
});
```
