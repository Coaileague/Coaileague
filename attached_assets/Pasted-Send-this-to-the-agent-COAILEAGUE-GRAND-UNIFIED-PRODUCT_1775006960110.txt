# Complete route inventory

  sort | uniq > /tmp/all_routes.txt

cat /tmp/all_routes.txt

# Complete table inventory
echo "=== ALL DATABASE TABLES ==="
psql $DATABASE_URL -t -c "
  SELECT table_name, 
    (SELECT COUNT(*) FROM information_schema.columns c 
     WHERE c.table_name = t.table_name) as col_count,
    pg_size_pretty(pg_total_relation_size(quote_ident(table_name))) as size
  FROM information_schema.tables t
  WHERE table_schema = 'public' 
  AND table_type = 'BASE TABLE'
  ORDER BY table_name
" > /tmp/all_tables.txt
wc -l /tmp/all_tables.txt
cat /tmp/all_tables.txt

DO $$
DECLARE r RECORD;
  today_str TEXT := TO_CHAR(NOW(), 'YYYYMMDD');
  next_num INTEGER;
BEGIN
  FOR r IN 
    SELECT id FROM document_vault 
    WHERE document_number IS NULL 
    ORDER BY created_at
  LOOP
    SELECT COALESCE(MAX(CAST(
      SUBSTRING(document_number FROM LENGTH('DOC-' || today_str || '-') + 1) 
      AS INTEGER)), 0) + 1
    INTO next_num
    FROM document_vault 
    WHERE document_number LIKE 'DOC-' || today_str || '-%';
    
    UPDATE document_vault 
    SET document_number = 'DOC-' || today_str || '-' || LPAD(next_num::TEXT, 5, '0')
    WHERE id = r.id;
  END LOOP;
  RAISE NOTICE 'Document number backfill complete';
END $$;
