// EXAMPLE 3: Speculative Execution Demo
// ═══════════════════════════════════════════════════════════════

async function example3_SpeculativeExecution() {
  console.log('\n╔═══════════════════════════════════════════════════════════╗');
  console.log('║   EXAMPLE 3: Speculative Execution - Ultra Fast          ║');
  console.log('╚═══════════════════════════════════════════════════════════╝\n');

  const task = 'Analyze this week\'s payroll for anomalies';
  const context = { weekEnding: '2024-12-08', employees: 50 };

  console.log('Task:', task);
  console.log('\n🔮 Speculative execution: Starting likely agents BEFORE decomposition...\n');

  // Create speculative executor
  const speculative = new SpeculativeExecutor(trinity);
  
  // Start speculative execution (non-blocking)
  const speculativePromises = await speculative.speculateAgents(task);

  console.log('🧠 Now running full decomposition in parallel...\n');

  // Run full orchestration
  const result = await trinity.orchestrateFast(task, context);

  // Check if speculative results were used
  const speculativeResults = speculative.getResults();
  console.log(`\n📊 Speculative results used: ${speculativeResults.size}`);

  console.log('\n⚡ TOTAL TIME:', (result.executionTime / 1000).toFixed(2) + 's');
  console.log('   (Some agents completed before decomposition finished!)\n');

  return result;
}

async function example5_EarlyStopping() {
  console.log('\n╔═══════════════════════════════════════════════════════════╗');
  console.log('║   EXAMPLE 5: Early Stopping - Smart Efficiency           ║');
  console.log('╚═══════════════════════════════════════════════════════════╝\n');

  // Configure with early stopping
  const fastTrinity = new TrinityFastMode(process.env.GEMINI_API_KEY, {
    enableEarlyStop: true,
    earlyStopConfidence: 85,
    maxParallelAgents: 10
  });

  const task = 'Should we hire 5 more warehouse workers? Current staff: 50, workload up 20%';
  const context = { currentStaff: 50, workloadIncrease: 0.20, budget: 500000 };

  console.log('Task:', task);
  console.log('Early stop enabled: Will stop at 85% confidence\n');

  const result = await fastTrinity.orchestrateFast(task, context);

  console.log('\n📊 RESULTS:');
  console.log(`Agents executed: ${result.agentsUsed}`);
  console.log(`Time: ${(result.executionTime / 1000).toFixed(2)}s`);
  console.log('Note: May have stopped early before running all planned agents!\n');

  return result;
}

async function runBenchmark() {
  console.log('\n╔═══════════════════════════════════════════════════════════╗');
  console.log('║        BENCHMARK: Trinity vs Replit Fast Mode            ║');
  console.log('╚═══════════════════════════════════════════════════════════╝\n');

  const tasks = [
    {
      name: 'Simple (1-2 agents)',
      task: 'Send welcome email',
      replitEstimate: 20,
      context: {}
    },
    {
      name: 'Medium (3-5 agents)',
      task: 'Create schedule and check compliance',
      replitEstimate: 45,
      context: { employees: 30 }
    },
    {
      name: 'Complex (6+ agents)',
      task: 'Complete onboarding with schedule, payroll, email, and compliance',
      replitEstimate: 90,
      context: { employee: 'John Doe' }
    }
  ];

  console.log('Running benchmarks...\n');
  console.log('Task Type               | Replit Est. | Trinity Actual | Speedup');
  console.log('━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━');

  for (const testCase of tasks) {
    const result = await trinity.orchestrateFast(testCase.task, testCase.context);
    const trinityTime = result.executionTime / 1000;
    const speedup = testCase.replitEstimate / trinityTime;
    
    console.log(
      `${testCase.name.padEnd(23)} | ${testCase.replitEstimate.toString().padEnd(11)}s | ${trinityTime.toFixed(2).padEnd(14)}s | ${speedup.toFixed(1)}x`
    );
  }

  console.log('\n✅ Benchmark complete!\n');
}
