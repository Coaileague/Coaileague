### **Step 3: Profit Calculation (Most Important)**
```javascript
const calculateProfitScore = (employee, shift) => {
  // What client pays per hour
  const clientRate = shift.client_pay_rate; // $45/hour
  
  // What employee costs
  const employeeCost = calculateTotalEmployeeCost(employee, shift);
  
  // Gross profit
  const profit = clientRate - employeeCost;
  
  // Profit margin %
  const profitMargin = (profit / clientRate) * 100;
  
  // Score based on profit margin
  // Higher margin = higher score
  return Math.min(profitMargin * 2, 100); // Cap at 100
};

const calculateTotalEmployeeCost = (employee, shift) => {
  let basePay = employee.hourly_rate; // $18/hour base
  
  // Add overtime multiplier if applicable
  const hoursThisWeek = getHoursWorkedThisWeek(employee, shift.date);
  if (hoursThisWeek + shift.hours > 40) {
    basePay *= 1.5; // Overtime rate
  }
  
  // Add burden rate (taxes, benefits, insurance)
  const burdenMultiplier = 1.25; // 25% burden
  
  // Total cost per hour
  return basePay * burdenMultiplier;
  
  // Example: $18 base × 1.25 burden = $22.50/hour total cost
  // Client pays $45, profit = $22.50/hour margin
};
```

### **Step 4: Qualification Score**
```javascript
const calculateQualificationScore = (employee, shift) => {
  let score = 0;
  
  // Required certifications (40 points)
  const hasCertifications = shift.required_certifications.every(cert =>
    employee.certifications.includes(cert)
  );
  score += hasCertifications ? 40 : 0;
  
  // Required skills (30 points)
  const matchingSkills = employee.skills.filter(skill =>
    shift.required_skills.includes(skill)
  ).length;
  score += (matchingSkills / shift.required_skills.length) * 30;
  
  // Experience level (20 points)
  const experienceYears = calculateExperience(employee, shift.position);
  score += Math.min(experienceYears * 4, 20); // 4 points per year, max 20
  
  // Seniority bonus (10 points)
  const tenureMonths = calculateTenure(employee);
  score += Math.min(tenureMonths / 6, 10); // 1 point per 6 months
  
  return score;
};
```

### **Step 5: Reliability Score**
```javascript
const calculateReliabilityScore = (employee) => {
  const history = employee.work_history;
  
  // On-time arrival rate (40 points)
  const onTimeRate = history.on_time_count / history.total_shifts;
  const onTimeScore = onTimeRate * 40;
  
  // Attendance rate (30 points) - no-shows/call-offs
  const attendanceRate = 1 - (history.no_shows / history.total_shifts);
  const attendanceScore = attendanceRate * 30;
  
  // Shift completion rate (20 points) - didn't leave early
  const completionRate = history.completed_shifts / history.total_shifts;
  const completionScore = completionRate * 20;
  
  // Recent trend (10 points) - improving or declining?
  const recentPerformance = calculateRecentTrend(employee, 30); // last 30 days
  const trendScore = recentPerformance * 10;
  
  return onTimeScore + attendanceScore + completionScore + trendScore;
};
```

### **Step 6: Compliance Score**
```javascript
const calculateComplianceScore = (employee, shift) => {
  let score = 100; // Start perfect, deduct for violations
  
  // Check rest period (8 hours between shifts)
  const lastShiftEnd = getLastShiftEnd(employee);
  const hoursSinceLastShift = (shift.start_time - lastShiftEnd) / 3600000;
  if (hoursSinceLastShift < 8) {
    score -= 30; // Major violation
  }
  
  // Check weekly hour limit
  const hoursThisWeek = getHoursWorkedThisWeek(employee, shift.date);
  if (hoursThisWeek + shift.hours > 60) {
    score -= 25; // Excessive hours
  }
  
  // Check consecutive days worked
  const consecutiveDays = getConsecutiveDaysWorked(employee, shift.date);
  if (consecutiveDays >= 6) {
    score -= 20; // Needs a day off
  }
  
  // Check break requirements during shift
  if (shift.hours > 6 && !shift.includes_break) {
    score -= 15; // Missing required break
  }
  
  // Check minor labor laws (if under 18)
  if (employee.age < 18) {
    if (shift.end_time > "22:00" || shift.hours > 8) {
      score -= 40; // Can't work late or long hours
    }
  }
  
  return Math.max(score, 0); // Can't go negative
};
```

      maxOutputTokens: 1024,

    shift_id: shift.id,
    decision: decision,
    candidates_considered: candidates.length,

  });

      manager: shift.manager_on_duty,
      type: "CALL_OFF_REPLACEMENT",

    // No replacements - send to employee pool

      shift: shift,

      notification_channels: ["push", "sms", "email"],

    });
