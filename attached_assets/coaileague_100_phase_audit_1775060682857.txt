GUARD 1 — STATEWIDE PROGRAMMATIC WRITE PROTECTION

Create server/middleware/statewideGuard.ts:

import { Request, Response, NextFunction } from 'express';
import { logger } from '../utils/logger';
import { pool } from '../db';

const STATEWIDE_WORKSPACE_ID = process.env.STATEWIDE_WORKSPACE_ID;

export function statewideWriteGuard(req: Request, res: Response, next: NextFunction) {
  if (!STATEWIDE_WORKSPACE_ID) return next();
  const workspaceId =
    req.user?.workspaceId || req.body?.workspaceId ||
    req.params?.workspaceId || req.query?.workspaceId;
  const isMutation = ['POST','PUT','PATCH','DELETE'].includes(req.method);
  const isAuthRoute = req.path.startsWith('/api/auth');
  const isHealthRoute = req.path === '/health';
  if (workspaceId === STATEWIDE_WORKSPACE_ID && isMutation && !isAuthRoute && !isHealthRoute) {
    logger.error('STATEWIDE WRITE BLOCKED', { path: req.path, method: req.method, userId: req.user?.userId });
    pool.query(`INSERT INTO admin_audit_log (action, actor_id, metadata, created_at)
      VALUES ('statewide_write_blocked', $1, $2, NOW())`,
      [req.user?.userId || 'anonymous', JSON.stringify({ path: req.path, method: req.method, ip: req.ip })]
    ).catch(() => {});
    return res.status(403).json({
      error: 'Write operations are blocked on this workspace during testing.',
      code: 'STATEWIDE_PROTECTED',
    });
  }
  next();
}

];
export function validateEnvironment(): void {
  const missing = CRITICAL_VARS.filter(v => !process.env[v]);
  const warned = BILLING_VARS.filter(v => !process.env[v]);
  if (missing.length > 0) {
    console.error('FATAL: Missing critical env vars:');
    missing.forEach(v => console.error('  MISSING: ' + v));
    process.exit(1);
  }
  if (warned.length > 0) {
    console.warn('WARNING: Missing billing env vars:');
    warned.forEach(v => console.warn('  MISSING: ' + v));
  }
  console.log('Environment validation passed');
}
