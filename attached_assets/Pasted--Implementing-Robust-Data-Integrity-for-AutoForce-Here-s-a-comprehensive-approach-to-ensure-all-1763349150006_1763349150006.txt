## 2. **ID Generation & Management**

**Never reuse IDs** using these strategies:

### UUID v7 (Recommended)
```javascript
// Time-ordered UUIDs - sortable and guaranteed unique
import { v7 as uuidv7 } from 'uuid';

function generateUserId() {
  return `user_${uuidv7()}`;
}

function generateOrgId() {
  return `org_${uuidv7()}`;
}
```

### Database Constraints
```sql
-- PostgreSQL example
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  deleted_at TIMESTAMPTZ NULL, -- Soft delete, never reuse
  -- ... other fields
);

-- Prevent ID reuse even after deletion
CREATE UNIQUE INDEX users_id_unique ON users(id);

-- Track all ID assignments
CREATE TABLE id_registry (
  id UUID PRIMARY KEY,
  entity_type VARCHAR(50) NOT NULL,
  issued_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  never_reuse BOOLEAN NOT NULL DEFAULT TRUE
);
```

## 3. **Transaction Logging System**

Create an audit trail for **every** action:

```javascript
class AuditLogger {
  async logAction({
    actorId,
    actorType,
    action,
    entityId,
    entityType,
    changes,
    success,
    errorMessage = null
  }) {
    const auditEntry = {
      id: uuidv7(),
      timestamp: new Date().toISOString(),
      actorId,
      actorType, // 'END_USER', 'SUPPORT', 'AI'
      action, // 'CREATE', 'UPDATE', 'DELETE', 'FIX'
      entityId,
      entityType, // 'USER', 'ORG', 'TASK'
      changesBefore: changes.before,
      changesAfter: changes.after,
      success,
      errorMessage,
      metadata: {
        timestamp: Date.now(),
        serverTime: new Date().toISOString()
      }
    };
    
    // Write to database with retry logic
    await this.writeToAuditLog(auditEntry);
    
    // Also write to separate immutable log store
    await this.writeToImmutableStore(auditEntry);
    
    return auditEntry.id;
  }
}
```

## 4. **Database Sync Patterns**

### Option A: Write-Ahead Logging (WAL)
```javascript
// Every action first writes to a log, then to main DB
async function performAction(action) {
  const logId = await writeToLog(action);
  
  try {
    const result = await executeAction(action);
    await markLogComplete(logId, result);
    return result;
  } catch (error) {
    await markLogFailed(logId, error);
    throw error;
  }
}
```

### Option B: Two-Phase Commit
```javascript
async function syncAction(action) {
  // Phase 1: Prepare
  const transactionId = await prepareTransaction(action);
  
  // Phase 2: Commit
  try {
    await commitToDatabase(transactionId, action);
    await commitToCache(transactionId, action);
    await notifyServices(transactionId, action);
  } catch (error) {
    await rollbackTransaction(transactionId);
    throw error;
  }
}
```
