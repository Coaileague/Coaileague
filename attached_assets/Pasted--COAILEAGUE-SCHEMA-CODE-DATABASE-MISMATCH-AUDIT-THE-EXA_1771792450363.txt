## 2.1 Extract Schema Column Definitions

```bash
#!/bin/bash
# extract_schema_columns.sh

echo "=== EXTRACTING COLUMNS FROM SCHEMA.TS ==="

# Parse each table and its columns
node << 'SCRIPT'
const fs = require('fs');
const content = fs.readFileSync('shared/schema.ts', 'utf8');

// Match table definitions
const tableRegex = /export const (\w+) = pgTable\(['"]([^'"]+)['"]/g;
const columnRegex = /(\w+):\s*(varchar|text|integer|boolean|timestamp|uuid|jsonb|serial|pgEnum)/g;

let match;
const tables = {};

// Find all tables
while ((match = tableRegex.exec(content)) !== null) {
  const [_, varName, tableName] = match;
  tables[varName] = { 
    sqlName: tableName, 
    jsName: varName,
    columns: [] 
  };
}

// For each table, find column section and extract
for (const [varName, table] of Object.entries(tables)) {
  const tableStart = content.indexOf(`export const ${varName} = pgTable`);
  const tableEnd = content.indexOf('});', tableStart);
  const tableBlock = content.substring(tableStart, tableEnd);
  
  // Extract columns
  const colMatches = tableBlock.matchAll(/(\w+):\s*(\w+)\(/g);
  for (const col of colMatches) {
    table.columns.push({
      jsName: col[1],
      type: col[2]
    });
  }
}

// Output
for (const [varName, table] of Object.entries(tables)) {
  console.log(`\nTABLE: ${varName} (SQL: ${table.sqlName})`);
  console.log('COLUMNS:');
  for (const col of table.columns) {
    // Predict SQL name (camelCase -> snake_case)
    const sqlName = col.jsName.replace(/([A-Z])/g, '_$1').toLowerCase();
    if (col.jsName !== sqlName.replace(/^_/, '')) {
      console.log(`  ${col.jsName} -> ${sqlName} (CASE CONVERSION)`);
    } else {
      console.log(`  ${col.jsName}`);
    }
  }
}
SCRIPT
```

# PART 5: AUTOMATED MISMATCH DETECTION

## 5.1 Comprehensive Detection Script

```bash
#!/bin/bash
# detect_all_mismatches.sh

echo "========================================================"
echo "COMPREHENSIVE SCHEMA/CODE/DATABASE MISMATCH DETECTION"
echo "========================================================"

ISSUES_FOUND=0
OUTPUT_FILE="/tmp/mismatch_report.txt"

echo "" > $OUTPUT_FILE

# Function to log issues
log_issue() {
  echo "[ISSUE] $1" | tee -a $OUTPUT_FILE
  ((ISSUES_FOUND++))
}

# 1. ENUM VALUE MISMATCHES
echo ""
echo "=== CHECKING ENUM VALUE MISMATCHES ==="

# Known problematic enum values from the document
KNOWN_BAD_VALUES=(
  "maintenance"      # Not in account_state
  "deactivated"      # Not in account_state
  "committed"        # Not in event_status
  "platform_staff"   # Should be SUPPORT_STAFF
)

for val in "${KNOWN_BAD_VALUES[@]}"; do
  count=$(grep -rn "'$val'\|\"$val\"" --include="*.ts" server/ | wc -l)
  if [ "$count" -gt 0 ]; then
    log_issue "Enum value '$val' used in $count places but may not exist in DB enum"
    grep -rn "'$val'\|\"$val\"" --include="*.ts" server/ | head -5 >> $OUTPUT_FILE
  fi
done

# 2. COLUMN NAME MISMATCHES
echo ""
echo "=== CHECKING COLUMN NAME MISMATCHES ==="

# Check for columns used in code that might not match schema
SUSPECT_COLUMNS=(
  "entityType"
  "entityId" 
  "resourceType"
  "resourceId"
  "actorId"
  "actorType"
  "details"
  "severity"
)

for col in "${SUSPECT_COLUMNS[@]}"; do
  # Check if used in inserts/updates
  usage=$(grep -rn "$col:" --include="*.ts" server/routes/ | wc -l)
  if [ "$usage" -gt 0 ]; then
    # Check if exists in schema
    in_schema=$(grep -c "$col:" shared/schema.ts)
    if [ "$in_schema" -eq 0 ]; then
      log_issue "Column '$col' used in code ($usage times) but not found in schema"
    fi
  fi
done

# 3. CASE SENSITIVITY ISSUES
echo ""
echo "=== CHECKING CASE SENSITIVITY ISSUES ==="

# Find uppercase enum values in schema
echo "Uppercase enum values in schema:"
grep -oP "'[A-Z][A-Z_]+'" shared/schema.ts | sort -u | while read val; do
  lowercase=$(echo $val | tr '[:upper:]' '[:lower:]')
  if grep -rq "$lowercase" --include="*.ts" server/; then
    log_issue "Enum value $val is uppercase in schema but lowercase version used in code"
  fi
done

# 4. TABLE REFERENCE MISMATCHES
echo ""
echo "=== CHECKING TABLE REFERENCE MISMATCHES ==="

# Get all table variable names from schema
grep -oP "export const \K\w+(?= = pgTable)" shared/schema.ts > /tmp/schema_tables.txt

# Check each table reference in code
grep -rn "db\.\(query\|insert\|update\|delete\)" --include="*.ts" server/ | \
  grep -oP "\.\K\w+(?=\(|\s|\.)" | sort -u | while read table; do
  if ! grep -q "^$table$" /tmp/schema_tables.txt; then
    log_issue "Table '$table' referenced in code but not found in schema"
  fi
done

# 5. SUMMARY
echo ""
echo "========================================================"
echo "TOTAL ISSUES FOUND: $ISSUES_FOUND"
echo "========================================================"
echo "Full report saved to: $OUTPUT_FILE"

cat $OUTPUT_FILE
```

## 5.2 SQL Verification Queries

```sql
-- Run these queries to verify database matches expectations

-- 1. List all enums and their values
SELECT 
  t.typname AS enum_name,
  array_agg(e.enumlabel ORDER BY e.enumsortorder) AS enum_values
FROM pg_type t
JOIN pg_enum e ON t.oid = e.enumtypid
JOIN pg_catalog.pg_namespace n ON n.oid = t.typnamespace
WHERE n.nspname = 'public'
GROUP BY t.typname
ORDER BY t.typname;

-- 2. Find columns using enum types
SELECT 
  c.table_name,
  c.column_name,
  c.udt_name AS enum_type
FROM information_schema.columns c
WHERE c.table_schema = 'public'
  AND c.data_type = 'USER-DEFINED'
ORDER BY c.table_name, c.column_name;

-- 3. Find potential varchar columns that should be enums
SELECT 
  table_name,
  column_name,
  data_type
FROM information_schema.columns
WHERE table_schema = 'public'
  AND data_type = 'character varying'
  AND (column_name LIKE '%status%' 
       OR column_name LIKE '%state%'
       OR column_name LIKE '%type%'
       OR column_name LIKE '%role%')
ORDER BY table_name, column_name;

-- 4. Check for missing columns referenced in common patterns
SELECT 
  table_name,
  CASE WHEN EXISTS (
    SELECT 1 FROM information_schema.columns c2 
    WHERE c2.table_name = c.table_name AND c2.column_name = 'entity_type'
  ) THEN 'YES' ELSE 'NO' END AS has_entity_type,
  CASE WHEN EXISTS (
    SELECT 1 FROM information_schema.columns c2 
    WHERE c2.table_name = c.table_name AND c2.column_name = 'entity_id'
  ) THEN 'YES' ELSE 'NO' END AS has_entity_id
FROM information_schema.tables c
WHERE table_schema = 'public'
  AND table_type = 'BASE TABLE'
  AND table_name LIKE '%audit%' OR table_name LIKE '%log%'
ORDER BY table_name;
```
