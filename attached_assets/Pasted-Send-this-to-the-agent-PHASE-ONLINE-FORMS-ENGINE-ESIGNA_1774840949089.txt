import { useRef, useState, useCallback } from 'react';
import SignatureCanvas from 'react-signature-canvas';
import { Pen, Type, RotateCcw, Check } from 'lucide-react';

interface SignatureCaptureProps {
  label?: string;
  required?: boolean;
  onCapture: (signature: {
    type: 'drawn' | 'typed';
    data: string; // base64 for drawn, string for typed
    typedName?: string;
    capturedAt: string;
  }) => void;
  onClear?: () => void;
  legalText?: string;
}

export function SignatureCapture({
  label = 'Sign here',
  required = true,
  onCapture,
  onClear,
  legalText = 'By signing, you agree this is your legal electronic signature per the ESIGN Act and UETA.'
}: SignatureCaptureProps) {
  const sigPad = useRef<SignatureCanvas>(null);
  const [mode, setMode] = useState<'draw' | 'type'>('draw');
  const [typedName, setTypedName] = useState('');
  const [captured, setCaptured] = useState(false);
  const [capturedData, setCapturedData] = useState<string | null>(null);

  const handleCapture = useCallback(() => {
    if (mode === 'draw') {
      if (!sigPad.current || sigPad.current.isEmpty()) {
        alert('Please draw your signature before confirming.');
        return;
      }
      const dataUrl = sigPad.current.toDataURL('image/png');
      const signatureData = {
        type: 'drawn' as const,
        data: dataUrl,
        capturedAt: new Date().toISOString()
      };
      setCapturedData(dataUrl);
      setCaptured(true);
      onCapture(signatureData);
    }

    if (mode === 'type') {
      if (!typedName.trim() || typedName.trim().length < 3) {
        alert('Please type your full legal name.');
        return;
      }
      // Render typed name as base64 image using canvas
      const canvas = document.createElement('canvas');
      canvas.width = 400;
      canvas.height = 100;
      const ctx = canvas.getContext('2d')!;
      ctx.fillStyle = 'white';
      ctx.fillRect(0, 0, 400, 100);
      ctx.fillStyle = '#1a1a2e';
      ctx.font = '48px "Dancing Script", cursive';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.fillText(typedName, 200, 55);
      const dataUrl = canvas.toDataURL('image/png');
      const signatureData = {
        type: 'typed' as const,
        data: dataUrl,
        typedName: typedName.trim(),
        capturedAt: new Date().toISOString()
      };
      setCapturedData(dataUrl);
      setCaptured(true);
      onCapture(signatureData);
    }
  }, [mode, typedName, onCapture]);

  const handleClear = useCallback(() => {
    sigPad.current?.clear();
    setTypedName('');
    setCaptured(false);
    setCapturedData(null);
    onClear?.();
  }, [onClear]);

  return (
    <div className="border-2 border-dashed border-gray-300 rounded-xl p-4 bg-gray-50">
      <div className="flex items-center justify-between mb-3">
        <label className="text-sm font-semibold text-gray-700">
          {label}
          {required && <span className="text-red-500 ml-1">*</span>}
        </label>

        {/* Mode toggle */}
        <div className="flex gap-1 bg-white border rounded-lg p-1">
          <button
            type="button"
            onClick={() => { setMode('draw'); handleClear(); }}
            className={`flex items-center gap-1 px-3 py-1 rounded text-xs font-medium transition-colors ${
              mode === 'draw' 
                ? 'bg-blue-600 text-white' 
                : 'text-gray-600 hover:bg-gray-100'
            }`}
          >
            <Pen className="h-3 w-3" />
            Draw
          </button>
          <button
            type="button"
            onClick={() => { setMode('type'); handleClear(); }}
            className={`flex items-center gap-1 px-3 py-1 rounded text-xs font-medium transition-colors ${
              mode === 'type' 
                ? 'bg-blue-600 text-white' 
                : 'text-gray-600 hover:bg-gray-100'
            }`}
          >
            <Type className="h-3 w-3" />
            Type
          </button>
        </div>
      </div>

      {/* Signature area */}
      {captured && capturedData ? (
        <div className="relative bg-white border rounded-lg overflow-hidden">
          <img
            src={capturedData}
            alt="Your signature"
            className="w-full h-24 object-contain"
          />
          <div className="absolute top-2 right-2 bg-green-500 text-white rounded-full p-1">
            <Check className="h-4 w-4" />
          </div>
        </div>
      ) : (
        <>
          {mode === 'draw' && (
            <div className="bg-white border rounded-lg overflow-hidden touch-none">
              <SignatureCanvas
                ref={sigPad}
                canvasProps={{
                  className: 'w-full',
                  style: { height: '120px', width: '100%' }
                }}
                penColor="#1a1a2e"
                dotSize={2}
                minWidth={1}
                maxWidth={3}
              />
            </div>
          )}

          {mode === 'type' && (
            <div className="bg-white border rounded-lg p-3">
              <input
                type="text"
                value={typedName}
                onChange={e => setTypedName(e.target.value)}
                placeholder="Type your full legal name"
                className="w-full border-0 outline-none text-3xl text-center text-gray-800"
                style={{ fontFamily: '"Dancing Script", cursive' }}
              />
              {typedName && (
                <p className="text-xs text-center text-gray-400 mt-1">
                  This will appear as your signature
                </p>
              )}
            </div>
          )}
        </>
      )}

      {/* Action buttons */}
      <div className="flex gap-2 mt-3">
        <button
          type="button"
          onClick={handleClear}
          className="flex items-center gap-1 px-3 py-2 text-sm border rounded-lg hover:bg-gray-100 transition-colors"
        >
          <RotateCcw className="h-3 w-3" />
          Clear
        </button>
        {!captured && (
          <button
            type="button"
            onClick={handleCapture}
            className="flex-1 flex items-center justify-center gap-1 px-3 py-2 text-sm bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition-colors font-medium"
          >
            <Check className="h-3 w-3" />
            Confirm Signature
          </button>
        )}
        {captured && (
          <div className="flex-1 flex items-center justify-center gap-1 px-3 py-2 text-sm bg-green-50 text-green-700 rounded-lg border border-green-200 font-medium">
            <Check className="h-3 w-3" />
            Signature Confirmed
          </div>
        )}
      </div>

      {/* Legal text */}
      <p className="text-xs text-gray-400 mt-3 leading-relaxed">
        {legalText}
      </p>
    </div>
  );
}

import { useState, useEffect, useCallback } from 'react';
import { useParams } from 'react-router-dom';
import { useQuery, useMutation } from '@tanstack/react-query';
import { SignatureCapture } from '@/components/forms/SignatureCapture';
import { Loader2, CheckCircle, AlertCircle, Clock } from 'lucide-react';

// Field renderers for each type
function FormField({ field, value, onChange, errors }: any) {
  const baseClass = `w-full border rounded-lg px-4 py-3 text-sm focus:outline-none focus:ring-2 focus:ring-blue-500 ${
    errors[field.id] ? 'border-red-500' : 'border-gray-300'
  }`;

  switch (field.type) {
    case 'display':
      return (
        <div className="bg-gray-50 border border-gray-200 rounded-lg px-4 py-3 text-sm text-gray-700">
          {value || field.defaultValue || '—'}
        </div>
      );

    case 'text':
    case 'email':
    case 'phone':
      return (
        <input
          type={field.type}
          value={value || ''}
          onChange={e => onChange(field.id, e.target.value)}
          placeholder={field.placeholder}
          readOnly={field.readOnly}
          className={baseClass + (field.readOnly ? ' bg-gray-50 text-gray-500' : '')}
        />
      );

    case 'date':
      return (
        <input
          type="date"
          value={value || ''}
          onChange={e => onChange(field.id, e.target.value)}
          className={baseClass}
        />
      );

    case 'textarea':
      return (
        <textarea
          value={value || ''}
          onChange={e => onChange(field.id, e.target.value)}
          placeholder={field.placeholder}
          rows={4}
          className={baseClass}
        />
      );

    case 'select':
      return (
        <select
          value={value || ''}
          onChange={e => onChange(field.id, e.target.value)}
          className={baseClass}
        >
          <option value="">Select an option...</option>
          {field.options?.map((opt: string) => (
            <option key={opt} value={opt}>{opt}</option>
          ))}
        </select>
      );

    case 'radio':
      return (
        <div className="space-y-2">
          {field.options?.map((opt: string) => (
            <label key={opt} className="flex items-center gap-3 cursor-pointer">
              <input
                type="radio"
                name={field.id}
                value={opt}
                checked={value === opt}
                onChange={() => onChange(field.id, opt)}
                className="h-4 w-4 text-blue-600"
              />
              <span className="text-sm text-gray-700">{opt}</span>
            </label>
          ))}
        </div>
      );

    case 'checkbox':
      return (
        <label className="flex items-start gap-3 cursor-pointer">
          <input
            type="checkbox"
            checked={!!value}
            onChange={e => onChange(field.id, e.target.checked)}
            className="h-4 w-4 mt-0.5 text-blue-600 rounded"
          />
          <span className="text-sm text-gray-700">{field.checkboxLabel || field.label}</span>
        </label>
      );

    case 'ssn':
      return (
        <input
          type="password"
          value={value || ''}
          onChange={e => {
            const clean = e.target.value.replace(/\D/g, '').slice(0, 9);
            onChange(field.id, clean);
          }}
          placeholder="XXX-XX-XXXX"
          maxLength={9}
          className={baseClass}
          autoComplete="off"
        />
      );

    case 'currency':
      return (
        <div className="relative">
          <span className="absolute left-3 top-1/2 -translate-y-1/2 text-gray-500 text-sm">$</span>
          <input
            type="number"
            value={value || ''}
            onChange={e => onChange(field.id, e.target.value)}
            placeholder="0.00"
            step="0.01"
            min="0"
            className={baseClass + ' pl-7'}
          />
        </div>
      );

    case 'file':
      return (
        <div className="border-2 border-dashed border-gray-300 rounded-lg p-6 text-center">
          <input
            type="file"
            onChange={e => {
              const file = e.target.files?.[0];
              if (file) {
                const reader = new FileReader();
                reader.onload = () => onChange(field.id, {
                  name: file.name,
                  type: file.type,
                  size: file.size,
                  data: reader.result
                });
                reader.readAsDataURL(file);
              }
            }}
            accept={field.acceptedTypes || '*'}
            className="hidden"
            id={`file-${field.id}`}
          />
          <label
            htmlFor={`file-${field.id}`}
            className="cursor-pointer text-blue-600 hover:text-blue-700 text-sm font-medium"
          >
            {value?.name || 'Click to upload file'}
          </label>
          {field.helpText && (
            <p className="text-xs text-gray-400 mt-1">{field.helpText}</p>
          )}
        </div>
      );

    default:
      return (
        <input
          type="text"
          value={value || ''}
          onChange={e => onChange(field.id, e.target.value)}
          className={baseClass}
        />
      );
  }
}

export default function PublicFormPage() {
  const { token } = useParams<{ token: string }>();
  const [formData, setFormData] = useState<Record<string, any>>({});
  const [signature, setSignature] = useState<any>(null);
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [submitted, setSubmitted] = useState(false);
  const [submitError, setSubmitError] = useState<string | null>(null);

  // Fetch form definition
  const { data: form, isLoading, error } = useQuery({
    queryKey: ['public-form', token],
    queryFn: async () => {
      const res = await fetch(`/api/forms/public/${token}`);
      if (!res.ok) {
        const err = await res.json();
        throw new Error(err.error || 'Form not found');
      }
      return res.json();
    },
    retry: false
  });

  // Pre-populate form with invitation data
  useEffect(() => {
    if (form?.pre_populated_data) {
      setFormData(prev => ({ ...prev, ...form.pre_populated_data }));
    }
  }, [form]);

  const handleFieldChange = useCallback((fieldId: string, value: any) => {
    setFormData(prev => ({ ...prev, [fieldId]: value }));
    setErrors(prev => ({ ...prev, [fieldId]: '' }));
  }, []);

  const validateForm = useCallback(() => {
    const newErrors: Record<string, string> = {};
    let isValid = true;

    form?.fields?.forEach((field: any) => {
      if (field.required && field.type !== 'display') {
        const value = formData[field.id];
        if (value === undefined || value === null || value === '') {
          newErrors[field.id] = `${field.label} is required`;
          isValid = false;
        }
      }

      // Email validation
      if (field.type === 'email' && formData[field.id]) {
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!emailRegex.test(formData[field.id])) {
          newErrors[field.id] = 'Please enter a valid email address';
          isValid = false;
        }
      }

      // SSN validation
      if (field.type === 'ssn' && field.required && formData[field.id]) {
        if (formData[field.id].length !== 9) {
          newErrors[field.id] = 'Please enter your full 9-digit SSN';
          isValid = false;
        }
      }
    });

    // Signature validation
    if (form?.requires_signature && !signature) {
      newErrors['signature'] = 'Signature is required';
      isValid = false;
    }

    setErrors(newErrors);
    return isValid;
  }, [form, formData, signature]);

  const submitMutation = useMutation({
    mutationFn: async () => {
      const res = await fetch(`/api/forms/public/${token}/submit`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          data: formData,
          signatureData: signature?.data || null,
          signatureType: signature?.type || null,
          typedName: signature?.typedName || null,
          deviceType: /Mobile|Tablet/i.test(navigator.userAgent) ? 'mobile' : 'desktop'
        })
      });
      if (!res.ok) {
        const err = await res.json();
        throw new Error(err.error || 'Submission failed');
      }
      return res.json();
    },
    onSuccess: () => {
      setSubmitted(true);
      window.scrollTo(0, 0);
    },
    onError: (err: Error) => {
      setSubmitError(err.message);
    }
  });

  const handleSubmit = useCallback(() => {
    if (validateForm()) {
      submitMutation.mutate();
    } else {
      // Scroll to first error
      const firstError = document.querySelector('[data-error="true"]');
      firstError?.scrollIntoView({ behavior: 'smooth', block: 'center' });
    }
  }, [validateForm, submitMutation]);

  // Loading state
  if (isLoading) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="text-center">
          <Loader2 className="h-8 w-8 animate-spin text-blue-600 mx-auto mb-3" />
          <p className="text-gray-600 text-sm">Loading form...</p>
        </div>
      </div>
    );
  }

  // Error state
  if (error || !form) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4">
        <div className="bg-white rounded-2xl shadow-lg p-8 max-w-md w-full text-center">
          <AlertCircle className="h-12 w-12 text-red-500 mx-auto mb-4" />
          <h2 className="text-xl font-bold text-gray-900 mb-2">Form Not Available</h2>
          <p className="text-gray-500 text-sm">
            {(error as Error)?.message || 'This form may have expired or is no longer available.'}
          </p>
        </div>
      </div>
    );
  }

  // Success state
  if (submitted) {
    return (

          {form.branding?.companyName && ` Managed by ${form.branding.companyName}.`}

  );

async function processFormSubmissionAsync(submission: any, invitation: any) {
  try {
    // Update status to processing
    await pool.query(
      'UPDATE form_submissions SET trinity_processing_status = $1 WHERE id = $2',
      ['processing', submission.id]

      'UPDATE form_submissions SET generated_document_url = $1, processed_at = NOW(), trinity_processing_status = $2 WHERE id = $3',
      [pdfUrl, 'complete', submission.id]
    );

    // 3. Handle submit_action
    switch (invitation.submit_action) {
      case 'complete_onboarding_task':
        await completeOnboardingTaskFromForm(submission, invitation);
        break;
      case 'create_candidate':
        await createCandidateFromApplication(submission, invitation);
        break;
      case 'advance_signing_sequence':
        await advanceSigningSequence(submission, invitation);
        break;
      case 'execute_proposal':
        await executeProposalFromSignature(submission, invitation);
        break;
    }

    // 4. Notify org about submission
    await notificationDeliveryService.send({
      workspaceId: invitation.workspace_id,
      type: 'form_submitted',
      data: {
        formTitle: invitation.form_title,
        submittedBy: submission.submitted_by_name || submission.submitted_by_email,
        documentUrl: pdfUrl,
        contextType: invitation.context_type,
        contextId: invitation.context_id
      },
      targetRoles: ['org_owner', 'org_admin']
    });

    // 5. Broadcast to workspace
    broadcastToWorkspace(invitation.workspace_id, {
      type: 'form_submission_received',
      data: {
        formType: invitation.form_type,
        submittedBy: submission.submitted_by_name,
        documentUrl: pdfUrl
      }
    });

    // 6. Send confirmation to submitter
    if (submission.submitted_by_email) {
      await notificationDeliveryService.send({
        workspaceId: invitation.workspace_id,
        type: 'form_submission_confirmation',
        toEmail: submission.submitted_by_email,
        data: {
          recipientName: submission.submitted_by_name,
          formTitle: invitation.form_title,
          documentUrl: pdfUrl,
          companyName: invitation.company_name
        }
      });
    }

  } catch (err) {
    await pool.query(
      'UPDATE form_submissions SET trinity_processing_status = $1 WHERE id = $2',
      ['failed', submission.id]
    );
    throw err;
  }
}

async function completeOnboardingTaskFromForm(submission: any, invitation: any) {
  if (!invitation.context_id) return;
  
  // Find matching onboarding task
  const task = await pool.query(`
    UPDATE onboarding_tasks 
    SET status = 'completed', completed_at = NOW()
    WHERE employee_id = $1 
    AND workspace_id = $2
    AND task_type = $3
    AND status = 'pending'
    RETURNING id
  `, [invitation.context_id, invitation.workspace_id, invitation.form_type]);

  if (task.rows[0]) {
    // Check if all Tier 1 tasks complete — activate officer
    const pendingTier1 = await pool.query(`
      SELECT COUNT(*) FROM onboarding_tasks 
      WHERE employee_id = $1 AND tier = 1 AND is_blocking = true AND status != 'completed'
    `, [invitation.context_id]);

    if (parseInt(pendingTier1.rows[0].count) === 0) {
      // Auto-activate officer
      await pool.query(
        'UPDATE employees SET status = $1 WHERE id = $2 AND status = $3',
        ['active', invitation.context_id, 'pending']
      );
      
      broadcastToWorkspace(invitation.workspace_id, {
        type: 'employee_activated',
        data: { employeeId: invitation.context_id }
      });
    }
  }
}

```typescript
import PDFDocument from 'pdfkit';
import { createCanvas } from 'canvas';
import crypto from 'crypto';
import { pool } from '../db';
import { documentVaultService } from './documentVaultService';

export class FormToPdfService {

  async generateFromSubmission(submission: any, invitation: any): Promise<string> {
    const workspace = await this.getWorkspace(invitation.workspace_id);
    const form = await this.getForm(submission.form_id);
    
    // Generate PDF based on form type
    let pdfBuffer: Buffer;
    
    switch (form.form_type) {
      case 'offer_letter':
        pdfBuffer = await this.generateOfferLetterPdf(submission, form, workspace);
        break;
      case 'onboarding_w4':
        pdfBuffer = await this.generateW4Pdf(submission, workspace);
        break;
      case 'onboarding_i9_section1':
        pdfBuffer = await this.generateI9Pdf(submission, workspace);
        break;
      case 'onboarding_direct_deposit':
        pdfBuffer = await this.generateDirectDepositPdf(submission, workspace);
        break;
      case 'handbook_acknowledgment':
        pdfBuffer = await this.generateAcknowledgmentPdf(submission, form, workspace);
        break;
      case 'client_service_agreement':
        pdfBuffer = await this.generateServiceAgreementPdf(submission, form, workspace);
        break;
      case 'sales_proposal':
        pdfBuffer = await this.generateProposalAcceptancePdf(submission, form, workspace);
        break;
      default:
        pdfBuffer = await this.generateGenericFormPdf(submission, form, workspace);
    }

    // Add signature to PDF if present
    if (submission.signature_data) {
      pdfBuffer = await this.embedSignatureIntoPdf(pdfBuffer, submission);
    }

    // Calculate integrity hash
    const hash = crypto.createHash('sha256').update(pdfBuffer).digest('hex');

    // Store in document vault
    const stored = await documentVaultService.storeBuffer({
      buffer: pdfBuffer,
      filename: `${form.form_type}_${submission.id}.pdf`,
      mimeType: 'application/pdf',
      workspaceId: invitation.workspace_id,
      employeeId: invitation.context_type === 'employee' ? invitation.context_id : null,
      clientId: invitation.context_type === 'client' ? invitation.context_id : null,
      documentType: form.form_type,
      integrityHash: hash,
      submissionId: submission.id
    });

    return stored.url;
  }

  async generateGenericFormPdf(submission: any, form: any, workspace: any): Promise<Buffer> {
    return new Promise((resolve, reject) => {
      const chunks: Buffer[] = [];
      const doc = new PDFDocument({ margin: 60 });
      
      doc.on('data', chunk => chunks.push(chunk));
      doc.on('end', () => resolve(Buffer.concat(chunks)));
      doc.on('error', reject);

      const data = submission.data;

      // Header
      doc.fontSize(16).font('Helvetica-Bold').text(workspace.company_name, { align: 'center' });
      doc.moveDown();
      doc.fontSize(14).text(form.title, { align: 'center' });
      doc.moveDown(2);

      // Submission metadata
      doc.fontSize(10).font('Helvetica').fillColor('#666666');
      doc.text(`Submitted: ${new Date(submission.submitted_at).toLocaleString()}`);
      doc.text(`By: ${submission.submitted_by_name || submission.submitted_by_email || 'Unknown'}`);
      doc.fillColor('#000000');
      doc.moveDown();

      // Form fields and responses
      form.fields?.forEach((field: any) => {
        if (field.type === 'display') return;
        const value = data[field.id];
        if (value === undefined || value === null || value === '') return;

        doc.font('Helvetica-Bold').fontSize(10).text(field.label + ':');
        doc.font('Helvetica').fontSize(10);
        
        if (typeof value === 'boolean') {
          doc.text(value ? 'Yes' : 'No');
        } else if (field.type === 'file') {
          doc.text(`[File: ${value.name || 'Attached'}]`);
        } else if (field.type === 'ssn') {
          doc.text(`XXX-XX-${String(value).slice(-4)}`); // Masked SSN
        } else {
          doc.text(String(value));
        }
        doc.moveDown(0.5);
      });

      // Signature block
      if (submission.signature_data) {
        doc.moveDown();
        doc.font('Helvetica-Bold').text('Electronic Signature:');
        doc.moveDown(0.5);
        // Signature will be embedded by embedSignatureIntoPdf
        doc.rect(60, doc.y, 200, 60).stroke();
        doc.moveDown(4);
        doc.font('Helvetica').fontSize(9).fillColor('#666666');
        doc.text(`Signed electronically by ${submission.submitted_by_name || submission.submitted_by_email}`);
        doc.text(`IP Address: ${submission.ip_address}`);
        doc.text(`Timestamp: ${new Date(submission.submitted_at).toISOString()}`);
        doc.fillColor('#000000');
      }

      this.addDocumentFooter(doc, submission, workspace);
      doc.end();
    });
  }

  async embedSignatureIntoPdf(pdfBuffer: Buffer, submission: any): Promise<Buffer> {
    // If signature_data is a base64 PNG, embed it into the PDF
    // Using pdf-lib for this operation
    try {
      const { PDFDocument, rgb } = await import('pdf-lib');
      const pdfDoc = await PDFDocument.load(pdfBuffer);
      const pages = pdfDoc.getPages();
      const lastPage = pages[pages.length - 1];

      if (submission.signature_data && submission.signature_data.startsWith('data:image')) {
        const base64Data = submission.signature_data.split(',')[1];
        const imageBytes = Buffer.from(base64Data, 'base64');
        const image = await pdfDoc.embedPng(imageBytes);
        
        // Place signature in signature area
        const { width, height } = lastPage.getSize();
        lastPage.drawImage(image, {
          x: 60,
          y: height - 400, // Adjust based on document layout
          width: 200,
          height: 60,
        });
      } else if (submission.signature_type === 'typed' && submission.typed_name) {
        // Add typed name as text in cursive-style
        lastPage.drawText(submission.typed_name, {
          x: 60,
          y: lastPage.getSize().height - 380,
          size: 28,
          color: rgb(0.1, 0.1, 0.5),
        });
      }

      const finalPdfBytes = await pdfDoc.save();
      return Buffer.from(finalPdfBytes);
    } catch (err) {
      // If signature embedding fails return original PDF
      console.error('Signature embedding failed:', err);
      return pdfBuffer;
    }
  }

  addDocumentFooter(doc: any, submission: any, workspace: any) {
    doc.fontSize(8).fillColor('#666666');
    doc.text('─'.repeat(80), { align: 'center' });
    doc.text(
      `This document was electronically submitted on ${new Date(submission.submitted_at).toLocaleString()} UTC | ` +
      `IP: ${submission.ip_address} | ` +
      `${workspace.company_name}`,
      { align: 'center' }
    );
    if (submission.id) {
      doc.text(`Submission Reference: ${submission.id}`, { align: 'center' });
    }
    doc.text(
      'Electronic signatures are legally binding under the ESIGN Act (15 U.S.C. § 7001) and UETA.',
      { align: 'center' }
    );
  }

  async getWorkspace(workspaceId: string) {
    const result = await pool.query('SELECT * FROM workspaces WHERE id = $1', [workspaceId]);
    return result.rows[0];
  }

  async getForm(formId: string) {
    const result = await pool.query('SELECT * FROM platform_forms WHERE id = $1', [formId]);
    return result.rows[0];
  }
}

            )}
          </div>
        </div>
      </div>

      <div className="max-w-2xl mx-auto px-4 py-6 space-y-4">

        {/* Services */}
        {content.services?.length > 0 && (
          <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-5">
            <div className="flex items-center gap-2 mb-4">
              <Shield className="h-5 w-5 text-blue-600" />
              <h2 className="font-bold text-gray-900">Security Services</h2>
            </div>
            <div className="space-y-3">
              {content.services.map((service: any, i: number) => (
                <div key={i} className="flex justify-between items-start py-2 border-b border-gray-50 last:border-0">
                  <div>
                    <p className="font-medium text-sm text-gray-800">{service.description}</p>
                    <p className="text-xs text-gray-500">{service.details}</p>
                  </div>
                  <p className="font-semibold text-sm text-gray-900 ml-4 whitespace-nowrap">
                    ${parseFloat(service.amount).toFixed(2)}/mo
                  </p>
                </div>
              ))}
            </div>
          </div>
        )}

        {/* Pricing Summary */}
        <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-5">
          <div className="flex items-center gap-2 mb-4">
            <DollarSign className="h-5 w-5 text-green-600" />
            <h2 className="font-bold text-gray-900">Pricing Summary</h2>
          </div>
          <div className="space-y-2">
            {content.pricing?.lineItems?.map((item: any, i: number) => (
              <div key={i} className="flex justify-between text-sm">
                <span className="text-gray-600">{item.label}</span>
                <span className="font-medium">${parseFloat(item.amount).toFixed(2)}</span>
              </div>
            ))}
            <div className="pt-2 mt-2 border-t border-gray-200 flex justify-between font-bold">
              <span>Total Monthly Investment</span>
              <span className="text-green-600 text-lg">
                ${parseFloat(proposal.total_monthly_value || 0).toFixed(2)}/mo
              </span>
            </div>
          </div>
        </div>

        {/* Terms */}
        {content.terms && (
          <div className="bg-blue-50 border border-blue-200 rounded-xl p-4">
            <h2 className="font-semibold text-blue-900 mb-2 text-sm">Key Terms</h2>
            <p className="text-sm text-blue-800 leading-relaxed">{content.terms}</p>
          </div>
        )}

        {/* Action area */}
        {!action ? (
          <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-5">
            <h2 className="font-bold text-gray-900 mb-4 text-center">How would you like to proceed?</h2>
            <div className="space-y-3">
              <button
                onClick={() => setAction('accept')}
                className="w-full py-4 bg-green-600 text-white rounded-xl font-bold text-base hover:bg-green-700 transition-colors flex items-center justify-center gap-2"
              >
                <CheckCircle className="h-5 w-5" />
                Accept This Proposal
              </button>
              <button
                onClick={() => setAction('changes')}
                className="w-full py-3 border-2 border-blue-600 text-blue-600 rounded-xl font-semibold hover:bg-blue-50 transition-colors flex items-center justify-center gap-2"
              >
                <MessageSquare className="h-4 w-4" />
                Request Changes
              </button>
              <button
                onClick={() => setAction('decline')}
                className="w-full py-3 border border-gray-300 text-gray-500 rounded-xl text-sm hover:bg-gray-50 transition-colors"
              >
                Decline Proposal
              </button>
            </div>
          </div>
        ) : action === 'accept' ? (
          <div className="space-y-4">
            <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-5">
              <h2 className="font-bold text-gray-900 mb-2">Sign to Accept</h2>
              <p className="text-sm text-gray-500 mb-4">
                By signing below you agree to the terms of this proposal and authorize {content.companyName} to begin providing security services as described.
              </p>
              <SignatureCapture
                label="Your Signature"
                required={true}
                onCapture={setSignature}
                onClear={() => setSignature(null)}
                legalText={`By signing, ${content.clientName || 'you'} agree to the services and pricing outlined in this proposal. This constitutes a binding service agreement.`}
              />
            </div>
            <div className="flex gap-3">
              <button
                onClick={() => { setAction(null); setSignature(null); }}
                className="flex-1 py-3 border rounded-xl text-gray-600 hover:bg-gray-50"
              >
                Back
              </button>
              <button
                onClick={handleAccept}
                disabled={!signature || actionMutation.isPending}
                className="flex-2 py-3 bg-green-600 text-white rounded-xl font-bold disabled:opacity-50 flex items-center justify-center gap-2 px-6"
              >
                {actionMutation.isPending ? <Loader2 className="h-4 w-4 animate-spin" /> : <CheckCircle className="h-4 w-4" />}
                Confirm Acceptance
              </button>
            </div>
          </div>
        ) : action === 'decline' ? (
          <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-5 space-y-4">
            <h2 className="font-bold text-gray-900">Decline Proposal</h2>
            <textarea
              value={declineReason}
              onChange={e => setDeclineReason(e.target.value)}
              placeholder="Please let us know why you are declining. This helps us improve our proposals..."
              rows={4}
              className="w-full border rounded-lg p-3 text-sm"
            />
            <div className="flex gap-3">
              <button onClick={() => setAction(null)} className="flex-1 py-3 border rounded-xl text-gray-600">Back</button>
              <button
                onClick={handleDecline}
                disabled={actionMutation.isPending}
                className="flex-1 py-3 bg-red-600 text-white rounded-xl font-semibold disabled:opacity-50"
              >
                Decline Proposal
              </button>
            </div>
          </div>
        ) : action === 'changes' ? (
          <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-5 space-y-4">
            <h2 className="font-bold text-gray-900">Request Changes</h2>
            <p className="text-sm text-gray-500">Describe the changes you would like to see in a revised proposal.</p>
            <textarea
              value={changeRequests}
              onChange={e => setChangeRequests(e.target.value)}
              placeholder="e.g. We need 3 officers instead of 2 at Site A, and would like a 6-month contract term instead of 12..."
              rows={5}
              className="w-full border rounded-lg p-3 text-sm"
            />
            <div className="flex gap-3">
              <button onClick={() => setAction(null)} className="flex-1 py-3 border rounded-xl text-gray-600">Back</button>
              <button
                onClick={handleChanges}
                disabled={actionMutation.isPending}
                className="flex-1 py-3 bg-blue-600 text-white rounded-xl font-semibold disabled:opacity-50"
              >
                Send Request
              </button>
            </div>
          </div>
        ) : null}

        <p className="text-xs text-center text-gray-400 pb-8">
          This proposal is secured and confidential. {content.companyName}
        </p>
      </div>
    </div>
  );
}
```
