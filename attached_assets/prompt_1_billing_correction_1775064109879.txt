export function selectGeminiModel(callType: string): string {
  // Simple read-only queries → Flash-Lite (cheapest)
  const liteCallTypes = [
    'simple_lookup', 'faq_answer', 'schedule_check',
    'who_is_on_shift', 'status_check', 'read_only_query',
  ];

  // Complex reasoning → Pro (most capable)
  const proCallTypes = [
    'payroll_dispute', 'complex_financial_analysis',
    'multi_site_optimization', 'strategic_scheduling',
    'complex_compliance_analysis',
  ];

  if (liteCallTypes.includes(callType)) return 'gemini-2.5-flash-lite';
  if (proCallTypes.includes(callType)) return 'gemini-2.5-pro';
  return 'gemini-2.5-flash'; // default for everything else
}

export function selectClaudeModel(callType: string): string | null {
  // Skip Claude entirely for read-only queries
  const skipClaudeTypes = [
    'simple_lookup', 'faq_answer', 'read_only_query',
    'schedule_check', 'status_check', 'who_is_on_shift',
  ];

  // Deep validation for financial and legal
  const deepValidationTypes = [
    'claude_validation_financial', 'payroll_validation',
    'legal_document_review', 'compliance_certification',
  ];

  if (skipClaudeTypes.includes(callType)) return null; // skip Claude
  if (deepValidationTypes.includes(callType)) return 'claude-sonnet-4-6';
  return 'claude-haiku-4-5'; // fast validation for everything else
}

export function selectGptModel(callType: string): string {
  // Complex documents → GPT-4o
  const fullGptTypes = [
    'proposal_generation', 'legal_document',
    'detailed_contract', 'client_facing_document',
    'compliance_report',
  ];

  if (fullGptTypes.includes(callType)) return 'gpt-4o';
  return 'gpt-4o-mini'; // default for everything else
}
