# COAILEAGUE - COMPREHENSIVE RACE CONDITION AUDIT

## WHAT ARE RACE CONDITIONS?

Race conditions occur when **two or more operations** try to access/modify the **same resource** at the **same time**, and the outcome depends on the **order of execution**.

**IN YOUR PLATFORM, THIS MEANS:**
- Two users assign the same shift → Both succeed, data corrupted
- User double-clicks submit → Duplicate records created
- Clock in called twice quickly → Two time entries
- Two tabs update same employee → One update lost
- WebSocket reconnects during message send → Message lost or duplicated
- Session expires during API call → Partial operation completed

# PART 1: DATABASE RACE CONDITIONS

## 1.1 The Classic Read-Modify-Write Race

```typescript
// ❌ RACE CONDITION - Check then act
async function assignShift(shiftId: string, employeeId: string) {
  // Step 1: Read
  const shift = await db.query.shifts.findFirst({
    where: eq(shifts.id, shiftId)
  });
  
  // ⚠️ RACE WINDOW: Another request could assign between read and write!
  
  if (shift.assignedEmployeeId) {
    throw new Error('Shift already assigned');
  }
  
  // Step 2: Write
  await db.update(shifts)
    .set({ assignedEmployeeId: employeeId })
    .where(eq(shifts.id, shiftId));
}

// ✅ FIXED - Atomic operation with condition
async function assignShiftSafe(shiftId: string, employeeId: string) {
  const result = await db.update(shifts)
    .set({ assignedEmployeeId: employeeId })
    .where(and(
      eq(shifts.id, shiftId),
      isNull(shifts.assignedEmployeeId)  // Condition in WHERE
    ))
    .returning();
  
  if (result.length === 0) {
    throw new Error('Shift already assigned or not found');
  }
  
  return result[0];
}
```

## 2.1 Double Submission Race

```typescript
// ❌ RACE CONDITION - No idempotency
router.post('/api/employees', async (req, res) => {
  // User double-clicks, two requests arrive
  const employee = await db.insert(employees).values(req.body);
  // RESULT: Two identical employees created!
  res.json(employee);
});

// ✅ FIXED - Idempotency key
router.post('/api/employees', async (req, res) => {
  const idempotencyKey = req.headers['x-idempotency-key'];
  
  if (idempotencyKey) {
    // Check if we've seen this request before
    const existing = await redis.get(`idempotency:${idempotencyKey}`);
    if (existing) {
      return res.json(JSON.parse(existing));
    }
  }
  
  const employee = await db.insert(employees).values(req.body).returning();
  
  if (idempotencyKey) {
    // Store result for duplicate requests
    await redis.setex(`idempotency:${idempotencyKey}`, 3600, JSON.stringify(employee));
  }
  
  res.json(employee);
});
```

## 2.3 Concurrent Update Race

```typescript
// ❌ RACE CONDITION - Lost update
router.put('/api/employees/:id', async (req, res) => {
  // Two users editing same employee
  // User A reads version 1, User B reads version 1
  // User A saves, version becomes 2
  // User B saves with version 1 data, overwrites User A's changes!
  
  await db.update(employees)
    .set(req.body)
    .where(eq(employees.id, req.params.id));
});

// ✅ FIXED - Optimistic locking with version
router.put('/api/employees/:id', async (req, res) => {
  const { version, ...data } = req.body;
  
  const result = await db.update(employees)
    .set({ 
      ...data, 
      version: sql`version + 1`,
      updatedAt: new Date()
    })
    .where(and(
      eq(employees.id, req.params.id),
      eq(employees.version, version)  // Only update if version matches
    ))
    .returning();
  
  if (result.length === 0) {
    return res.status(409).json({
      error: 'CONFLICT',
      message: 'Record was modified by another user. Please refresh and try again.'
    });
  }
  
  res.json(result[0]);
});
```

# PART 4: WEBSOCKET RACE CONDITIONS

## 4.1 Connection State Race

```typescript
// ❌ RACE CONDITION - Send during reconnect
function useChat() {
  const socketRef = useRef(null);
  
  const sendMessage = (msg) => {
    // Socket might be reconnecting!
    socketRef.current.emit('message', msg);  // FAILS
  };
}

// ✅ FIXED - Queue messages during reconnect
function useChat() {
  const socketRef = useRef(null);
  const queueRef = useRef([]);
  const [connected, setConnected] = useState(false);
  
  useEffect(() => {
    const socket = io();
    socketRef.current = socket;
    
    socket.on('connect', () => {
      setConnected(true);
      // Flush queued messages
      while (queueRef.current.length > 0) {
        const msg = queueRef.current.shift();
        socket.emit('message', msg);
      }
    });
    
    socket.on('disconnect', () => setConnected(false));
    
    return () => socket.disconnect();
  }, []);
  
  const sendMessage = (msg) => {
    if (connected && socketRef.current?.connected) {
      socketRef.current.emit('message', msg);
    } else {
      queueRef.current.push(msg);  // Queue for later
    }
  };
  
  return { sendMessage, connected };
}
```

## 5.1 Overlapping Interval Race

```typescript
// ❌ RACE CONDITION - Intervals overlap
useEffect(() => {
  const interval = setInterval(async () => {
    // If fetch takes longer than interval, they overlap!
    await fetchUpdates();
  }, 5000);
  
  return () => clearInterval(interval);
}, []);

// ✅ FIXED - Wait for completion before next run
useEffect(() => {
  let timeoutId;
  let cancelled = false;
  
  const poll = async () => {
    if (cancelled) return;
    
    try {
      await fetchUpdates();
    } finally {
      if (!cancelled) {
        timeoutId = setTimeout(poll, 5000);  // Schedule AFTER completion
      }
    }
  };
  
  poll();
  
  return () => {
    cancelled = true;
    clearTimeout(timeoutId);
  };
}, []);
```

## 5.2 Debounce Race

```typescript
// ❌ RACE CONDITION - Multiple debounces running
function SearchInput() {
  const [query, setQuery] = useState('');
  
  const search = async (q) => {
    const results = await api.search(q);
    setResults(results);  // Older query might resolve after newer one!
  };
  
  useEffect(() => {
    const timer = setTimeout(() => search(query), 300);
    return () => clearTimeout(timer);
  }, [query]);
}

// ✅ FIXED - Track latest request
function SearchInput() {
  const [query, setQuery] = useState('');
  const latestRequestRef = useRef(0);
  
  useEffect(() => {
    const requestId = ++latestRequestRef.current;
    
    const timer = setTimeout(async () => {
      const results = await api.search(query);
      
      // Only update if this is still the latest request
      if (requestId === latestRequestRef.current) {
        setResults(results);
      }
    }, 300);
    
    return () => clearTimeout(timer);
  }, [query]);
}
```

## 8.2 Database-Specific Queries

```sql
-- Find tables without version column (need optimistic locking)
SELECT table_name
FROM information_schema.tables t
WHERE table_schema = 'public'
  AND table_type = 'BASE TABLE'
  AND NOT EXISTS (
    SELECT 1 FROM information_schema.columns c
    WHERE c.table_name = t.table_name
      AND c.column_name = 'version'
  )
ORDER BY table_name;

-- Find tables without unique constraints (duplicate risk)
SELECT t.table_name
FROM information_schema.tables t
WHERE t.table_schema = 'public'
  AND t.table_type = 'BASE TABLE'
  AND NOT EXISTS (
    SELECT 1 FROM information_schema.table_constraints tc
    WHERE tc.table_name = t.table_name
      AND tc.constraint_type = 'UNIQUE'
  )
ORDER BY t.table_name;
```

## 9.1 Transaction Template

```typescript
// Use for any multi-step database operation
async function multiStepOperation() {
  return await db.transaction(async (tx) => {
    // All operations use 'tx' not 'db'
    const step1 = await tx.insert(table1).values({...}).returning();
    const step2 = await tx.update(table2).set({...}).returning();
    const step3 = await tx.insert(table3).values({...}).returning();
    
    // If any step fails, ALL are rolled back
    return { step1, step2, step3 };
  });
}
```

## 9.2 Atomic Update Template

```typescript
// Use for any status/assignment change
async function atomicStatusChange(id: string, fromStatus: string, toStatus: string) {
  const result = await db.update(table)
    .set({ status: toStatus, updatedAt: new Date() })
    .where(and(
      eq(table.id, id),
      eq(table.status, fromStatus)  // Only if current status matches
    ))
    .returning();
  
  if (result.length === 0) {
    throw new ConflictError(`Cannot change status: current status is not '${fromStatus}'`);
  }
  
  return result[0];
}
```

## 9.3 Idempotent Insert Template

```typescript
// Use for any creation that could be duplicated
async function idempotentCreate(data: any, uniqueKey: string) {
  // Try to find existing
  const existing = await db.query.table.findFirst({
    where: eq(table.uniqueField, uniqueKey)
  });
  
  if (existing) {
    return existing;  // Return existing instead of creating duplicate
  }
  
  // Create with conflict handling
  const result = await db.insert(table)
    .values(data)
    .onConflictDoNothing()  // Or onConflictDoUpdate
    .returning();
  
  // If insert was skipped due to conflict, fetch existing
  if (result.length === 0) {
    return await db.query.table.findFirst({
      where: eq(table.uniqueField, uniqueKey)
    });
  }
  
  return result[0];
}
```

## 9.4 Frontend Async Handler Template

```typescript
// Use for any button that triggers async operation
function SafeAsyncButton({ onClick, children }) {
  const [loading, setLoading] = useState(false);
  const processingRef = useRef(false);
  
  const handleClick = async () => {
    // Immediate guard with ref
    if (processingRef.current) return;
    processingRef.current = true;
    setLoading(true);
    
    try {
      await onClick();
    } catch (error) {
      // Handle error
    } finally {
      processingRef.current = false;
      setLoading(false);
    }
  };
  
  return (
    <button 
      onClick={handleClick} 
      disabled={loading}
    >
      {loading ? <Spinner /> : children}
    </button>
  );
}
```
