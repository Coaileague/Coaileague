1. WebSocket Connection Architecture:
typescript// CURRENT (BROKEN): One connection per room
// User enters room → Connect to room's WebSocket
// User leaves room → Disconnect from WebSocket
// Result: Misses messages

// REQUIRED: Single persistent connection, multiple room subscriptions

// User navigates anywhere → Connection stays open
// Messages for ANY room come through same connection
// Result: Never misses messages
2. Connection Manager (New Service):
typescript// /client/src/services/chatConnectionManager.ts

class ChatConnectionManager {
  private socket: WebSocket | null = null;
  private subscribedRooms: Set<string> = new Set();
  private messageHandlers: Map<string, (message: ChatMessage) => void> = new Map();
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 10;
  
  // Initialize on app load (not on room enter)
  async initialize(userId: string, workspaceId: string) {
    if (this.socket?.readyState === WebSocket.OPEN) return;
    
    this.socket = new WebSocket(`wss://${host}/ws/chat`);
    
    this.socket.onopen = () => {
      this.reconnectAttempts = 0;
      // Authenticate
      this.send({ type: 'auth', userId, workspaceId });
      // Subscribe to all user's rooms
      this.subscribeToAllRooms();
    };
    
    this.socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleIncomingMessage(data);
    };
    
    this.socket.onclose = () => {
      this.handleDisconnect();
    };
  }
  
  // Called on app load - get user's rooms and subscribe
  async subscribeToAllRooms() {
    const rooms = await this.fetchUserRooms();
    rooms.forEach(room => {
      this.subscribedRooms.add(room.id);
      this.send({ type: 'subscribe', roomId: room.id });
    });
  }
  
  // Handle message for ANY room
  handleIncomingMessage(data: any) {
    if (data.type === 'message') {
      const { roomId, message } = data;
      
      // Update unread count (even if not viewing this room)
      this.incrementUnreadCount(roomId);
      
      // Trigger notification (even if not viewing this room)
      this.triggerNotification(roomId, message);
      
      // If there's a handler for this room (user is viewing it), call it
      const handler = this.messageHandlers.get(roomId);
      if (handler) {
        handler(message);
      }
      
      // Always update the room list preview
      this.updateRoomPreview(roomId, message);
    }
  }
  
  // User opens a specific room - register handler for live updates
  setActiveRoom(roomId: string, handler: (message: ChatMessage) => void) {
    this.messageHandlers.set(roomId, handler);
    // Mark messages as read
    this.markAsRead(roomId);
  }
  
  // User leaves room view - remove handler but STAY SUBSCRIBED
  clearActiveRoom(roomId: string) {
    this.messageHandlers.delete(roomId);
    // DO NOT unsubscribe - keep receiving messages
  }
  
  // Auto-reconnect on disconnect
  handleDisconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      setTimeout(() => this.initialize(), 1000 * this.reconnectAttempts);
    }
  }
  
  // Send message to any room
  sendMessage(roomId: string, content: string, attachments?: string[]) {
    this.send({
      type: 'message',
      roomId,
      content,
      attachments,
      timestamp: new Date().toISOString()
    });
  }
}

export const chatManager = new ChatConnectionManager();

7. Server-Side WebSocket Handler:
typescript// /server/websocket/chatHandler.ts

class ChatWebSocketServer {
  private connections: Map<string, WebSocket> = new Map(); // userId -> socket
  private userRooms: Map<string, Set<string>> = new Map(); // userId -> roomIds
  
  handleConnection(ws: WebSocket) {
    ws.on('message', (data) => {
      const message = JSON.parse(data);
      
      switch (message.type) {
        case 'auth':
          this.authenticateUser(ws, message.userId, message.workspaceId);
          break;
          
        case 'subscribe':
          this.subscribeToRoom(ws, message.userId, message.roomId);
          break;
          
        case 'message':
          this.broadcastToRoom(message.roomId, message);
          this.saveMessage(message);
          this.sendPushNotifications(message);
          break;
      }
    });
  }
  
  // Send to ALL users in a room, regardless of what they're viewing
  broadcastToRoom(roomId: string, message: any) {
    const roomMembers = await this.getRoomMembers(roomId);
    
    roomMembers.forEach(userId => {
      const socket = this.connections.get(userId);
      if (socket && socket.readyState === WebSocket.OPEN) {
        socket.send(JSON.stringify({
          type: 'message',
          roomId,
          message
        }));
      }
    });
  }
  
  // For offline users, queue and send push notification
  async sendPushNotifications(message: any) {
    const roomMembers = await this.getRoomMembers(message.roomId);
    
    for (const userId of roomMembers) {
      const socket = this.connections.get(userId);
      
      // User not connected - send push notification
      if (!socket || socket.readyState !== WebSocket.OPEN) {
        await this.sendPushNotification(userId, message);
      }
    }
  }
}

PART 3: DATABASE SCHEMA FOR CHAT PERSISTENCE
sql-- Ensure these tables exist and have correct structure

-- Chat rooms
CREATE TABLE IF NOT EXISTS chat_rooms (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  workspace_id UUID NOT NULL REFERENCES workspaces(id),
  name VARCHAR(255) NOT NULL,
  type VARCHAR(50) NOT NULL DEFAULT 'group', -- 'group', 'direct', 'broadcast'
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Room memberships (who is in what rooms)
CREATE TABLE IF NOT EXISTS chat_room_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  room_id UUID NOT NULL REFERENCES chat_rooms(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES users(id),
  workspace_id UUID NOT NULL,
  joined_at TIMESTAMP DEFAULT NOW(),
  last_read_at TIMESTAMP DEFAULT NOW(), -- For unread tracking
  is_muted BOOLEAN DEFAULT FALSE,
  role VARCHAR(50) DEFAULT 'member', -- 'admin', 'member'
  UNIQUE(room_id, user_id)
);

-- Messages
CREATE TABLE IF NOT EXISTS chat_messages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  room_id UUID NOT NULL REFERENCES chat_rooms(id) ON DELETE CASCADE,
  sender_id UUID NOT NULL REFERENCES users(id),
  workspace_id UUID NOT NULL,
  content TEXT NOT NULL,
  attachments JSONB DEFAULT '[]', -- [{url, type, name}]
  reply_to_id UUID REFERENCES chat_messages(id), -- For threading
  created_at TIMESTAMP DEFAULT NOW(),
  edited_at TIMESTAMP,
  is_deleted BOOLEAN DEFAULT FALSE
);

-- Read receipts (optional, for "seen by")
CREATE TABLE IF NOT EXISTS chat_read_receipts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  message_id UUID NOT NULL REFERENCES chat_messages(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES users(id),
  read_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(message_id, user_id)
);

-- Push notification subscriptions
CREATE TABLE IF NOT EXISTS push_subscriptions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id),
  workspace_id UUID NOT NULL,
  endpoint TEXT NOT NULL,
  keys JSONB NOT NULL, -- {p256dh, auth}
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(user_id, endpoint)
);

-- Indexes for performance
CREATE INDEX idx_chat_messages_room_created ON chat_messages(room_id, created_at DESC);
CREATE INDEX idx_chat_room_members_user ON chat_room_members(user_id);
CREATE INDEX idx_chat_room_members_room ON chat_room_members(room_id);
