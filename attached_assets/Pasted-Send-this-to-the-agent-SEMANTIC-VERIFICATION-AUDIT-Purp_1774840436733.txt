**Q2: Are payroll calculations isolated per officer — no cross-contamination?**

```bash
# Run payroll for ACME
curl -s -X POST http://localhost:5000/api/payroll/runs \
  -H "Authorization: Bearer [token]" \
  -d '{"periodStart":"2026-03-15","periodEnd":"2026-03-29","workspaceId":"[acme_id]"}'

# Verify each officer's pay stub shows ONLY their hours
SELECT 
  pr.id as payroll_run_id,
  prd.employee_id,
  prd.regular_hours,
  prd.overtime_hours,
  prd.gross_pay,
  prd.net_pay,
  -- Verify hours match actual time entries
  (SELECT SUM(EXTRACT(EPOCH FROM (clock_out - clock_in))/3600)
   FROM time_entries 
   WHERE employee_id = prd.employee_id
   AND clock_in BETWEEN '2026-03-15' AND '2026-03-29') as actual_hours
FROM payroll_run_details prd
JOIN payroll_runs pr ON pr.id = prd.payroll_run_id
WHERE pr.workspace_id = '[acme_id]'
ORDER BY prd.employee_id;
-- regular_hours must match actual_hours for each officer
-- Any discrepancy = payroll calculation bug — fix

# Verify officers from other workspaces are NOT included
SELECT COUNT(*) FROM payroll_run_details prd
JOIN payroll_runs pr ON pr.id = prd.payroll_run_id
JOIN employees e ON e.id = prd.employee_id
WHERE pr.workspace_id = '[acme_id]'
AND e.workspace_id != '[acme_id]';
-- Must be 0 — no cross-workspace contamination
```
