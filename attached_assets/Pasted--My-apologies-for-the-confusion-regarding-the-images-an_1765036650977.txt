const canvas = document.getElementById('trinityMascot');
const ctx = canvas.getContext('2d');
const W = canvas.width;
const H = canvas.height;
const CENTER_X = W / 2;
const CENTER_Y = H / 2;
const R_OUTER = 160; 
const R_INNER = 50;  
const NUM_POINTS = 5;

let currentJitter = 0;

// --- DRAWING THE KNOT ---

// 1. Calculate the 10 core points (5 outer, 5 inner)
function getKnotPoints(outerR, innerR, jitter) {
    let points = [];
    for (let i = 0; i < 2 * NUM_POINTS; i++) {
        const radius = (i % 2 === 0) ? outerR : innerR;
        const angle = (Math.PI / NUM_POINTS) * i - (Math.PI / 2);

        // Apply time-based breathing pulse (Micro-mutation)
        const pulse = 3 * Math.sin(time * 0.05); 
        
        points.push({
            x: CENTER_X + (radius + pulse + (Math.random() * jitter)) * Math.cos(angle),
            y: CENTER_Y + (radius + pulse + (Math.random() * jitter)) * Math.sin(angle)
        });
    }
    return points;
}

// 2. Function to draw segments with overlap control
function drawKnot(points) {
    // This function must draw overlapping segments in layers (Z-order)
    // Example simplified drawing order for the 5-point knot (A-B-C-D-E segments)
    const segmentOrder = [
        [0, 1, COLORS.TEAL], 
        [2, 3, COLORS.GOLD],
        [4, 5, COLORS.TEAL],
        [6, 7, COLORS.GOLD],
        [8, 9, COLORS.TEAL],
        // The remaining segments need to be drawn over the top of the first set:
        [1, 2, COLORS.GOLD], 
        [3, 4, COLORS.TEAL],
        [5, 6, COLORS.GOLD],
        [7, 8, COLORS.TEAL],
        [9, 0, COLORS.GOLD]
    ];
    
    segmentOrder.forEach(([startIdx, endIdx, color]) => {
        // A real implementation would use ctx.bezierCurveTo() for smooth ribbons
        ctx.beginPath();
        ctx.strokeStyle = color;
        ctx.lineWidth = 20;
        ctx.lineCap = 'round';
        ctx.moveTo(points[startIdx].x, points[startIdx].y);
        ctx.lineTo(points[endIdx].x, points[endIdx].y);
        ctx.stroke();
    });
}

// 3. Draw Digital Overlay/Data Flow
function drawDataFlow(points) {
    ctx.fillStyle = getCssVar('--data-flow-color');
    
    // Draw small squares moving along the ribbons
    points.forEach((p, i) => {
        // Calculate a point slightly offset from the main ribbon point
        const offset = 20 * Math.sin(time * 0.2 + i); 
        ctx.fillRect(p.x + offset, p.y - 10, 4, 4); // Example square particle
        ctx.fillRect(p.x - 10, p.y + offset, 2, 2); // Smaller particle
    });
}

// --- 4. ANIMATION LOOP ---

function animate() {
    time += 1;
    
    // Clear the canvas with the background color
    ctx.fillStyle = getCssVar('--bg-dark');
    ctx.fillRect(0, 0, W, H);

    // Get current points with subtle breathing
    const points = getKnotPoints(R_OUTER, R_INNER, currentJitter);

    drawKnot(points); 
    drawDataFlow(points);
    
    // Draw Central Core Glow
    const coreGradient = ctx.createRadialGradient(CENTER_X, CENTER_Y, 5, CENTER_X, CENTER_Y, 40);
    coreGradient.addColorStop(0, getCssVar('--core-light'));
    coreGradient.addColorStop(1, 'rgba(0,0,0,0)'); 
    ctx.fillStyle = coreGradient;
    ctx.fillRect(CENTER_X - 40, CENTER_Y - 40, 80, 80);

    requestAnimationFrame(animate);
}

// Start the animation
animate();

IV. Animation & Overlay Control
The key to easy manipulation is the window.trinitySetState function:

Breathing (Default): The 3 * Math.sin(time * 0.05) in getKnotPoints handles the subtle, default "breathing" or pulsing.

Frantic Swarm/Processing (Jitter Overlay): When your platform runs a heavy AI workflow, call trinitySetState('JITTER', 5). This adds random noise to the geometry, making the mascot appear frantic and active.

Color Change/Warning (Theming Overlay): For critical alerts, modify the CSS variables directly using JavaScript.

JavaScript
// Example JS to trigger a critical alert color theme
function triggerCriticalAlert() {
    document.documentElement.style.setProperty('--ribbon-teal', '#FF0000'); // Red
    document.documentElement.style.setProperty('--ribbon-gold', '#FFA07A'); // Light Red
    trinitySetState('JITTER', 4); // Frantic movement
}
