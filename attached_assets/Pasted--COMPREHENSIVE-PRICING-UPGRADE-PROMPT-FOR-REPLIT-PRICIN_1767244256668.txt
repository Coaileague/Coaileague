async function handleEmployeeCountChange(
  workspaceId: string, 
  newCount: number
) {
  const subscription = await getSubscription(workspaceId);
  const tier = PRICING_TIERS[subscription.tier];
  
  const overageEmployees = Math.max(0, newCount - tier.includedEmployees);

  await stripe.subscriptionItems.createUsageRecord(

      timestamp: Math.floor(Date.now() / 1000),
      action: 'set' // Replace previous value

  );

    employeeCount: newCount,
    overageEmployees,
    overageCharge: overageEmployees * tier.overagePrice,
    totalMonthlyCharge: tier.basePrice + (overageEmployees * tier.overagePrice)
  });

typescript// When employee clocks in
async function validateClockIn(employeeId: string, latitude: number, longitude: number) {
  // Get assigned shift for this employee right now
  const currentShift = await getCurrentShift(employeeId);
  if (!currentShift) {
    return {
      allowed: false,
      reason: 'No shift scheduled for this time'
    };
  }
  
  // Get client site location
  const site = await getSiteLocation(currentShift.siteId);
  
  // Calculate distance using Haversine formula
  const distanceInMeters = calculateDistance(
    { lat: latitude, lng: longitude },
    { lat: site.latitude, lng: site.longitude }
  );
  
  // Geofence radius: 100 meters (328 feet)
  const GEOFENCE_RADIUS = 100;
  
  if (distanceInMeters > GEOFENCE_RADIUS) {
    // FRAUD ATTEMPT - not at correct location
    await logGPSViolation({
      employeeId,
      shiftId: currentShift.id,
      attemptedLocation: { lat: latitude, lng: longitude },
      correctLocation: { lat: site.latitude, lng: site.longitude },
      distanceOff: distanceInMeters,
      timestamp: new Date()
    });
    
    // Notify manager immediately
    await notifyManager({
      type: 'GPS_VIOLATION',
      employee: employeeId,
      message: `${employee.name} attempted to clock in ${Math.round(distanceInMeters)}m away from ${site.name}`,
      severity: 'high'
    });
    
    return {
      allowed: false,
      reason: `You must be within 100m of ${site.name} to clock in. You are ${Math.round(distanceInMeters)}m away.`
    };
  }
  
  // Valid - within geofence
  return {
    allowed: true,
    location: { lat: latitude, lng: longitude },
    siteVerified: site.name
  };
}

// Add Haversine distance calculation
function calculateDistance(point1: {lat: number, lng: number}, point2: {lat: number, lng: number}): number {
  const R = 6371e3; // Earth radius in meters
  const φ1 = point1.lat * Math.PI / 180;
  const φ2 = point2.lat * Math.PI / 180;
  const Δφ = (point2.lat - point1.lat) * Math.PI / 180;
  const Δλ = (point2.lng - point1.lng) * Math.PI / 180;

  const a = Math.sin(Δφ/2) * Math.sin(Δφ/2) +
          Math.cos(φ1) * Math.cos(φ2) *
          Math.sin(Δλ/2) * Math.sin(Δλ/2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

  return R * c; // Distance in meters
}

PLATFORM ISSUE DETECTED:

Generate the exact code changes needed.

    });

    await this.escalateToDeveloper({

      analysis,
      proposedFix: codeFix,
      reason: judgeApproval.reason
    });
  }

typescriptclass TrinityCore {
  private context: Map<string, any> = new Map(); // Workspace contexts
  private learnings: TrinityLearning[] = [];
  private geminiClient: GeminiClient;
  private llmJudge: LLMJudge;
  private eventBus: PlatformEventBus;
  
  constructor() {
    this.initializeEventListeners();
    this.startContinuousAnalysis();
    this.loadHistoricalLearnings();
  }
  
  // Trinity listens to EVERYTHING
  private initializeEventListeners() {
    this.eventBus.on('*', async (event) => {
      await this.observe(event);
      await this.analyzeIfNeeded(event);
      await this.actIfNeeded(event);
    });
  }
  
  // Trinity's decision-making interface
  async orchestrate(request: OrchestrationRequest): Promise<OrchestrationResult> {
    const context = await this.gatherContext(request);
    const analysis = await this.analyze(context);
    const decision = await this.decide(analysis);
    const validation = await this.validate(decision);
    
    if (validation.approved) {
      const result = await this.execute(decision);
      await this.learn(decision, result);
      return result;
    } else {
      await this.escalate(decision, validation.reason);
      throw new Error(validation.reason);
    }
  }
  
  // Continuous analysis loops
  private startContinuousAnalysis() {
    setInterval(() => this.analyzeRealTimeOperations(), 5 * 60 * 1000); // 5 min
    setInterval(() => this.analyzeEmployeePerformance(), 60 * 60 * 1000); // 1 hour
    setInterval(() => this.analyzeClientRelationships(), 6 * 60 * 60 * 1000); // 6 hours
    setInterval(() => this.analyzeBusinessStrategy(), 24 * 60 * 60 * 1000); // 24 hours
  }
}
