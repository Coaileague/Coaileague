                            <span className="text-xs font-bold text-purple-600">{workflow.aiConfidence}%</span>
                          </div>
                          <span className={`text-xs font-medium px-2 py-0.5 rounded bg-${priorityColor}-100 text-${priorityColor}-700 capitalize`}>
                            {workflow.priority}
                          </span>
                        </div>
                      </div>

                      <div className={`inline-flex items-center px-2 py-1 rounded text-xs font-medium bg-${color}-100 text-${color}-700 capitalize mb-2`}>
                        {workflow.type}
                      </div>

                      {/* Approval Chain Progress */}
                      <div className="mb-3">
                        <div className="flex items-center space-x-1 mb-1">
                          <GitBranch className="w-3 h-3 text-gray-400" />
                          <span className="text-xs text-gray-600">
                            Approval {workflow.currentApprovalLevel + 1}/{workflow.approvalChain.length}
                          </span>
                        </div>
                        <div className="flex space-x-1">
                          {workflow.approvalChain.map((level, idx) => (
                            <div
                              key={idx}
                              className={`flex-1 h-1 rounded ${
                                level.status === 'approved' ? 'bg-green-500' :
                                idx === workflow.currentApprovalLevel ? 'bg-orange-500' :
                                'bg-gray-200'
                              }`}
                            />
                          ))}
                        </div>
                      </div>

                      <div className="space-y-1 text-sm">
                        {workflow.type === 'payroll' && (
                          <>
                            <div className="flex justify-between">
                              <span className="text-gray-600">Total Amount:</span>
                              <span className="font-bold text-gray-900">{formatCurrency(workflow.summary.totalAmount)}</span>
                            </div>
                            <div className="flex justify-between">
                              <span className="text-gray-600">Employees:</span>
                              <span className="font-medium text-gray-900">{workflow.summary.employees + workflow.summary.contractors}</span>
                            </div>
                          </>
                        )}
                        {workflow.type === 'invoice' && (
                          <>
                            <div className="flex justify-between">
                              <span className="text-gray-600">Total Amount:</span>
                              <span className="font-bold text-gray-900">{formatCurrency(workflow.summary.totalAmount)}</span>
                            </div>
                            <div className="flex justify-between">
                              <span className="text-gray-600">Invoices:</span>
                              <span className="font-medium text-gray-900">{workflow.summary.invoiceCount}</span>
                            </div>
                          </>
                        )}
                        {workflow.type === 'schedule' && (
                          <>
                            <div className="flex justify-between">
                              <span className="text-gray-600">Total Hours:</span>
                              <span className="font-bold text-gray-900">{workflow.summary.totalHours}</span>
                            </div>
                            <div className="flex justify-between">
                              <span className="text-gray-600">Employees:</span>
                              <span className="font-medium text-gray-900">{workflow.summary.employees}</span>
                            </div>
                          </>
                        )}
                      </div>

                      {/* Approval Status Badge */}
                      {approvalStatus.canApprove ? (
                        <div className="mt-3 pt-3 border-t border-gray-200">
                          <div className="flex items-center space-x-2 text-green-600">
                            <CheckSquare className="w-4 h-4" />
                            <span className="text-xs font-medium">Ready for your approval</span>
                          </div>
                        </div>
                      ) : (
                        <div className="mt-3 pt-3 border-t border-gray-200">
                          <div className="flex items-center space-x-2 text-gray-500">
                            <Lock className="w-4 h-4" />
                            <span className="text-xs">{approvalStatus.reason}</span>
                          </div>
                        </div>
                      )}
                    </div>
                  );
                })}

                {filteredWorkflows.length === 0 && (
                  <div className="p-8 text-center">
                    <CheckCircle className="w-12 h-12 text-green-500 mx-auto mb-3" />
                    <p className="text-gray-600 font-medium">All caught up!</p>
                    <p className="text-sm text-gray-500">No pending approvals for your role</p>
                  </div>
                )}
              </div>
            </div>
          </div>

          {/* Main Content - Workflow Details */}
          <div className="col-span-2">
            {selectedWorkflow ? (
              <div className="space-y-6">
                {/* Workflow Header */}
                <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
                  <div className="flex items-start justify-between mb-6">
                    <div>
                      <div className="flex items-center space-x-3 mb-2">
                        <h2 className="text-2xl font-bold text-gray-900">{selectedWorkflow.id}</h2>
                        <span className={`px-3 py-1 rounded-full text-sm font-medium capitalize ${
                          selectedWorkflow.type === 'payroll' ? 'bg-blue-100 text-blue-700' :
                          selectedWorkflow.type === 'invoice' ? 'bg-green-100 text-green-700' :
                          'bg-purple-100 text-purple-700'
                        }`}>
                          {selectedWorkflow.type}
                        </span>
                        <span className={`px-3 py-1 rounded-full text-sm font-medium capitalize bg-${getPriorityColor(selectedWorkflow.priority)}-100 text-${getPriorityColor(selectedWorkflow.priority)}-700`}>
                          {selectedWorkflow.priority} Priority
                        </span>
                      </div>
                      <p className="text-gray-600">
                        Created by <span className="font-medium">{selectedWorkflow.createdBy.name}</span> • {formatDate(selectedWorkflow.createdAt)}
                      </p>
                      <p className="text-sm text-gray-500 mt-1">
                        Scheduled for: <span className="font-medium text-gray-900">{formatDate(selectedWorkflow.scheduledFor)}</span>
                      </p>
                      <p className="text-sm text-gray-500">
                        Department: <span className="font-medium text-gray-900">{selectedWorkflow.department}</span>
                      </p>
                    </div>

                    <div className="flex items-center space-x-2">
                      <div className="bg-purple-50 border border-purple-200 rounded-lg px-4 py-2 text-center">
                        <div className="text-xs text-purple-600 font-medium">AI Confidence</div>
                        <div className="text-2xl font-bold text-purple-600">{selectedWorkflow.aiConfidence}%</div>
                      </div>
                    </div>
                  </div>

                  {/* Approval Chain Visualization */}
                  <div className="bg-gradient-to-r from-blue-50 to-purple-50 border border-blue-200 rounded-lg p-4 mb-4">
                    <div className="flex items-center space-x-2 mb-3">
                      <GitBranch className="w-5 h-5 text-blue-600" />
                      <h3 className="font-bold text-gray-900">Approval Chain</h3>
                    </div>
                    
                    <div className="space-y-3">
                      {selectedWorkflow.approvalChain.map((level, idx) => (
                        <div key={idx} className="flex items-center space-x-3">
                          <div className={`w-8 h-8 rounded-full flex items-center justify-center ${
                            level.status === 'approved' ? 'bg-green-500' :
                            idx === selectedWorkflow.currentApprovalLevel ? 'bg-orange-500' :
                            'bg-gray-300'
                          }`}>
                            {level.status === 'approved' ? (
                              <CheckCircle className="w-5 h-5 text-white" />
                            ) : idx === selectedWorkflow.currentApprovalLevel ? (
                              <Clock3 className="w-5 h-5 text-white" />
                            ) : (
                              <span className="text-white font-bold text-sm">{idx + 1}</span>
                            )}
                          </div>
                          
                          <div className="flex-1">
                            <div className="flex items-center justify-between">
                              <div>
                                <div className="font-medium text-gray-900 capitalize">
                                  {level.role.replace('_', ' ')}
                                </div>
                                {level.status === 'approved' && (
                                  <div className="text-xs text-gray-600">
                                    Approved by {level.approvedBy} • {formatDate(level.approvedAt)}
                                  </div>
                                )}
                                {level.status === 'pending' && idx === selectedWorkflow.currentApprovalLevel && (
                                  <div className="text-xs text-orange-600 font-medium">
                                    Awaiting approval
                                  </div>
                                )}
                              </div>
                              
                              {level.status === 'approved' && (
                                <span className="text-green-600 text-xs font-medium flex items-center space-x-1">
                                  <CheckCircle className="w-4 h-4" />
                                  <span>Approved</span>
                                </span>
                              )}
                            </div>
                            {level.comment && (
                              <div className="text-xs text-gray-600 mt-1 italic">
                                "{level.comment}"
                              </div>
                            )}
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>

                  {/* Summary Cards - Same as before */}
                  <div className="grid grid-cols-4 gap-4">
                    {selectedWorkflow.type === 'payroll' && (
                      <>
                        <div className="bg-blue-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <DollarSign className="w-4 h-4 text-blue-600" />
                            <span className="text-xs font-medium text-blue-600">Total Payroll</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {formatCurrency(selectedWorkflow.summary.totalAmount)}
                          </div>
                        </div>
                        <div className="bg-green-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <Users className="w-4 h-4 text-green-600" />
                            <span className="text-xs font-medium text-green-600">Employees</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.employees}
                          </div>
                        </div>
                        <div className="bg-purple-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <UserCheck className="w-4 h-4 text-purple-600" />
                            <span className="text-xs font-medium text-purple-600">Contractors</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.contractors}
                          </div>
                        </div>
                        <div className="bg-orange-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <Clock className="w-4 h-4 text-orange-600" />
                            <span className="text-xs font-medium text-orange-600">Total Hours</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.totalHours}
                          </div>
                        </div>
                      </>
                    )}

                    {selectedWorkflow.type === 'invoice' && (
                      <>
                        <div className="bg-green-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <DollarSign className="w-4 h-4 text-green-600" />
                            <span className="text-xs font-medium text-green-600">Total Amount</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {formatCurrency(selectedWorkflow.summary.totalAmount)}
                          </div>
                        </div>
                        <div className="bg-blue-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <FileText className="w-4 h-4 text-blue-600" />
                            <span className="text-xs font-medium text-blue-600">Invoices</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.invoiceCount}
                          </div>
                        </div>
                        <div className="bg-purple-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <Building className="w-4 h-4 text-purple-600" />
                            <span className="text-xs font-medium text-purple-600">Clients</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.clients}
                          </div>
                        </div>
                        <div className="bg-orange-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <Calendar className="w-4 h-4 text-orange-600" />
                            <span className="text-xs font-medium text-orange-600">Payment Terms</span>
                          </div>
                          <div className="text-lg font-bold text-gray-900">
                            {selectedWorkflow.summary.paymentTerms}
                          </div>
                        </div>
                      </>
                    )}

                    {selectedWorkflow.type === 'schedule' && (
                      <>
                        <div className="bg-purple-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <Clock className="w-4 h-4 text-purple-600" />
                            <span className="text-xs font-medium text-purple-600">Total Hours</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.totalHours}
                          </div>
                        </div>
                        <div className="bg-blue-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <Users className="w-4 h-4 text-blue-600" />
                            <span className="text-xs font-medium text-blue-600">Employees</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.employees}
                          </div>
                        </div>
                        <div className="bg-orange-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <AlertCircle className="w-4 h-4 text-orange-600" />
                            <span className="text-xs font-medium text-orange-600">Open Shifts</span>
                          </div>
                          <div className="text-xl font-bold text-gray-900">
                            {selectedWorkflow.summary.openShifts}
                          </div>
                        </div>
                        <div className="bg-green-50 rounded-lg p-4">
                          <div className="flex items-center space-x-2 mb-2">
                            <TrendingUp className="w-4 h-4 text-green-600" />
                            <span className="text-xs font-medium text-green-600">Profit Impact</span>
                          </div>
                          <div className="text-xl font-bold text-green-600">
                            {selectedWorkflow.profitImpact}
                          </div>
                        </div>
                      </>
                    )}
                  </div>
                </div>

                {/* Comments Section */}
                {selectedWorkflow.comments.length > 0 && (
                  <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
                    <h3 className="font-bold text-gray-900 mb-4 flex items-center space-x-2">
                      <MessageSquare className="w-5 h-5" />
                      <span>Comments & Notes</span>
                    </h3>
                    <div className="space-y-3">
                      {selectedWorkflow.comments.map((comment, idx) => (
                        <div key={idx} className="bg-gray-50 rounded-lg p-3">
                          <div className="flex items-center justify-between mb-1">
                            <span className="font-medium text-gray-900">{comment.userName}</span>
                            <span className="text-xs text-gray-500">{formatDate(comment.timestamp)}</span>
                          </div>
                          <p className="text-sm text-gray-700">{comment.comment}</p>
                        </div>
                      ))}
                    </div>
                  </div>
                )}

                {/* Approval Actions */}
                <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
                  {(() => {
                    const approvalStatus = canApproveWorkflow(selectedWorkflow);
                    
                    if (!approvalStatus.canApprove) {
                      return (
                        <div className="bg-yellow-50 border border-yellow-200 rounded-lg p-4">
                          <div className="flex items-start space-x-3">
                            <Lock className="w-5 h-5 text-yellow-600 mt-0.5" />
                            <div>
                              <h3 className="font-bold text-yellow-900 mb-1">Cannot Approve</h3>
                              <p className="text-sm text-yellow-800">{approvalStatus.reason}</p>
                            </div>
                          </div>
                        </div>
                      );
                    }

                    return (
                      <>
                        <div className="mb-4">
                          <label className="block text-sm font-medium text-gray-700 mb-2">
                            Add Comment (Optional)
                          </label>
                          <textarea
                            value={comment}
                            onChange={(e) => setComment(e.target.value)}
                            placeholder="Add a comment about your decision..."
                            rows={3}
                            className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                          />
                        </div>

                        <div className="flex items-center justify-between">
                          <div className="flex items-center space-x-3">
                            {currentUser.permissions.delegateApprovals && (
                              <button
                                onClick={() => setShowDelegationModal(true)}
                                className="px-4 py-2 border border-gray-300 text-gray-700 rounded-lg hover:bg-gray-50 transition-colors font-medium flex items-center space-x-2"
                              >
                                <UserPlus className="w-4 h-4" />
                                <span>Delegate</span>
                              </button>
                            )}
                          </div>
                          
                          <div className="flex items-center space-x-3">
                            <button
                              onClick={() => handleReject(selectedWorkflow.id)}
                              disabled={processingApproval}
                              className="px-6 py-3 border-2 border-red-300 text-red-700 rounded-lg hover:bg-red-50 transition-colors font-medium disabled:opacity-50 disabled:cursor-not-allowed flex items-center space-x-2"
                            >
                              <XCircle className="w-5 h-5" />
                              <span>Reject</span>
                            </button>
                            
                            <button
                              onClick={() => handleApprove(selectedWorkflow.id)}
                              disabled={processingApproval}
                              className="px-8 py-3 bg-gradient-to-r from-green-600 to-emerald-600 text-white rounded-lg hover:from-green-700 hover:to-emerald-700 transition-colors font-medium shadow-lg disabled:opacity-50 disabled:cursor-not-allowed flex items-center space-x-2"
                            >
                              {processingApproval ? (
                                <>
                                  <Loader className="w-5 h-5 animate-spin" />
                                  <span>Processing...</span>
                                </>
                              ) : (
                                <>
                                  <CheckCircle className="w-5 h-5" />
                                  <span>Approve</span>
                                </>
                              )}
                            </button>
                          </div>
                        </div>
                      </>
                    );
                  })()}
                </div>
              </div>
            ) : (
              <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-12 text-center">
                <Clock className="w-16 h-16 text-gray-300 mx-auto mb-4" />
                <h3 className="text-xl font-bold text-gray-900 mb-2">Select a Workflow to Review</h3>
                <p className="text-gray-600">Choose a pending workflow from the left sidebar to view details and approve</p>
              </div>
            )}
          </div>
        </div>
      </div>

      {/* Role & Permissions Panel */}
      {showRolePanel && (
        <div className="fixed inset-y-0 right-0 w-96 bg-white shadow-2xl border-l border-gray-200 z-50 overflow-y-auto">
          <div className="p-6">
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-xl font-bold text-gray-900">Role & Permissions</h2>
              <button onClick={() => setShowRolePanel(false)} className="text-gray-400 hover:text-gray-600">
                <XCircle className="w-6 h-6" />
              </button>
            </div>

            <div className="space-y-6">
              {/* Current Role Card */}
              <div className={`bg-${roles[currentUser.role].color}-50 border-2 border-${roles[currentUser.role].color}-200 rounded-lg p-4`}>
                <div className="flex items-center space-x-3 mb-3">
                  {roles[currentUser.role].icon}
                  <div>
                    <div className="font-bold text-gray-900">{roles[currentUser.role].name}</div>
                    <div className="text-sm text-gray-600">{currentUser.email}</div>
                  </div>
                </div>
                <p className="text-sm text-gray-700">{roles[currentUser.role].description}</p>
              </div>

              {/* Permissions List */}
              <div>
                <h3 className="font-bold text-gray-900 mb-3">Your Permissions</h3>
                <div className="space-y-2">
                  {Object.entries(currentUser.permissions).map(([key, value]) => (
                    <div key={key} className="flex items-center justify-between p-2 bg-gray-50 rounded">
                      <span className="text-sm text-gray-700 capitalize">{key.replace(/([A-Z])/g, ' $1').trim()}</span>
                      {value ? (
                        <CheckCircle className="w-4 h-4 text-green-600" />
                      ) : (
                        <XCircle className="w-4 h-4 text-gray-300" />
                      )}
                    </div>
                  ))}
                </div>
              </div>

              {/* All Roles Reference */}
              <div>
                <h3 className="font-bold text-gray-900 mb-3">All Roles</h3>
                <div className="space-y-2">
                  {Object.entries(roles).map(([roleKey, role]) => (
                    <div key={roleKey} className={`p-3 rounded-lg border-2 ${
                      currentUser.role === roleKey ? `border-${role.color}-500 bg-${role.color}-50` : 'border-gray-200'
                    }`}>
                      <div className="flex items-center space-x-2 mb-1">
                        {role.icon}
                        <span className="font-medium text-gray-900">{role.name}</span>
                      </div>
                      <p className="text-xs text-gray-600">{role.description}</p>
                    </div>
                  ))}
                </div>
              </div>
            </div>
          </div>
        </div>
      )}

      {/* Audit Log Panel */}
      {showAuditLog && (
        <div className="fixed inset-y-0 right-0 w-[500px] bg-white shadow-2xl border-l border-gray-200 z-50 overflow-y-auto">
          <div className="p-6">
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-xl font-bold text-gray-900 flex items-center space-x-2">
                <History className="w-6 h-6" />
                <span>Audit Log</span>
              </h2>
              <button onClick={() => setShowAuditLog(false)} className="text-gray-400 hover:text-gray-600">
                <XCircle className="w-6 h-6" />
              </button>
            </div>

            <div className="space-y-3">
              {auditLog.map(entry => (
                <div key={entry.id} className="border border-gray-200 rounded-lg p-4 hover:bg-gray-50 transition-colors">
                  <div className="flex items-start justify-between mb-2">
                    <div className="flex items-center space-x-2">
                      {entry.action === 'approved' && <CheckCircle className="w-4 h-4 text-green-600" />}
                      {entry.action === 'rejected' && <XCircle className="w-4 h-4 text-red-600" />}
                      {entry.action === 'viewed' && <Eye className="w-4 h-4 text-blue-600" />}
                      {entry.action === 'delegated' && <UserPlus className="w-4 h-4 text-purple-600" />}
                      <span className="font-medium text-gray-900 capitalize">{entry.action}</span>
                    </div>
                    <span className="text-xs text-gray-500">{formatDate(entry.timestamp)}</span>
                  </div>
                  <div className="text-sm text-gray-700 mb-1">
                    <span className="font-medium">{entry.userName}</span> {entry.action} workflow <span className="font-medium">{entry.workflowId}</span>
                  </div>
                  <div className="text-xs text-gray-600">{entry.details}</div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}

      {/* Delegation Modal */}
      {showDelegationModal && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-xl shadow-2xl max-w-md w-full">
            <div className="p-6">
              <h3 className="text-xl font-bold text-gray-900 mb-4">Delegate Approval</h3>
              <p className="text-sm text-gray-600 mb-4">
                Select a team member to delegate this approval to:
              </p>
              
              <div className="space-y-2 mb-6">
                {teamMembers.map(member => (
                  <button
                    key={member.id}
                    onClick={() => handleDelegate(selectedWorkflow.id, member.id)}
                    className="w-full p-3 border-2 border-gray-200 rounded-lg hover:border-blue-500 hover:bg-blue-50 transition-all text-left"
                  >
                    <div className="font-medium text-gray-900">{member.name}</div>
                    <div className="text-sm text-gray-600">{member.email}</div>
                    <div className="text-xs text-gray-500 mt-1">
                      {roles[member.role]?.name} • {member.department}
                    </div>
                  </button>
                ))}
              </div>

              <button
                onClick={() => setShowDelegationModal(false)}
                className="w-full px-4 py-2 border border-gray-300 rounded-lg hover:bg-gray-50 transition-colors"
              >
                Cancel
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default AutomationWorkflowApprovalRBAC;
