// Determine which flow to trigger based on user message
export function detectIntakeFlow(message: string): string | null {
  const lower = message.toLowerCase();

  if (lower.includes('locked out') || lower.includes('cannot log in') ||
      lower.includes('cant log in') || lower.includes('account locked')) {
    return 'account_locked';
  }

  if (lower.includes('call off') || lower.includes('calloff') ||
      lower.includes('calling off') || lower.includes('cant make') ||
      lower.includes('sick') || lower.includes('emergency')) {
    return 'calloff_report';
  }

  if (lower.includes('paycheck') || lower.includes('payroll') ||
      lower.includes('missing hours') || lower.includes('wrong pay') ||
      lower.includes('not paid')) {
    return 'payroll_dispute';
  }

  if (lower.includes('incident') || lower.includes('report') ||
      lower.includes('fight') || lower.includes('trespass') ||
      lower.includes('theft') || lower.includes('assault')) {
    return 'incident_report';
  }

  if (lower.includes('notification') || lower.includes('not receiving') ||
      lower.includes('no alerts') || lower.includes('missing alerts')) {
    return 'notification_fix';
  }

  return null; // Use standard Trinity chat response
}

```typescript
import express from 'express';
import { requireAuth } from '../middleware/auth';
import { pool } from '../db';
import { 
  TRINITY_INTAKE_FLOWS, 
  detectIntakeFlow 
} from '../services/trinityIntakeFlows';
import { broadcastToWorkspace } from '../websocket';
import { universalAuditService } from '../services/universalAuditService';

const router = express.Router();

// Start an intake session
router.post('/api/trinity/intake/start', requireAuth, async (req, res, next) => {
  try {
    const { workspaceId, userId } = req.user;
    const { flowType, chatRoomId, triggerMessage } = req.body;

    // Auto-detect flow if not specified
    const resolvedFlowType = flowType || 
      detectIntakeFlow(triggerMessage || '') || 
      'support_triage';

    const flow = TRINITY_INTAKE_FLOWS[resolvedFlowType];
    if (!flow) return res.status(404).json({ error: 'Flow not found' });

    // Create session
    const session = await pool.query(`
      INSERT INTO trinity_intake_sessions
      (workspace_id, chat_room_id, user_id, session_type, 
       flow_definition, current_step_index, status)
      VALUES ($1, $2, $3, $4, $5, 0, 'active')
      RETURNING *
    `, [workspaceId, chatRoomId, userId, 
        resolvedFlowType, JSON.stringify(flow.steps)]);

    const sessionData = session.rows[0];

    // Get first step
    const firstStep = flow.steps[0];

    // Broadcast Trinity greeting + first widget to DockChat
    broadcastToWorkspace(workspaceId, {
      type: 'trinity_intake_widget',
      data: {
        sessionId: sessionData.id,
        chatRoomId,
        messageType: 'intake_greeting',
        greeting: flow.greeting,
        step: firstStep,
        stepIndex: 0,
        totalSteps: flow.steps.length,
        flowTitle: flow.title
      }
    });

    res.json({
      sessionId: sessionData.id,
      flowType: resolvedFlowType,
      firstStep,
      greeting: flow.greeting
    });
  } catch (err) {
    next(err);
  }
});

// Submit a step response and get next step
router.post('/api/trinity/intake/:sessionId/respond', 
  requireAuth, async (req, res, next) => {
  try {
    const { workspaceId, userId } = req.user;
    const { sessionId } = req.params;
    const { fieldId, value, stepIndex } = req.body;

    // Get session
    const sessionResult = await pool.query(
      'SELECT * FROM trinity_intake_sessions WHERE id = $1 AND workspace_id = $2',
      [sessionId, workspaceId]
    );

    if (!sessionResult.rows[0]) {
      return res.status(404).json({ error: 'Session not found' });
    }

    const session = sessionResult.rows[0];
    const flow = TRINITY_INTAKE_FLOWS[session.session_type];

    if (!flow) return res.status(404).json({ error: 'Flow not found' });

    // Store response
    await pool.query(`
      INSERT INTO trinity_intake_responses
      (session_id, workspace_id, step_index, field_id, 
       field_label, field_type, raw_value, parsed_value)
      VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
    `, [
      sessionId, workspaceId, stepIndex, fieldId,
      flow.steps[stepIndex]?.question,
      flow.steps[stepIndex]?.widgetType,
      String(value),
      JSON.stringify(value)
    ]);

    // Update collected data
    const updatedData = {
      ...(session.collected_data || {}),
      [fieldId]: value
    };

    await pool.query(
      'UPDATE trinity_intake_sessions SET collected_data = $1 WHERE id = $2',
      [JSON.stringify(updatedData), sessionId]
    );

    // Find next step (skipping conditional steps if needed)
    let nextStepIndex = stepIndex + 1;
    while (nextStepIndex < flow.steps.length) {
      const nextStep = flow.steps[nextStepIndex];
      if (nextStep.skipIf) {
        const skipFieldValue = updatedData[nextStep.skipIf.fieldId];
        if (skipFieldValue === nextStep.skipIf.value) {
          nextStepIndex++;
          continue;
        }
      }
      break;
    }

    // Check if flow is complete
    if (nextStepIndex >= flow.steps.length) {
      // Mark complete
      await pool.query(`
        UPDATE trinity_intake_sessions 
        SET status = 'completed', 
            current_step_index = $1,
            completed_at = NOW()
        WHERE id = $2
      `, [nextStepIndex, sessionId]);

      // Broadcast completion message
      broadcastToWorkspace(workspaceId, {
        type: 'trinity_intake_widget',
        data: {
          sessionId,
          chatRoomId: session.chat_room_id,
          messageType: 'intake_complete',
          completionMessage: flow.completionMessage,
          collectedData: updatedData
        }
      });

      // Trigger Trinity action asynchronously
      processTrinityIntakeAction(
        sessionId, workspaceId, userId,
        flow.trinityAction, updatedData
      ).catch(err => console.error('Intake action error:', err));

      return res.json({ 
        complete: true, 
        completionMessage: flow.completionMessage 
      });
    }

    // Advance to next step
    const nextStep = flow.steps[nextStepIndex];

    await pool.query(
      'UPDATE trinity_intake_sessions SET current_step_index = $1 WHERE id = $2',
      [nextStepIndex, sessionId]
    );

    // Broadcast next widget
    broadcastToWorkspace(workspaceId, {
      type: 'trinity_intake_widget',
      data: {
        sessionId,
        chatRoomId: session.chat_room_id,
        messageType: 'intake_next_step',
        step: nextStep,
        stepIndex: nextStepIndex,
        totalSteps: flow.steps.length,
        previousAnswer: value
      }
    });

    res.json({
      complete: false,
      nextStep,
      nextStepIndex
    });
  } catch (err) {
    next(err);
  }
});

// Abandon an intake session
router.post('/api/trinity/intake/:sessionId/abandon',
  requireAuth, async (req, res, next) => {
  try {
    const { workspaceId } = req.user;
    await pool.query(
      'UPDATE trinity_intake_sessions SET status = $1 WHERE id = $2 AND workspace_id = $3',
      ['abandoned', req.params.sessionId, workspaceId]
    );
    res.json({ success: true });
  } catch (err) {
    next(err);
  }
});

// Async processor for completed intake sessions
async function processTrinityIntakeAction(
  sessionId: string,
  workspaceId: string,
  userId: string,
  actionType: string,
  collectedData: Record<string, any>
) {
  try {
    // Route to appropriate Trinity action based on flow type
    // Each action uses the collected data to resolve the issue

    await universalAuditService.log({
      workspaceId,
      actorId: 'trinity-system-actor-000000000000',
      entityType: 'trinity_intake_session',
      entityId: sessionId,
      action: actionType,
      data: { collectedData, actionType }
    });

    // Additional action routing handled by existing Trinity action registry
    // The action type maps directly to a registered Trinity action
    
    await pool.query(
      `UPDATE trinity_intake_sessions 
       SET trinity_action_taken = $1 
       WHERE id = $2`,
      [actionType, sessionId]
    );
  } catch (err) {
    console.error('Intake action processing error:', err);
  }
}

export default router;

  greeting,
  flowTitle,
  onComplete,
  onSkip
}: TrinityIntakeWidgetProps) {
  const { workspaceId } = useAuth();
  const [value, setValue] = useState<any>('');
  const [error, setError] = useState<string>('');
  const [submitted, setSubmitted] = useState(false);

  const respondMutation = useMutation({
    mutationFn: async () => {
      const res = await fetch(`/api/trinity/intake/${sessionId}/respond`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          fieldId: step.fieldId,
          value,
          stepIndex
        })
      });
      if (!res.ok) throw new Error('Response failed');
      return res.json();
    },
    onSuccess: (data) => {
      setSubmitted(true);
      onComplete(data);
    }
  });

  const validate = useCallback(() => {
    if (step.required && (!value || value === '')) {
      setError('This field is required');
      return false;
    }
    if (step.validation?.minLength && String(value).length < step.validation.minLength) {
      setError(`Must be at least ${step.validation.minLength} characters`);
      return false;
    }
    setError('');
    return true;
  }, [value, step]);

  const handleSubmit = useCallback(() => {
    if (validate()) {
      respondMutation.mutate();
    }
  }, [validate, respondMutation]);

  const renderInput = () => {
    const baseClass = `w-full rounded-xl border px-4 py-3 text-sm 
      focus:outline-none focus:ring-2 focus:ring-purple-500
      ${error ? 'border-red-400' : 'border-gray-200'}
      bg-white text-gray-900`;

    switch (step.widgetType) {
      case 'select':
        return (
          <div className="relative">
            <select
              value={value}
              onChange={e => { setValue(e.target.value); setError(''); }}
              className={baseClass + ' appearance-none cursor-pointer'}
            >
              <option value="">Select an option...</option>
              {step.options?.map(opt => (
                <option key={opt} value={opt}>{opt}</option>
              ))}
            </select>
            <ChevronDown className="absolute right-3 top-1/2 -translate-y-1/2 h-4 w-4 text-gray-400 pointer-events-none" />
          </div>
        );

      case 'radio':
        return (
          <div className="space-y-2">
            {step.options?.map(opt => (
              <label
                key={opt}
                onClick={() => { setValue(opt); setError(''); }}
                className={`flex items-center gap-3 p-3 rounded-xl border cursor-pointer transition-all ${
                  value === opt
                    ? 'border-purple-500 bg-purple-50'
                    : 'border-gray-200 hover:border-gray-300 bg-white'
                }`}
              >
                <div className={`h-4 w-4 rounded-full border-2 flex items-center justify-center ${
                  value === opt ? 'border-purple-500' : 'border-gray-300'
                }`}>
                  {value === opt && (
                    <div className="h-2 w-2 rounded-full bg-purple-500" />
                  )}
                </div>
                <span className="text-sm text-gray-800">{opt}</span>
              </label>
            ))}
          </div>
        );

      case 'textarea':
        return (
          <textarea
            value={value}
            onChange={e => { setValue(e.target.value); setError(''); }}
            placeholder={step.placeholder}
            rows={4}
            className={baseClass + ' resize-none'}
          />
        );

      case 'date':
        return (
          <input
            type="date"
            value={value}
            onChange={e => { setValue(e.target.value); setError(''); }}
            className={baseClass}
          />
        );

      case 'number':
        return (
          <input
            type="number"
            value={value}
            onChange={e => { setValue(e.target.value); setError(''); }}
            placeholder={step.placeholder}
            min={step.validation?.min}
            max={step.validation?.max}
            step="0.5"
            className={baseClass}
          />
        );

      case 'phone':
        return (
          <input
            type="tel"
            value={value}
            onChange={e => { setValue(e.target.value); setError(''); }}
            placeholder={step.placeholder || '(xxx) xxx-xxxx'}
            className={baseClass}
          />
        );

      case 'email':
        return (
          <input
            type="email"
            value={value}
            onChange={e => { setValue(e.target.value); setError(''); }}
            placeholder={step.placeholder || 'email@example.com'}
            className={baseClass}
          />
        );

      default: // text
        return (
          <input
            type="text"
            value={value}
            onChange={e => { setValue(e.target.value); setError(''); }}
            placeholder={step.placeholder}
            className={baseClass}
            onKeyDown={e => { if (e.key === 'Enter') handleSubmit(); }}
          />
        );
    }
  };

  if (submitted) {
    return (
      <div className="flex items-center gap-2 py-2 text-sm text-gray-500">
        <Check className="h-4 w-4 text-green-500" />
        <span className="italic">{String(value)}</span>
      </div>
    );
  }

  return (
    <div className="my-2">
      {/* Trinity greeting card — only on first step */}
      {greeting && stepIndex === 0 && (
        <div className="flex items-start gap-3 mb-3">
          <div className="h-8 w-8 rounded-full bg-purple-600 flex items-center justify-center flex-shrink-0">
            <span className="text-white text-xs font-bold">T</span>
          </div>
          <div className="bg-white border border-gray-100 rounded-2xl rounded-tl-sm px-4 py-3 shadow-sm max-w-xs">
            <p className="text-sm text-gray-800 leading-relaxed">{greeting}</p>
          </div>
        </div>
      )}

      {/* Widget card */}
      <div className="flex items-start gap-3">
        <div className="h-8 w-8 rounded-full bg-purple-600 flex items-center justify-center flex-shrink-0">
          <span className="text-white text-xs font-bold">T</span>
        </div>
        <div className="flex-1 max-w-sm">
          <div className="bg-white border border-gray-100 rounded-2xl rounded-tl-sm px-4 py-4 shadow-sm">
            {/* Progress indicator */}
            <div className="flex items-center gap-2 mb-3">
              <div className="flex gap-1">
                {Array.from({ length: totalSteps }).map((_, i) => (
                  <div
                    key={i}
                    className={`h-1 rounded-full transition-all ${
                      i < stepIndex ? 'bg-purple-500 w-4' :
                      i === stepIndex ? 'bg-purple-500 w-6' :
                      'bg-gray-200 w-4'
                    }`}
                  />
                ))}
              </div>
              <span className="text-xs text-gray-400">
                {stepIndex + 1} of {totalSteps}
              </span>
            </div>

            {/* Question */}
            <p className="text-sm font-semibold text-gray-900 mb-1">
              {step.question}
            </p>

            {/* Subtext */}
            {step.subtext && (
              <p className="text-xs text-gray-500 mb-3">{step.subtext}</p>
            )}

            {/* Input */}
            <div className="mb-3">
              {renderInput()}
              {error && (
                <p className="text-xs text-red-500 mt-1">{error}</p>
              )}
            </div>

            {/* Submit button */}
            <div className="flex gap-2">
              <button
                onClick={handleSubmit}
                disabled={respondMutation.isPending}
                className="flex-1 flex items-center justify-center gap-2 py-2.5 bg-purple-600 text-white rounded-xl text-sm font-semibold hover:bg-purple-700 transition-colors disabled:opacity-50"
              >
                {respondMutation.isPending ? (
                  <Loader2 className="h-4 w-4 animate-spin" />
                ) : (
                  <>
                    Continue
                    <ChevronRight className="h-4 w-4" />
                  </>
                )}
              </button>

              {!step.required && (
                <button
                  onClick={onSkip}
                  className="px-3 py-2 text-xs text-gray-400 hover:text-gray-600 transition-colors"
                >
                  Skip
                </button>
              )}
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
```
