async function finalizeInvoice(draftInvoice: Invoice) {
  const lineItems = draftInvoice.items.map(item => ({
    description: item.description,
    amount: Math.round(item.amount * 100), // Stripe uses cents
    quantity: 1
  }));

  const stripeInvoice = await stripe.invoices.create({
    customer: draftInvoice.client.stripeCustomerId,
    collection_method: 'send_invoice',
    days_until_due: 30,
    auto_advance: false, // Keep manual finalization for safety
    metadata: {
      autoforceInvoiceId: draftInvoice.id,
      workspaceId: draftInvoice.workspaceId
    }
  });

  for (const item of lineItems) {
    await stripe.invoiceItems.create({
      customer: draftInvoice.client.stripeCustomerId,
      invoice: stripeInvoice.id,
      ...item
    });
  }

  // Finalize and send
  await stripe.invoices.finalizeInvoice(stripeInvoice.id);
  await stripe.invoices.sendInvoice(stripeInvoice.id);

  // Update your DB
  await db.update(invoices)
    .set({ 
      status: 'sent',
      stripeInvoiceId: stripeInvoice.id,
      sentAt: new Date()
    })
    .where(eq(invoices.id, draftInvoice.id));
}

async function persistSchedule(scheduleDecision: ScheduleDecision) {
  const createdShifts = [];
  
  for (const shift of scheduleDecision.shifts) {
    const created = await db.insert(shifts).values({
      workspaceId: scheduleDecision.workspaceId,
      employeeId: shift.employeeId,
      clientId: shift.clientId,
      startTime: new Date(shift.startTime),
      endTime: new Date(shift.endTime),
      role: shift.role,
      status: scheduleDecision.confidence > 0.95 
        ? 'confirmed' 
        : 'pending_review',
      generatedBy: 'ai_brain',
      confidence: scheduleDecision.confidence
    }).returning();
    
    createdShifts.push(created[0]);
  }

  // Log the decision
  await db.insert(aiEventStream).values({
    workspaceId: scheduleDecision.workspaceId,
    eventType: 'schedule_generated',
    payload: {
      shiftIds: createdShifts.map(s => s.id),
      reasoning: scheduleDecision.reasoning,
      confidence: scheduleDecision.confidence
    },
    metadata: {
      model: 'gemini-2.0-flash-exp',
      tokensUsed: scheduleDecision.tokensUsed
    }
  });

  return createdShifts;
}

typescript// In payrollAutomation.ts
async function submitToGusto(payrollRun: PayrollRun) {
  const gustoPayroll = await gustoClient.payrolls.create({
    company_id: workspace.gustoCompanyId,
    pay_period: {
      start_date: payrollRun.periodStart.toISOString(),
      end_date: payrollRun.periodEnd.toISOString()
    },
    payroll_deadline: payrollRun.processingDate.toISOString(),
    // CRITICAL: Set to false initially
    process_payroll: false // Manual approval step
  });

  // Add each employee's pay
  for (const calc of payrollRun.calculations) {
    await gustoClient.employeePayments.create({
      payroll_id: gustoPayroll.id,
      employee_id: calc.gustoEmployeeId,
      regular_hours: calc.regularHours,
      overtime_hours: calc.overtimeHours,
      wage: calc.hourlyRate,
      // Gusto calculates taxes automatically
    });
  }

  // Update DB with Gusto ID
  await db.update(payrollRuns)
    .set({ 
      gustoPayrollId: gustoPayroll.id,
      status: 'submitted_to_gusto'
    })
    .where(eq(payrollRuns.id, payrollRun.id));

  return gustoPayroll;
}

async function checkDuplicate(type: string, key: string) {
  const exists = await redis.get(`processed:${type}:${key}`);
  if (exists) throw new Error('Already processed');
  await redis.setex(`processed:${type}:${key}`, 86400 * 14, '1');
}
