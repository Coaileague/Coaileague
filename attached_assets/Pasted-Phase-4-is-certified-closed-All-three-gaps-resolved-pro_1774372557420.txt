**SHORTCUT PATTERNS THAT WILL BE REJECTED**

**REJECTED — Frontend-only business rule enforcement.** Business rules enforced only in the UI can be bypassed by calling the API directly. Every rule must be enforced server-side. Frontend enforcement is supplementary UX only.

**REJECTED — Generic error messages.** A business rule rejection that returns `"Invalid input"` is not acceptable. Every rejection names the specific field, the specific rule violated, and the acceptable range or values. The user must know exactly what to fix.

**REJECTED — Partial domain coverage.** If payroll has business rules, every payroll endpoint enforces them. Not the ones that seem risky. Every endpoint that accepts a value governed by a business rule validates against that rule.

**REJECTED — Rules that conflict with Phase 2 Zod schemas.** Phase 2 Zod schemas validate structure and type. Phase 5 business rules validate logical constraints. They must not duplicate each other or contradict each other. Rules are applied after Zod validation passes — not instead of it.

**REJECTED — Silent rule violations.** A business rule that logs a warning and proceeds is not enforcement. Every rule violation returns a 400 response with a specific error before any database operation executes.

**STEP 1 — CREATE THE CENTRAL BUSINESSRULES MODULE**

Create `server/lib/businessRules.ts`. This is the single source of truth for every business constraint in the platform. Every rule is named, typed, and testable. No business constraint is defined anywhere else in the codebase. Every route that needs to enforce a constraint imports from this module.

```typescript
import { z } from 'zod';

// ─── PAY RATE RULES ───────────────────────────────────────────────────────────
export const PAY_RATE_RULES = {
  minimum: '0.01',        // No zero or negative pay rates
  maximum: '999.99',      // Configurable ceiling — prevents data entry errors
  decimalPlaces: 4,       // Matches FinancialCalculator precision
} as const;

export const HOURS_RULES = {

export const SHIFT_TIME_RULES = {
  endMustBeAfterStart: true,
  minimumDurationMinutes: 0,     // Zero-duration allowed for instantaneous entries
  maximumDurationHours: 24,      // No shift longer than 24 hours

  futureLookAheadDays: 365,      // Shifts cannot be scheduled more than 1 year ahead
} as const;

} as const;

// ─── BILLING RATE RULES ──────────────────────────────────────────────────────
export const BILLING_RATE_RULES = {

} as const;

export const PAYROLL_PERIOD_RULES = {

// ─── ROLE ASSIGNMENT RULES ───────────────────────────────────────────────────
export const ROLE_HIERARCHY = ['officer', 'supervisor', 'manager', 'owner'] as const;
export type RoleLevel = typeof ROLE_HIERARCHY[number];

export function getRoleLevel(role: string): number {
  return ROLE_HIERARCHY.indexOf(role as RoleLevel);
}

export function canAssignRole(assignerRole: string, targetRole: string): boolean {
  // Owners can assign any role
  if (assignerRole === 'owner') return true;
  // Others can only assign roles strictly below their own level
  return getRoleLevel(assignerRole) > getRoleLevel(targetRole);
}

// ─── VALIDATION FUNCTIONS ────────────────────────────────────────────────────
import { Decimal } from 'decimal.js';

export interface BusinessRuleViolation {
  field: string;
  rule: string;
  message: string;
  received: string | number | null;
  allowed: string;
}

export function validatePayRate(
  rate: string, 
  fieldName: string = 'payRate'
): BusinessRuleViolation | null {
  const d = new Decimal(rate);
  if (d.lessThan(PAY_RATE_RULES.minimum)) {
    return {
      field: fieldName,
      rule: 'PAY_RATE_MINIMUM',
      message: `Pay rate must be at least $${PAY_RATE_RULES.minimum}`,
      received: rate,
      allowed: `>= ${PAY_RATE_RULES.minimum}`
    };
  }
  if (d.greaterThan(PAY_RATE_RULES.maximum)) {
    return {
      field: fieldName,
      rule: 'PAY_RATE_MAXIMUM',
      message: `Pay rate cannot exceed $${PAY_RATE_RULES.maximum}`,
      received: rate,
      allowed: `<= ${PAY_RATE_RULES.maximum}`
    };
  }
  return null;
}

export function validateShiftTimes(
  startTime: Date,
  endTime: Date
): BusinessRuleViolation | null {
  if (endTime <= startTime) {
    return {
      field: 'endTime',
      rule: 'SHIFT_END_AFTER_START',
      message: 'Shift end time must be after start time',
      received: endTime.toISOString(),
      allowed: `> ${startTime.toISOString()}`
    };
  }
  const durationHours = 
    (endTime.getTime() - startTime.getTime()) / (1000 * 60 * 60);
  if (durationHours > SHIFT_TIME_RULES.maximumDurationHours) {
    return {
      field: 'endTime',
      rule: 'SHIFT_DURATION_MAXIMUM',
      message: `Shift duration cannot exceed ${SHIFT_TIME_RULES.maximumDurationHours} hours`,
      received: `${durationHours.toFixed(2)} hours`,
      allowed: `<= ${SHIFT_TIME_RULES.maximumDurationHours} hours`
    };
  }
  return null;
}

export function validateInvoiceAmount(
  amount: string,
  fieldName: string = 'amount'
): BusinessRuleViolation | null {
  const d = new Decimal(amount);
  if (d.lessThan(INVOICE_RULES.minimumAmount)) {
    return {
      field: fieldName,
      rule: 'INVOICE_AMOUNT_MINIMUM',
      message: `Invoice amount must be at least $${INVOICE_RULES.minimumAmount}`,
      received: amount,
      allowed: `>= ${INVOICE_RULES.minimumAmount}`
    };
  }
  if (d.greaterThan(INVOICE_RULES.maximumAmount)) {
    return {
      field: fieldName,
      rule: 'INVOICE_AMOUNT_MAXIMUM',
      message: `Invoice amount cannot exceed $${INVOICE_RULES.maximumAmount}`,
      received: amount,
      allowed: `<= ${INVOICE_RULES.maximumAmount}`
    };
  }
  return null;
}

export function validateBillingRate(
  rate: string,
  fieldName: string = 'billingRate'
): BusinessRuleViolation | null {
  const d = new Decimal(rate);
  if (d.lessThan(BILLING_RATE_RULES.minimum)) {
    return {
      field: fieldName,
      rule: 'BILLING_RATE_MINIMUM',
      message: `Billing rate must be at least $${BILLING_RATE_RULES.minimum}`,
      received: rate,
      allowed: `>= ${BILLING_RATE_RULES.minimum}`
    };
  }
  if (d.greaterThan(BILLING_RATE_RULES.maximum)) {
    return {
      field: fieldName,
      rule: 'BILLING_RATE_MAXIMUM',
      message: `Billing rate cannot exceed $${BILLING_RATE_RULES.maximum}`,
      received: rate,
      allowed: `<= ${BILLING_RATE_RULES.maximum}`
    };
  }
  return null;
}

export function validatePayrollPeriod(
  startDate: Date,
  endDate: Date
): BusinessRuleViolation | null {
  if (endDate <= startDate) {
    return {
      field: 'periodEnd',
      rule: 'PAYROLL_PERIOD_END_AFTER_START',
      message: 'Payroll period end date must be after start date',
      received: endDate.toISOString(),
      allowed: `> ${startDate.toISOString()}`
    };
  }
  const durationDays = 
    (endDate.getTime() - startDate.getTime()) / (1000 * 60 * 60 * 24);
  if (durationDays > PAYROLL_PERIOD_RULES.maximumPeriodDays) {
    return {
      field: 'periodEnd',
      rule: 'PAYROLL_PERIOD_MAXIMUM_DURATION',
      message: `Payroll period cannot exceed ${PAYROLL_PERIOD_RULES.maximumPeriodDays} days`,
      received: `${durationDays} days`,
      allowed: `<= ${PAYROLL_PERIOD_RULES.maximumPeriodDays} days`
    };
  }
  return null;
}

// ─── MIDDLEWARE HELPER ────────────────────────────────────────────────────────
export function businessRuleResponse(
  res: any,
  violations: (BusinessRuleViolation | null)[]
): boolean {
  const active = violations.filter(Boolean) as BusinessRuleViolation[];
  if (active.length === 0) return false;
  
  res.status(400).json({
    error: 'Business rule violation',
    code: 'BUSINESS_RULE_VIOLATION',
    violations: active.map(v => ({
      field: v.field,
      rule: v.rule,
      message: v.message,
      received: v.received,
      allowed: v.allowed
    }))
  });
  return true;
}
```
