## 📊 DATABASE SCHEMA

```sql
-- Users (Customers & Agents)
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email VARCHAR(255) UNIQUE NOT NULL,
  name VARCHAR(255) NOT NULL,
  role VARCHAR(50) NOT NULL, -- 'customer', 'agent', 'admin'
  avatar_url TEXT,
  online_status VARCHAR(20) DEFAULT 'offline', -- 'online', 'offline', 'away'
  last_seen TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Conversations
CREATE TABLE conversations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  customer_id UUID REFERENCES users(id),
  assigned_agent_id UUID REFERENCES users(id),
  status VARCHAR(50) DEFAULT 'open', -- 'open', 'pending', 'resolved', 'closed'
  priority VARCHAR(20) DEFAULT 'normal', -- 'low', 'normal', 'high', 'urgent'
  channel VARCHAR(50) DEFAULT 'web', -- 'web', 'mobile', 'email'
  ai_confidence_score DECIMAL(5,2),
  issue_category VARCHAR(100),
  metadata JSONB, -- Custom fields
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  resolved_at TIMESTAMP,
  first_response_time INTEGER, -- seconds
  resolution_time INTEGER -- seconds
);

-- Messages
CREATE TABLE messages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  conversation_id UUID REFERENCES conversations(id) ON DELETE CASCADE,
  sender_id UUID REFERENCES users(id),
  sender_type VARCHAR(20) NOT NULL, -- 'customer', 'agent', 'bot'
  content TEXT NOT NULL,
  content_type VARCHAR(50) DEFAULT 'text', -- 'text', 'image', 'file', 'system'
  attachments JSONB, -- [{url, filename, size, type}]
  metadata JSONB, -- AI decision data, etc.
  read_by JSONB DEFAULT '[]', -- [user_id, user_id]
  created_at TIMESTAMP DEFAULT NOW()
);

-- AI Decisions Log
CREATE TABLE ai_decisions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  conversation_id UUID REFERENCES conversations(id),
  decision_type VARCHAR(100), -- 'route', 'response', 'escalate'
  confidence_score DECIMAL(5,2),
  input_data JSONB,
  output_data JSONB,
  human_override BOOLEAN DEFAULT false,
  override_reason TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Agent Performance
CREATE TABLE agent_metrics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  agent_id UUID REFERENCES users(id),
  date DATE NOT NULL,
  conversations_handled INTEGER DEFAULT 0,
  avg_response_time INTEGER, -- seconds
  avg_resolution_time INTEGER, -- seconds
  customer_satisfaction DECIMAL(3,2), -- 1-5 rating
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(agent_id, date)
);

-- Canned Responses
CREATE TABLE canned_responses (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title VARCHAR(255) NOT NULL,
  content TEXT NOT NULL,
  category VARCHAR(100),
  shortcut VARCHAR(50), -- e.g., '/login-help'
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW()
);

-- Tags
CREATE TABLE tags (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) UNIQUE NOT NULL,
  color VARCHAR(7), -- hex color
  created_at TIMESTAMP DEFAULT NOW()
);

-- Conversation Tags (many-to-many)
CREATE TABLE conversation_tags (
  conversation_id UUID REFERENCES conversations(id) ON DELETE CASCADE,
  tag_id UUID REFERENCES tags(id) ON DELETE CASCADE,
  PRIMARY KEY (conversation_id, tag_id)
);

-- Indexes for performance
CREATE INDEX idx_conversations_customer ON conversations(customer_id);
CREATE INDEX idx_conversations_agent ON conversations(assigned_agent_id);
CREATE INDEX idx_conversations_status ON conversations(status);
CREATE INDEX idx_messages_conversation ON messages(conversation_id);
CREATE INDEX idx_messages_created ON messages(created_at DESC);
CREATE INDEX idx_ai_decisions_conversation ON ai_decisions(conversation_id);
```

## 🤖 AI INTEGRATION WORKFLOW

```javascript
// AI Decision Engine
class AIRouter {
  async processMessage(conversationId, message) {
    // 1. Analyze message intent
    const intent = await this.analyzeIntent(message);
    
    // 2. Check confidence threshold
    const confidence = await this.calculateConfidence(conversationId);
    
    // 3. Decide action
    if (confidence >= 95) {
      // Full autonomy - AI responds directly
      return await this.generateAIResponse(conversationId, intent);
    } 
    else if (confidence >= 75) {
      // Supervised - AI suggests, agent approves
      return await this.suggestResponse(conversationId, intent);
    } 
    else {
      // Training - Escalate to human immediately
      return await this.escalateToHuman(conversationId, intent);
    }
  }
  
  async escalateToHuman(conversationId, intent) {
    // 1. Find available agent
    const agent = await this.findBestAgent(intent.category);
    
    // 2. Create handoff summary
    const summary = await this.generateHandoffSummary(conversationId);
    
    // 3. Assign conversation
    await this.assignConversation(conversationId, agent.id);
    
    // 4. Notify agent
    socket.to(agent.socketId).emit('conversation:new', {
      conversationId,
      summary,
      aiConfidence: intent.confidence,
      recommendedActions: summary.recommendations
    });
    
    // 5. Notify customer
    socket.to(conversationId).emit('ai:escalate', {
      message: 'Connecting you with a live agent...',
      estimatedWaitTime: await this.getAverageWaitTime()
    });
  }
}
```
