To make this system production-ready and safe (not just a demo), you need three critical "invisible" layers. Without these, your AI is a "loose cannon" that could hallucinate a refund or delete data.

The Commitment Manager (Safety Layer): To prevent the AI from taking dangerous actions without approval.

The Observability Layer (Black Box Recorder): To see why the AI made a decision.

The Identity Layer (System Prompt): To ensure the AI behaves like a corporate employee, not a creative writer.

Here is the updated code integrating these three missing pieces.

from database import Base

class ToolDefinition(Base):
    __tablename__ = 'tool_definitions'

    parameters_schema = Column(JSON, nullable=False)

    __tablename__ = 'workflows'
    id = Column(Integer, primary_key=True, index=True)
    user_id = Column(Integer, index=True)
    request_text = Column(String)
    # Status can now be: pending, running, waiting_for_approval, completed, failed
    status = Column(String, default="pending") 
    plan_data = Column(JSON) 
    created_at = Column(DateTime(timezone=True), server_default=func.now())

# NEW: The Black Box Recorder
class AuditLog(Base):
    __tablename__ = 'audit_logs'
    id = Column(Integer, primary_key=True, index=True)
    workflow_id = Column(Integer, ForeignKey('workflows.id'))
    step_description = Column(String) # e.g., "AI decided to call send_email"
    tool_used = Column(String)
    tool_input = Column(JSON)
    tool_output = Column(JSON)
    timestamp = Column(DateTime(timezone=True), server_default=func.now())

SENSITIVE_TOOLS = ["send_email_notification", "refund_payment"]

class CommitmentManager:
    @staticmethod
    def is_action_allowed(tool_name, arguments, user_role="admin"):
        """
        Decides if the AI is allowed to do this RIGHT NOW.
        """
        if tool_name in SENSITIVE_TOOLS:
            # Logic: If it's sensitive, we might need manual approval or check limits.
            # For this demo, we'll log a warning but allow it (or return False to block).
            print(f"⚠️ [COMMITMENT MANAGER] Reviewing sensitive action: {tool_name}")
            
            # Example Policy: Block emails to non-company domains
            if tool_name == "send_email_notification" and "@company.com" not in arguments.get("recipient", ""):
                return False, "Security Policy Violation: Cannot email external domains without manual approval."
                
        return True, "Approved"

# --- TOOLS ---

def send_email_notification(recipient: str, subject: str, body: str):
    """
    Sends an email notification.
    """
    # 1. CHECK COMMITMENT
    allowed, reason = CommitmentManager.is_action_allowed("send_email_notification", locals())

        return {"status": "blocked", "error": reason}

    # 2. EXECUTE
    print(f"\n📧 [SYSTEM ACTION] Sending Email to {recipient}")
    return {"status": "sent", "recipient": recipient}

We give the AI a "Personality" so it adheres to rules, and we log its thoughts.

import os
import google.generativeai as genai
from sqlalchemy.orm import Session
from models import Workflow, AuditLog # Import AuditLog
from tools_impl import AVAILABLE_TOOLS

API_KEY = os.environ.get("GEMINI_API_KEY")
if API_KEY:
    genai.configure(api_key=API_KEY)

# NEW: System Instruction (The "Constitution")
SYSTEM_INSTRUCTION = """
You are the Orchestration Brain for a corporate automation system.
1. You must be precise and professional.
2. DO NOT hallucinate data. If you cannot find info in the database, stop and ask the user.
3. If a tool returns a "Security Policy Violation", inform the user clearly.
"""

class AIEngineService:
    def __init__(self, db: Session):
        self.db = db
        self.model = genai.GenerativeModel(
            model_name='gemini-1.5-flash',
            tools=AVAILABLE_TOOLS,
            system_instruction=SYSTEM_INSTRUCTION # Apply identity
        )

    def process_request(self, user_id: int, request_text: str):
        workflow = Workflow(user_id=user_id, request_text=request_text, status="running")
        self.db.add(workflow)
        self.db.commit()

        chat = self.model.start_chat(enable_automatic_function_calling=True)

        try:
            response = chat.send_message(request_text)

            # NEW: Log the "Thought Process" (History) into Audit Logs
            for part in chat.history:
                if part.role == "model" and part.parts[0].function_call:
                    fc = part.parts[0].function_call
                    log = AuditLog(
                        workflow_id=workflow.id,
                        step_description="AI invoked tool",
                        tool_used=fc.name,
                        tool_input=str(dict(fc.args)),
                        tool_output="Check Tools Impl" # Simplified for demo
                    )
                    self.db.add(log)
            
            workflow.status = "completed"
            workflow.plan_data = {"ai_response": response.text}
            self.db.commit()
            return workflow

        except Exception as e:
            workflow.status = "failed"
            workflow.plan_data = {"error": str(e)}
            self.db.commit()
            raise e

@app.get("/workflow/{workflow_id}/logs")
def get_workflow_logs(workflow_id: int, db: Session = Depends(get_db)):
    """
    Observability: See exactly what the AI did and what tools it called.
    """
    logs = db.query(AuditLog).filter(AuditLog.workflow_id == workflow_id).all()
    return {"workflow_id": workflow_id, "logs": logs}
