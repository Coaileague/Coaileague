CREATE TABLE IF NOT EXISTS platform_email_addresses (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  -- Ownership
  workspace_id UUID REFERENCES workspaces(id),
  user_id UUID REFERENCES users(id),
  client_id UUID REFERENCES clients(id),

  -- The address
  address VARCHAR(320) NOT NULL UNIQUE,
  local_part VARCHAR(255) NOT NULL,    -- everything before @
  subdomain VARCHAR(100),              -- slug for tenant addresses

  -- Type

  -- 'user_personal'    — john.smith@acme.coaileague.com (billed per seat)
  -- 'user_client'      — client@acme.coaileague.com (billed per seat)

  -- 'platform_team'    — trinity@coaileague.com (platform reserved)
  -- 'protected_support' — name@support.coaileague.com (protected subdomain)

  -- Status
  is_active BOOLEAN DEFAULT false,
  is_protected BOOLEAN DEFAULT false,    -- cannot be created by tenants
  is_outbound_only BOOLEAN DEFAULT false, -- noreply — no inbound routing
  activated_at TIMESTAMPTZ,
  activated_by UUID REFERENCES users(id),
  deactivated_at TIMESTAMPTZ,

  -- Billing
  billing_seat_id VARCHAR(255),          -- links to Stripe subscription item
  fair_use_monthly_limit INTEGER DEFAULT 500, -- emails per month included
  emails_sent_this_period INTEGER DEFAULT 0,
  emails_received_this_period INTEGER DEFAULT 0,

  -- Trinity processing
  auto_trinity_process BOOLEAN DEFAULT false,
  trinity_calltype VARCHAR(100),

  -- Timestamps
  created_at TIMESTAMPTZ DEFAULT NOW(),

-- ============================================================
-- EMAIL ROUTING TABLE
-- Maps every inbound address to correct handler
-- Resend webhook looks up this table to route
-- ============================================================

CREATE TABLE IF NOT EXISTS email_routing (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  address VARCHAR(320) NOT NULL UNIQUE,
  email_address_id UUID REFERENCES platform_email_addresses(id),

  route_type VARCHAR(50) NOT NULL,
  -- 'user_inbox'       — deliver to user's mailbox
  -- 'workspace_inbox'  — deliver to workspace shared inbox
  -- 'trinity_process'  — Trinity processes automatically
  -- 'support_queue'    — platform support queue
  -- 'discard'          — accept but discard (spam protection)

  target_user_id UUID REFERENCES users(id),
  target_inbox_type VARCHAR(100),

  auto_process BOOLEAN DEFAULT false,
  process_as VARCHAR(100),
  -- 'calloff', 'incident', 'support_ticket', 'application',
  -- 'client_message', 'general'

  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_email_routing_address
  ON email_routing(address);

-- EMAILS TABLE

  workspace_id UUID REFERENCES workspaces(id),

  from_name VARCHAR(255),

  bcc_addresses TEXT[] DEFAULT '{}',

  body_html TEXT,
  body_text TEXT,

  folder VARCHAR(50) DEFAULT 'inbox',
  -- 'inbox', 'sent', 'drafts', 'starred', 'archive', 'trash', 'spam'

  is_starred BOOLEAN DEFAULT false,
  is_archived BOOLEAN DEFAULT false,
  is_deleted BOOLEAN DEFAULT false,

import { pool } from '../../db';

export class EmailProvisioningService {

  // Called automatically after workspace registration
  async provisionWorkspaceAddresses(
    workspaceId: string,

  ): Promise<void> {
    const domain = `${emailSlug}.coaileague.com`;

    // Update workspace with email domain

      },

        local: `calloffs-${emailSlug}`,
        full: `calloffs-${emailSlug}@coaileague.com`,
        displayName: 'Calloffs',
        trinityType: 'calloff',
        autoProcess: true,
      },
      {
        local: `incidents-${emailSlug}`,
        full: `incidents-${emailSlug}@coaileague.com`,
        displayName: 'Incidents',
        trinityType: 'incident',
        autoProcess: true,
      },
      {
        local: `support-${emailSlug}`,
        full: `support-${emailSlug}@coaileague.com`,
        displayName: 'Support',
        trinityType: 'support_ticket',
        autoProcess: true,
      },
      {
        local: `docs-${emailSlug}`,
        full: `docs-${emailSlug}@coaileague.com`,

        trinityType: 'document_submission',

        local: `billing-${emailSlug}`,
        full: `billing-${emailSlug}@coaileague.com`,
        displayName: 'Billing',
        trinityType: 'billing_inquiry',
        autoProcess: true,
      },
    ];

    for (const addr of systemAddresses) {
      await pool.query(`
        INSERT INTO platform_email_addresses (
          workspace_id, address, local_part, subdomain, display_name,
          address_type, is_active, auto_trinity_process, trinity_calltype
        ) VALUES ($1, $2, $3, $4, $5, 'workspace_system', true, $6, $7)
        ON CONFLICT (address) DO NOTHING
      `, [
        workspaceId, addr.full, addr.local, emailSlug,
        addr.displayName, addr.autoProcess, addr.trinityType
      ]);

      await pool.query(`
        INSERT INTO email_routing (
          address, route_type, target_workspace_id,
          auto_process, process_as
        ) VALUES ($1, 'trinity_process', $2, $3, $4)
        ON CONFLICT (address) DO NOTHING
      `, [addr.full, workspaceId, addr.autoProcess, addr.trinityType]);
    }
  }

  // Called when org owner creates a new officer
  async reserveUserEmailAddress(
    workspaceId: string,
    userId: string,
    firstName: string,
    lastName: string,
    emailSlug: string
  ): Promise<string> {
    const localPart = `${firstName.toLowerCase()}.${lastName.toLowerCase()}`
      .replace(/[^a-z0-9.]/g, '');
    const address = `${localPart}@${emailSlug}.coaileague.com`;

    // Reserve address — not active until org owner activates
    await pool.query(`
      INSERT INTO platform_email_addresses (
        workspace_id, user_id, address, local_part, subdomain,
        display_name, address_type, is_active
      ) VALUES ($1, $2, $3, $4, $5, $6, 'user_personal', false)
      ON CONFLICT (address) DO UPDATE SET
        user_id = EXCLUDED.user_id,
        display_name = EXCLUDED.display_name
    `, [
      workspaceId, userId, address, localPart, emailSlug,
      `${firstName} ${lastName}`
    ]);

    // Update user record with reserved address
    await pool.query(
      `UPDATE users SET platform_email = $1 WHERE id = $2`,
      [address, userId]
    );

    return address;
  }

  // Called when org owner creates a client
  async reserveClientEmailAddress(
    workspaceId: string,
    clientId: string,
    clientName: string,
    emailSlug: string
  ): Promise<string> {
    const localPart = clientName.toLowerCase()
      .replace(/[^a-z0-9]/g, '').slice(0, 20);
    const address = `${localPart}@${emailSlug}.coaileague.com`;

    await pool.query(`
      INSERT INTO platform_email_addresses (
        workspace_id, client_id, address, local_part, subdomain,
        display_name, address_type, is_active
      ) VALUES ($1, $2, $3, $4, $5, $6, 'user_client', false)
      ON CONFLICT (address) DO NOTHING
    `, [
      workspaceId, clientId, address, localPart, emailSlug, clientName
    ]);

    await pool.query(
      `UPDATE clients SET platform_email = $1 WHERE id = $2`,
      [address, clientId]
    );

    return address;
  }

  // Called when org owner activates an email seat
  async activateEmailAddress(
    emailAddressId: string,
    activatedByUserId: string,
    stripeSubscriptionItemId: string
  ): Promise<void> {
    const client = await pool.connect();
    try {
      await client.query('BEGIN');

      // Activate address
      const result = await client.query(`
        UPDATE platform_email_addresses SET
          is_active = true,
          activated_at = NOW(),
          activated_by = $1,
          billing_seat_id = $2
        WHERE id = $3
        RETURNING address, route_type, target_workspace_id, address_type,
                  user_id, workspace_id
      `, [activatedByUserId, stripeSubscriptionItemId, emailAddressId]);

      if (!result.rows[0]) throw new Error('Email address not found');
      const addr = result.rows[0];

      // Create routing entry
      await client.query(`
        INSERT INTO email_routing (
          address, email_address_id, route_type,
          target_workspace_id, target_user_id, is_active
        ) VALUES ($1, $2, 'user_inbox', $3, $4, true)
        ON CONFLICT (address) DO UPDATE SET
          is_active = true,
          target_user_id = EXCLUDED.target_user_id
      `, [
        addr.address, emailAddressId,
        addr.workspace_id, addr.user_id
      ]);

      // Write audit log
      await client.query(`
        INSERT INTO universal_audit_log (
          workspace_id, entity_type, entity_id,
          action_type, actor_id, new_value
        ) VALUES ($1, 'email_address', $2, 'email_activated', $3, $4)
      `, [
        addr.workspace_id, emailAddressId,
        activatedByUserId,
        JSON.stringify({ address: addr.address })
      ]);

      await client.query('COMMIT');
    } catch (err) {
      await client.query('ROLLBACK');
      throw err;
    } finally {
      client.release();
    }
  }

  // Called when org owner deactivates an email seat
  async deactivateEmailAddress(
    emailAddressId: string,
    deactivatedByUserId: string
  ): Promise<void> {
    await pool.query(`
      UPDATE platform_email_addresses SET
        is_active = false,
        deactivated_at = NOW()
      WHERE id = $1
    `, [emailAddressId]);

    await pool.query(`
      UPDATE email_routing SET is_active = false
      WHERE email_address_id = $1
    `, [emailAddressId]);
  }
}

import { emailProvisioningService } from '../../services/email/emailProvisioningService';
import { pool } from '../../db';
import { verifyResendWebhookSignature } from '../../utils/resendWebhook';
import { notificationDeliveryService } from '../../services/notificationDeliveryService';
import { aiMeteringService } from '../../services/billing/aiMeteringService';
import { withGemini } from '../../services/ai/aiCallWrapper';

router.post('/api/inbound/email', async (req, res) => {
  // 1. Verify Resend webhook signature
  const signature = req.headers['resend-signature'] as string;
  const isValid = verifyResendWebhookSignature(
    req.rawBody, signature, process.env.RESEND_WEBHOOK_SECRET!
  );
  if (!isValid) {
    return res.status(401).json({ error: 'Invalid webhook signature' });
  }

  // 2. Deduplicate by Resend email ID
  const resendEmailId = req.body.email_id || req.body.id;
  if (resendEmailId) {
    const existing = await pool.query(
      `SELECT id FROM platform_emails WHERE resend_email_id = $1`,
      [resendEmailId]
    );
    if (existing.rows[0]) {
      return res.status(200).json({ status: 'already_processed' });
    }
  }

  // 3. Parse inbound email payload from Resend
  const {
    from, to, cc, subject,
    html, text, message_id, in_reply_to,
    attachments = [],
  } = req.body;

  // 4. Determine routing for each To address
  const toAddresses = Array.isArray(to) ? to : [to];
  let primaryRoute = null;

  for (const toAddress of toAddresses) {
    const clean = toAddress.toLowerCase().trim();
    const route = await pool.query(
      `SELECT er.*, pea.workspace_id as email_workspace_id,
              pea.user_id as email_user_id, pea.address_type,
              pea.auto_trinity_process, pea.trinity_calltype,
              pea.is_active
       FROM email_routing er
       JOIN platform_email_addresses pea ON pea.id = er.email_address_id
       WHERE er.address = $1 AND er.is_active = true`,
      [clean]
    );

    if (route.rows[0]) {
      primaryRoute = route.rows[0];
      break;
    }
  }

  // 5. If no route found — accept but discard (prevents bounces)
  if (!primaryRoute) {
    return res.status(200).json({ status: 'no_route' });
  }

  // 6. Store email in platform_emails
  const snippet = (text || '').slice(0, 200);
  const emailResult = await pool.query(`
    INSERT INTO platform_emails (
      workspace_id, resend_email_id, message_id, in_reply_to,
      direction, from_address, to_addresses, cc_addresses,
      subject, body_html, body_text, snippet,
      owner_user_id, folder, received_at
    ) VALUES (
      $1,$2,$3,$4,'inbound',$5,$6,$7,$8,$9,$10,$11,$12,'inbox',NOW()
    )
    RETURNING id
  `, [
    primaryRoute.email_workspace_id,
    resendEmailId,
    message_id, in_reply_to,
    from, toAddresses,
    Array.isArray(cc) ? cc : [],
    subject, html, text, snippet,
    primaryRoute.email_user_id || null,
  ]);

  const emailId = emailResult.rows[0].id;

  // 7. Store attachments
  if (attachments.length > 0) {
    for (const att of attachments) {
      await pool.query(`
        INSERT INTO email_attachments (
          email_id, filename, content_type, size_bytes
        ) VALUES ($1, $2, $3, $4)
      `, [emailId, att.filename, att.content_type, att.size]);
    }
  }

  // 8. Route to correct handler
  if (primaryRoute.auto_trinity_process && primaryRoute.process_as) {
    // Trinity processes automatically — fire and forget
    processTrinityEmail({
      emailId,
      workspaceId: primaryRoute.email_workspace_id,
      subject: subject || '',
      body: text || '',
      from,
      processAs: primaryRoute.process_as,
      callType: primaryRoute.trinity_calltype,
    }).catch(err => console.error('Trinity email processing error:', err));
  } else {
    // User inbox delivery — notify via WebSocket
    broadcastToWorkspace(primaryRoute.email_workspace_id, {
      type: 'new_email',
      emailId,
      from,
      subject: subject || '(no subject)',
      snippet,
      userId: primaryRoute.email_user_id,
    });

    // In-app notification
    if (primaryRoute.email_user_id) {
      await notificationDeliveryService.send({
        workspaceId: primaryRoute.email_workspace_id,
        type: 'new_email_received' as any,
        targetUserIds: [primaryRoute.email_user_id],
        data: { from, subject, emailId }
      });
    }
  }

  // 9. Acknowledge immediately to Resend
  return res.status(200).json({ status: 'accepted', emailId });
});

// Trinity auto-processes specific email types
async function processTrinityEmail(params: {
  emailId: string;
  workspaceId: string;
  subject: string;
  body: string;
  from: string;
  processAs: string;
  callType: string;
}) {
  // Get workspace tier for AI metering
  const ws = await pool.query(
    `SELECT tier FROM workspaces WHERE id = $1`, [params.workspaceId]
  );
  const tier = ws.rows[0]?.tier || 'starter';

  // AI classification and processing
  const result = await withGemini(
    'gemini-2.5-flash',
    {
      workspaceId: params.workspaceId,
      tier,
      callType: params.callType || 'email_classification',
    },
    async () => {
      // Gemini classifies intent and extracts key data
      const prompt = `
        Classify this email and extract key information.
        Process type: ${params.processAs}
        From: ${params.from}
        Subject: ${params.subject}
        Body: ${params.body.slice(0, 2000)}

        Return JSON:
        {
          "intent": "${params.processAs}",
          "summary": "one line summary",
          "priority": "urgent|high|normal|low",
          "extractedData": {},
          "suggestedAction": "what Trinity should do",
          "draftReply": "suggested reply text"
        }
      `;
      // actual Gemini call here
      const raw = await geminiModel.generateContent(prompt);
      return {
        result: raw.response.text(),
        rawResponse: raw,
      };
    }
  );

  // Parse result and take action based on processAs type
  let actionTaken = '';
  let actionRecordId = null;
  let parsedResult: any = {};

  try {
    parsedResult = JSON.parse(result);
  } catch {
    parsedResult = { summary: 'Email received', priority: 'normal' };
  }

  // Route to correct Trinity action based on processAs
  if (params.processAs === 'calloff') {
    // Create calloff record
    const calloff = await pool.query(`
      INSERT INTO calloff_records (
        workspace_id, source, source_email_id,
        raw_content, status
      ) VALUES ($1, 'email', $2, $3, 'pending')
      RETURNING id
    `, [params.workspaceId, params.emailId, params.body]);
    actionRecordId = calloff.rows[0]?.id;
    actionTaken = 'calloff_created';

  } else if (params.processAs === 'incident') {
    // Create incident record
    const incident = await pool.query(`
      INSERT INTO incident_reports (
        workspace_id, source, source_email_id,
        description, status
      ) VALUES ($1, 'email', $2, $3, 'open')
      RETURNING id
    `, [params.workspaceId, params.emailId, params.body]);
    actionRecordId = incident.rows[0]?.id;
    actionTaken = 'incident_created';

  } else if (params.processAs === 'support_ticket') {
    // Create support ticket
    const ticket = await pool.query(`
      INSERT INTO support_tickets (
        workspace_id, source, source_email_id,
        description, status, priority
      ) VALUES ($1, 'email', $2, $3, 'open', $4)
      RETURNING id
    `, [
      params.workspaceId, params.emailId,
      params.body, parsedResult.priority || 'normal'
    ]);
    actionRecordId = ticket.rows[0]?.id;
    actionTaken = 'support_ticket_created';
  }

  // Update email with Trinity processing results
  await pool.query(`
    UPDATE platform_emails SET
      trinity_processed = true,
      trinity_processed_at = NOW(),
      trinity_category = $1,
      trinity_summary = $2,
      trinity_action_taken = $3,
      trinity_action_record_id = $4,
      trinity_draft_reply = $5,
      trinity_priority = $6
    WHERE id = $7
  `, [
    params.processAs,
    parsedResult.summary,
    actionTaken,
    actionRecordId,
    parsedResult.draftReply,
    parsedResult.priority || 'normal',
    params.emailId,
  ]);

  // Notify org owner of auto-processed email
  await notificationDeliveryService.send({
    workspaceId: params.workspaceId,
    type: 'trinity_email_processed' as any,
    targetRoles: ['org_owner'],
    data: {
      from: params.from,
      processAs: params.processAs,
      actionTaken,
      summary: parsedResult.summary,
    }
  });
}
