async verifyBankAccount(accessToken: string): Promise<{ valid: boolean, status: string }> {

**P1-FIX-5 — retry-failed-transfers endpoint**
In `payrollRoutes.ts` add after the mark-paid endpoint:
```typescript
router.post('/runs/:id/retry-failed-transfers', requireAuth, requireRole(['org_owner', 'org_manager']), async (req, res) => {
  const { id } = req.params
  const workspaceId = req.workspaceId
  try {
    const failedStubs = await db.select()
      .from(payStubs)
      .where(and(
        eq(payStubs.payrollRunId, id),
        eq(payStubs.workspaceId, workspaceId),
        eq(payStubs.status, 'failed')
      ))
    if (failedStubs.length === 0) {
      return res.json({ success: true, message: 'No failed transfers to retry', retried: 0 })
    }
    const results = []
    for (const stub of failedStubs) {
      try {
        const employee = await db.select().from(employeeBankAccounts)
          .where(and(
            eq(employeeBankAccounts.employeeId, stub.employeeId),
            eq(employeeBankAccounts.workspaceId, workspaceId),
            eq(employeeBankAccounts.isPrimary, true)
          )).limit(1)
        if (!employee[0]?.plaidAccessTokenEncrypted) {
          results.push({ employeeId: stub.employeeId, status: 'skipped', reason: 'no_plaid_account' })
          continue
        }
        const accessToken = decrypt(employee[0].plaidAccessTokenEncrypted)
        const transfer = await plaidService.initiateTransfer({
          accessToken,
          accountId: employee[0].plaidItemId,
          amount: Number(stub.netPay),
          description: `Payroll retry - ${stub.payPeriodStart}`,
          workspaceId
        })
        await db.update(payStubs)
          .set({ status: 'transfer_initiated', plaidTransferId: transfer.transferId, consecutivePollFailures: 0 })
          .where(eq(payStubs.id, stub.id))
        results.push({ employeeId: stub.employeeId, status: 'retried', transferId: transfer.transferId })
      } catch (err) {
        results.push({ employeeId: stub.employeeId, status: 'retry_failed', reason: err.message })
      }
    }
    res.json({ success: true, retried: results.filter(r => r.status === 'retried').length, results })
  } catch (err) {
    res.status(500).json({ error: 'Failed to retry transfers', details: err.message })
  }
})
```
