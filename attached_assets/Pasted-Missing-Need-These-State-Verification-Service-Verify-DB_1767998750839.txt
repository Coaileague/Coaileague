class StateVerificationService {
  async verifyActionResult(action, expectedOutcome) {
    // Query database to check actual state
    const actualState = await this.queryDatabaseState(action);
    
    if (!statesMatch(actualState, expectedOutcome)) {
      return {
        verified: false,
        discrepancy: findDiscrepancy(actualState, expectedOutcome),
        needsRollback: true
      };
    }
    
    return { verified: true };
  }
  
  private async queryDatabaseState(action) {
    switch (action.type) {
      case 'ASSIGN_GUARD':
        return await db.shiftAssignments.findUnique({
          where: { id: action.targetId }
        });
      case 'CREATE_SHIFT':
        return await db.shifts.findUnique({
          where: { id: action.targetId }
        });
      // Add other action types
    }
  }
}

class AlternativeStrategyService {
  async generateAlternatives(failedAction, error, context) {
    // Use existing agent parity layer
    const alternatives = await agentParityLayer.reflect({
      action: failedAction,
      error,
      context
    });
    
    // Rank by success probability
    return alternatives
      .map(alt => ({
        ...alt,
        probability: this.calculateSuccessProbability(alt, context),
        estimatedCost: this.estimateCost(alt, context)
      }))
      .sort((a, b) => b.probability - a.probability);
  }
}

class BusinessImpactCalculator {
  async calculateImpact(actions, context) {
    let totalLaborCost = 0;
    let totalBilling = 0;
    let peopleAffected = new Set();
    
    for (const action of actions) {
      if (action.type === 'ASSIGN_GUARD') {
        const assignment = await db.shiftAssignments.findUnique({
          where: { id: action.targetId },
          include: { shift: true, guard: true }
        });
        
        totalLaborCost += assignment.shift.hours * assignment.guard.hourlyRate;
        totalBilling += assignment.shift.hours * assignment.shift.billingRate;
        peopleAffected.add(assignment.guard.id);
      }
    }
    
    return {
      cost: totalLaborCost,
      billing: totalBilling,
      netProfit: totalBilling - totalLaborCost,
      timeSaved: this.estimateTimeSaved(actions),
      peopleAffected: peopleAffected.size,
      compliance: await this.checkCompliance(actions)
    };
  }
}
