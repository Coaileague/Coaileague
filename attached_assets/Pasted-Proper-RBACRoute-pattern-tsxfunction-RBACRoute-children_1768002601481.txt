Proper RBACRoute pattern:tsxfunction RBACRoute({ children, requiredPermission, publicPath = false }) {
  const { user, isAuthenticated, isLoading } = useAuth();
  
  // 1. Show loading spinner during auth check
  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-screen">
        <Spinner /> {/* Or TrinityRedesign with loading animation */}
      </div>
    );
  }
  
  // 2. Public paths render immediately (no auth needed)
  if (publicPath) {
    return children;
  }
  
  // 3. Not authenticated? Redirect to login
  if (!isAuthenticated) {
    return <Navigate to="/login" replace />;
  }
  
  // 4. Check permission
  if (requiredPermission && !hasPermission(user, requiredPermission)) {
    return (
      <div className="flex items-center justify-center h-screen">
        <Alert variant="destructive">
          <AlertTitle>Access Denied</AlertTitle>
          <AlertDescription>
            You don't have permission to view this page.
          </AlertDescription>
        </Alert>
      </div>
    );
  }
  
  // 5. All checks passed - render children
  return children;
}🎯 KEY CHANGES NEEDED1. Never Return null or false from Route Guards
