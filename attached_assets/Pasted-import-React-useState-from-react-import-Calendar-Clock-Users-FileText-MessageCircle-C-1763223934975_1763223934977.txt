import React, { useState } from 'react';
import { Calendar, Clock, Users, FileText, MessageCircle, Camera, CheckSquare, Plus, X, Menu, Bell, Shield, ChevronLeft, ChevronRight } from 'lucide-react';

export default function WorkforceManagerSchedule() {
  const [activeTab, setActiveTab] = useState('schedule');
  const [viewMode, setViewMode] = useState('day'); // 'day' or 'week'
  const [selectedDate, setSelectedDate] = useState(new Date());
  const [showShiftModal, setShowShiftModal] = useState(false);
  const [selectedShift, setSelectedShift] = useState(null);
  const [shifts, setShifts] = useState([
    {
      id: 1,
      employee: 'John Doe',
      position: 'Security Officer',
      client: 'Tech Corp Building A',
      date: '2025-11-14',
      clockIn: '08:00',
      clockOut: '16:00',
      status: 'scheduled',
      notes: 'Main entrance patrol',
      postOrders: true,
      requirePicsHourly: true,
      requireDAR: true,
      chatId: 'shift-001'
    },
    {
      id: 2,
      employee: 'Jane Smith',
      position: 'Supervisor',
      client: 'Tech Corp Building B',
      date: '2025-11-14',
      clockIn: '12:00',
      clockOut: '20:00',
      status: 'active',
      notes: 'Overnight coverage',
      postOrders: true,
      requirePicsHourly: false,
      requireDAR: true,
      chatId: 'shift-002'
    }
  ]);

  // RBAC-based tabs
  const tabs = [
    { id: 'dashboard', label: 'Dashboard', icon: Shield, role: 'all' },
    { id: 'schedule', label: 'Schedule', icon: Calendar, role: 'all' },
    { id: 'team', label: 'Team', icon: Users, role: 'manager' },
    { id: 'reports', label: 'Reports', icon: FileText, role: 'manager' },
    { id: 'approvals', label: 'Approvals', icon: CheckSquare, role: 'admin' },
  ];

  const timeSlots = Array.from({ length: 24 }, (_, i) => {
    const hour = i.toString().padStart(2, '0');
    return `${hour}:00`;
  });

  const employees = [
    'John Doe',
    'Jane Smith',
    'Mike Johnson',
    'Sarah Williams',
    'Tom Brown',
    'Emily Davis'
  ];

  const positions = [
    'Security Officer',
    'Supervisor',
    'Senior Officer',
    'Site Manager',
    'Patrol Officer'
  ];

  const clients = [
    'Tech Corp Building A',
    'Tech Corp Building B',
    'Medical Center',
    'Shopping Mall',
    'Corporate Plaza'
  ];

  const formatDate = (date) => {
    return date.toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
  };

  const getWeekDates = () => {
    const week = [];
    const start = new Date(selectedDate);
    start.setDate(start.getDate() - start.getDay());
    
    for (let i = 0; i < 7; i++) {
      const date = new Date(start);
      date.setDate(start.getDate() + i);
      week.push(date);
    }
    return week;
  };

  const changeDate = (days) => {
    const newDate = new Date(selectedDate);
    newDate.setDate(newDate.getDate() + days);
    setSelectedDate(newDate);
  };

  const handleCellClick = (time, employee) => {
    setSelectedShift({
      employee,
      position: '',
      client: '',
      date: selectedDate.toISOString().split('T')[0],
      clockIn: time,
      clockOut: '',
      notes: '',
      postOrders: false,
      requirePicsHourly: false,
      requireDAR: false,
    });
    setShowShiftModal(true);
  };

  const handleSaveShift = () => {
    if (selectedShift.id) {
      setShifts(shifts.map(s => s.id === selectedShift.id ? selectedShift : s));
    } else {
      const newShift = {
        ...selectedShift,
        id: Date.now(),
        status: 'scheduled',
        chatId: `shift-${Date.now()}`
      };
      setShifts([...shifts, newShift]);
    }
    setShowShiftModal(false);
    setSelectedShift(null);
  };

  const ShiftModal = () => (
    <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4">
      <div className="bg-white rounded-xl max-w-2xl w-full max-h-[90vh] overflow-y-auto">
        <div className="sticky top-0 bg-gradient-to-r from-blue-600 to-purple-600 text-white p-6 flex justify-between items-center">
          <h2 className="text-2xl font-bold flex items-center gap-2">
            <Calendar className="w-6 h-6" />
            {selectedShift?.id ? 'Edit Shift' : 'Add New Shift'}
          </h2>
          <button onClick={() => setShowShiftModal(false)} className="text-white hover:bg-white hover:bg-opacity-20 rounded-full p-2">
            <X className="w-6 h-6" />
          </button>
        </div>

        <div className="p-6 space-y-4">
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">Employee</label>
              <select
                value={selectedShift?.employee || ''}
                onChange={(e) => setSelectedShift({ ...selectedShift, employee: e.target.value })}
                className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
              >
                <option value="">Select Employee</option>
                {employees.map(emp => (
                  <option key={emp} value={emp}>{emp}</option>
                ))}
              </select>
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">Position/Type</label>
              <select
                value={selectedShift?.position || ''}
                onChange={(e) => setSelectedShift({ ...selectedShift, position: e.target.value })}
                className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
              >
                <option value="">Select Position</option>
                {positions.map(pos => (
                  <option key={pos} value={pos}>{pos}</option>
                ))}
              </select>
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">Client</label>
              <select
                value={selectedShift?.client || ''}
                onChange={(e) => setSelectedShift({ ...selectedShift, client: e.target.value })}
                className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
              >
                <option value="">Select Client</option>
                {clients.map(client => (
                  <option key={client} value={client}>{client}</option>
                ))}
              </select>
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">Date</label>
              <input
                type="date"
                value={selectedShift?.date || ''}
                onChange={(e) => setSelectedShift({ ...selectedShift, date: e.target.value })}
                className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
              />
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">Clock In Time</label>
              <input
                type="time"
                value={selectedShift?.clockIn || ''}
                onChange={(e) => setSelectedShift({ ...selectedShift, clockIn: e.target.value })}
                className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
              />
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">Clock Out Time</label>
              <input
                type="time"
                value={selectedShift?.clockOut || ''}
                onChange={(e) => setSelectedShift({ ...selectedShift, clockOut: e.target.value })}
                className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
              />
            </div>
          </div>

          <div>
            <label className="block text-sm font-semibold text-gray-700 mb-2">Notes</label>
            <textarea
              value={selectedShift?.notes || ''}
              onChange={(e) => setSelectedShift({ ...selectedShift, notes: e.target.value })}
              rows="3"
              className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
              placeholder="Add shift notes..."
            />
          </div>

          <div className="border-t pt-4">
            <h3 className="text-lg font-semibold text-gray-800 mb-3">Requirements</h3>
            <div className="space-y-3">
              <label className="flex items-center gap-3 cursor-pointer">
                <input
                  type="checkbox"
                  checked={selectedShift?.postOrders || false}
                  onChange={(e) => setSelectedShift({ ...selectedShift, postOrders: e.target.checked })}
                  className="w-5 h-5 text-blue-600 rounded focus:ring-2 focus:ring-blue-500"
                />
                <div className="flex items-center gap-2">
                  <FileText className="w-5 h-5 text-blue-600" />
                  <span className="font-medium text-gray-700">Acknowledge Post Orders</span>
                </div>
              </label>

              <label className="flex items-center gap-3 cursor-pointer">
                <input
                  type="checkbox"
                  checked={selectedShift?.requirePicsHourly || false}
                  onChange={(e) => setSelectedShift({ ...selectedShift, requirePicsHourly: e.target.checked })}
                  className="w-5 h-5 text-blue-600 rounded focus:ring-2 focus:ring-blue-500"
                />
                <div className="flex items-center gap-2">
                  <Camera className="w-5 h-5 text-purple-600" />
                  <span className="font-medium text-gray-700">Require Hourly Pictures</span>
                </div>
              </label>

              <label className="flex items-center gap-3 cursor-pointer">
                <input
                  type="checkbox"
                  checked={selectedShift?.requireDAR || false}
                  onChange={(e) => setSelectedShift({ ...selectedShift, requireDAR: e.target.checked })}
                  className="w-5 h-5 text-blue-600 rounded focus:ring-2 focus:ring-blue-500"
                />
                <div className="flex items-center gap-2">
                  <CheckSquare className="w-5 h-5 text-green-600" />
                  <span className="font-medium text-gray-700">Daily Activity Report (DAR)</span>
                </div>
              </label>
            </div>
          </div>

          <div className="flex gap-3 pt-4">
            <button
              onClick={handleSaveShift}
              className="flex-1 bg-gradient-to-r from-blue-600 to-purple-600 text-white py-3 rounded-lg font-semibold hover:shadow-lg transform hover:scale-[1.02] transition-all"
            >
              Save Shift
            </button>
            <button
              onClick={() => setShowShiftModal(false)}
              className="px-6 py-3 border-2 border-gray-300 rounded-lg font-semibold text-gray-700 hover:bg-gray-50"
            >
              Cancel
            </button>
          </div>
        </div>
      </div>
    </div>
  );

  const DayView = () => (
    <div className="bg-white rounded-xl shadow-lg overflow-hidden">
      <div className="overflow-x-auto">
        <table className="w-full">
          <thead className="bg-gradient-to-r from-blue-600 to-purple-600 text-white">
            <tr>
              <th className="px-4 py-3 text-left font-semibold sticky left-0 bg-blue-600">Time</th>
              {employees.map(emp => (
                <th key={emp} className="px-4 py-3 text-left font-semibold min-w-[200px]">{emp}</th>
              ))}
            </tr>
          </thead>
          <tbody>
            {timeSlots.map((time, idx) => (
              <tr key={time} className={idx % 2 === 0 ? 'bg-gray-50' : 'bg-white'}>
                <td className="px-4 py-3 font-semibold text-gray-700 sticky left-0 bg-gray-100 border-r">
                  {time}
                </td>
                {employees.map(emp => {
                  const shift = shifts.find(s => 
                    s.employee === emp && 
                    s.clockIn <= time && 
                    s.clockOut > time &&
                    s.date === selectedDate.toISOString().split('T')[0]
                  );
                  
                  return (
                    <td
                      key={emp}
                      onClick={() => !shift && handleCellClick(time, emp)}
                      className={`px-4 py-3 border-b border-gray-200 cursor-pointer hover:bg-blue-50 transition-colors ${
                        shift ? 'bg-gradient-to-r from-blue-100 to-purple-100' : ''
                      }`}
                    >
                      {shift && (
                        <div 
                          onClick={(e) => {
                            e.stopPropagation();
                            setSelectedShift(shift);
                            setShowShiftModal(true);
                          }}
                          className="text-xs space-y-1 p-2 rounded-lg bg-white shadow-sm hover:shadow-md transition-shadow"
                        >
                          <div className="font-semibold text-blue-700">{shift.client}</div>
                          <div className="text-gray-600">{shift.position}</div>
                          <div className="flex items-center gap-1 text-gray-500">
                            <Clock className="w-3 h-3" />
                            {shift.clockIn} - {shift.clockOut}
                          </div>
                          {shift.chatId && (
                            <button className="flex items-center gap-1 text-blue-600 hover:text-blue-800 font-medium">
                              <MessageCircle className="w-3 h-3" />
                              Join Chat
                            </button>
                          )}
                        </div>
                      )}
                    </td>
                  );
                })}
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );

  const WeekView = () => {
    const weekDates = getWeekDates();
    
    return (
      <div className="bg-white rounded-xl shadow-lg overflow-hidden">
        <div className="overflow-x-auto">
          <table className="w-full">
            <thead className="bg-gradient-to-r from-blue-600 to-purple-600 text-white">
              <tr>
                <th className="px-4 py-3 text-left font-semibold sticky left-0 bg-blue-600">Employee</th>
                {weekDates.map(date => (
                  <th key={date.toISOString()} className="px-4 py-3 text-center font-semibold min-w-[150px]">
                    <div>{date.toLocaleDateString('en-US', { weekday: 'short' })}</div>
                    <div className="text-sm font-normal">{date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}</div>
                  </th>
                ))}
              </tr>
            </thead>
            <tbody>
              {employees.map((emp, idx) => (
                <tr key={emp} className={idx % 2 === 0 ? 'bg-gray-50' : 'bg-white'}>
                  <td className="px-4 py-3 font-semibold text-gray-700 sticky left-0 bg-gray-100 border-r">
                    {emp}
                  </td>
                  {weekDates.map(date => {
                    const dateStr = date.toISOString().split('T')[0];
                    const dayShifts = shifts.filter(s => s.employee === emp && s.date === dateStr);
                    
                    return (
                      <td
                        key={date.toISOString()}
                        onClick={() => handleCellClick('08:00', emp)}
                        className="px-2 py-3 border-b border-gray-200 cursor-pointer hover:bg-blue-50 transition-colors"
                      >
                        <div className="space-y-1">
                          {dayShifts.map(shift => (
                            <div
                              key={shift.id}
                              onClick={(e) => {
                                e.stopPropagation();
                                setSelectedShift(shift);
                                setShowShiftModal(true);
                              }}
                              className="text-xs p-2 rounded-lg bg-gradient-to-r from-blue-100 to-purple-100 hover:shadow-md transition-shadow"
                            >
                              <div className="font-semibold text-blue-700 truncate">{shift.client}</div>
                              <div className="flex items-center gap-1 text-gray-600">
                                <Clock className="w-3 h-3" />
                                {shift.clockIn}-{shift.clockOut}
                              </div>
                            </div>
                          ))}
                        </div>
                      </td>
                    );
                  })}
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    );
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-50 to-blue-50">
      {/* Header */}
      <div className="bg-gradient-to-r from-blue-600 to-purple-600 text-white p-4 shadow-lg">
        <div className="max-w-7xl mx-auto flex items-center justify-between">
          <div className="flex items-center gap-3">
            <Menu className="w-6 h-6" />
            <div>
              <div className="flex items-center gap-2">
                <Shield className="w-6 h-6" />
                <h1 className="text-2xl font-bold">Manager</h1>
              </div>
              <div className="text-sm opacity-90">WorkforceOS</div>
            </div>
          </div>
          <Bell className="w-6 h-6" />
        </div>
      </div>

      {/* Tabs */}
      <div className="bg-white border-b shadow-sm sticky top-0 z-40">
        <div className="max-w-7xl mx-auto">
          <div className="flex overflow-x-auto">
            {tabs.map(tab => {
              const Icon = tab.icon;
              return (
                <button
                  key={tab.id}
                  onClick={() => setActiveTab(tab.id)}
                  className={`flex items-center gap-2 px-6 py-4 font-semibold whitespace-nowrap transition-all ${
                    activeTab === tab.id
                      ? 'text-blue-600 border-b-2 border-blue-600 bg-blue-50'
                      : 'text-gray-600 hover:text-blue-600 hover:bg-gray-50'
                  }`}
                >
                  <Icon className="w-5 h-5" />
                  {tab.label}
                </button>
              );
            })}
          </div>
        </div>
      </div>

      {/* Main Content */}
      <div className="max-w-7xl mx-auto p-4">
        {activeTab === 'schedule' && (
          <div className="space-y-4">
            {/* Controls */}
            <div className="bg-white rounded-xl shadow-lg p-4">
              <div className="flex flex-col md:flex-row items-center justify-between gap-4">
                <div className="flex items-center gap-4">
                  <button
                    onClick={() => changeDate(-1)}
                    className="p-2 hover:bg-gray-100 rounded-lg transition-colors"
                  >
                    <ChevronLeft className="w-5 h-5" />
                  </button>
                  <div className="text-center">
                    <div className="text-2xl font-bold text-gray-800">{formatDate(selectedDate)}</div>
                    <div className="text-sm text-gray-500">
                      {selectedDate.toLocaleDateString('en-US', { year: 'numeric' })}
                    </div>
                  </div>
                  <button
                    onClick={() => changeDate(1)}
                    className="p-2 hover:bg-gray-100 rounded-lg transition-colors"
                  >
                    <ChevronRight className="w-5 h-5" />
                  </button>
                </div>

                <div className="flex items-center gap-3">
                  <div className="bg-gray-100 rounded-lg p-1 flex gap-1">
                    <button
                      onClick={() => setViewMode('day')}
                      className={`px-4 py-2 rounded-lg font-semibold transition-all ${
                        viewMode === 'day'
                          ? 'bg-gradient-to-r from-blue-600 to-purple-600 text-white shadow-md'
                          : 'text-gray-600 hover:bg-gray-200'
                      }`}
                    >
                      Day
                    </button>
                    <button
                      onClick={() => setViewMode('week')}
                      className={`px-4 py-2 rounded-lg font-semibold transition-all ${
                        viewMode === 'week'
                          ? 'bg-gradient-to-r from-blue-600 to-purple-600 text-white shadow-md'
                          : 'text-gray-600 hover:bg-gray-200'
                      }`}
                    >
                      Week
                    </button>
                  </div>

                  <button
                    onClick={() => {
                      setSelectedShift({
                        employee: '',
                        position: '',
                        client: '',
                        date: selectedDate.toISOString().split('T')[0],
                        clockIn: '08:00',
                        clockOut: '16:00',
                        notes: '',
                        postOrders: false,
                        requirePicsHourly: false,
                        requireDAR: false,
                      });
                      setShowShiftModal(true);
                    }}
                    className="flex items-center gap-2 bg-gradient-to-r from-blue-600 to-purple-600 text-white px-6 py-2 rounded-lg font-semibold hover:shadow-lg transform hover:scale-105 transition-all"
                  >
                    <Plus className="w-5 h-5" />
                    Add Shift
                  </button>
                </div>
              </div>
            </div>

            {/* Schedule Grid */}
            {viewMode === 'day' ? <DayView /> : <WeekView />}

            {/* Legend */}
            <div className="bg-white rounded-xl shadow-lg p-4">
              <h3 className="font-semibold text-gray-800 mb-3">Legend</h3>
              <div className="flex flex-wrap gap-4">
                <div className="flex items-center gap-2">
                  <div className="w-4 h-4 bg-gradient-to-r from-blue-100 to-purple-100 rounded"></div>
                  <span className="text-sm text-gray-600">Scheduled Shift</span>
                </div>
                <div className="flex items-center gap-2">
                  <MessageCircle className="w-4 h-4 text-blue-600" />
                  <span className="text-sm text-gray-600">Chat Available</span>
                </div>
                <div className="flex items-center gap-2">
                  <Camera className="w-4 h-4 text-purple-600" />
                  <span className="text-sm text-gray-600">Hourly Pics Required</span>
                </div>
                <div className="flex items-center gap-2">
                  <FileText className="w-4 h-4 text-blue-600" />
                  <span className="text-sm text-gray-600">Post Orders</span>
                </div>
              </div>
            </div>
          </div>
        )}

        {activeTab === 'dashboard' && (
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            <div className="bg-white rounded-xl shadow-lg p-6">
              <h3 className="text-lg font-bold text-gray-800 mb-4">Quick Stats</h3>
              <div className="space-y-3">
                <div className="flex items-center justify-between">
                  <span className="text-gray-600">Active Shifts</span>
                  <span className="text-2xl font-bold text-blue-600">12</span>
                </div>
                <div className="flex items-center justify-between">
                  <span className="text-gray-600">Total Employees</span>
                  <span className="text-2xl font-bold text-purple-600">24</span>
                </div>
                <div className="flex items-center justify-between">
                  <span className="text-gray-600">Pending Approvals</span>
                  <span className="text-2xl font-bold text-orange-600">5</span>
                </div>
              </div>
            </div>
          </div>
        )}
      </div>

      {/* Shift Modal */}
      {showShiftModal && <ShiftModal />}
    </div>
  );
}
