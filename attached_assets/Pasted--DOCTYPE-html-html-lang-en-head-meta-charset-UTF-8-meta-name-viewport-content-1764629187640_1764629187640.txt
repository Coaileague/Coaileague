    <script>
        /**
         * COAI LEAGUE - AGENT BEE ENGINE
         * Used for both the small FAB and the large Overlay
         */
        class BeeEngine {
            constructor(canvasId, isSmall = false) {
                this.canvas = document.getElementById(canvasId);
                this.ctx = this.canvas.getContext('2d', { alpha: true });
                this.isSmall = isSmall;
                this.workers = [];
                this.time = 0;
                
                // Config
                this.count = isSmall ? 4 : 8;
                this.color = '#ffd700'; // Gold
                this.mode = 'IDLE';

                this.resize();
                this.initBees();
                
                // Animation Loop
                const loop = () => {
                    this.update();
                    this.draw();
                    requestAnimationFrame(loop);
                };
                loop();
            }

            resize() {
                const rect = this.canvas.parentElement.getBoundingClientRect();
                const dpr = window.devicePixelRatio || 1;
                this.width = rect.width;
                this.height = rect.height;
                this.canvas.width = this.width * dpr;
                this.canvas.height = this.height * dpr;
                this.ctx.scale(dpr, dpr);
                this.cx = this.width / 2;
                this.cy = this.height / 2;
                this.scale = this.width / (this.isSmall ? 60 : 300); // Relative scaling
            }

            initBees() {
                for(let i=0; i<this.count; i++) {
                    this.workers.push({
                        angle: (Math.PI*2/this.count) * i,
                        radius: this.isSmall ? 15 : 80,
                        speed: 0.05 + Math.random()*0.02,
                        offset: Math.random() * 100
                    });
                }
            }

            setMode(mode) {
                this.mode = mode;
                if(mode === 'VOICE') this.color = '#f43f5e';
                else if(mode === 'SUCCESS') this.color = '#10b981';
                else this.color = '#ffd700';
            }

            update() {
                this.time++;
                this.workers.forEach(w => {
                    w.angle += w.speed;
                    // Breathing radius
                    const breath = Math.sin(this.time * 0.05) * (this.isSmall ? 2 : 10);
                    w.currentR = w.radius + breath;
                });
            }

            draw() {
                this.ctx.clearRect(0, 0, this.width, this.height);
                
                // Draw Queen (Center)
                const qSize = this.isSmall ? 6 : 25;
                this.ctx.fillStyle = this.color;
                this.ctx.beginPath();
                this.ctx.arc(this.cx, this.cy, qSize, 0, Math.PI*2);
                this.ctx.fill();
                
                // Glow
                this.ctx.shadowBlur = this.isSmall ? 5 : 20;
                this.ctx.shadowColor = this.color;

                // Draw Workers
                this.workers.forEach(w => {
                    const x = this.cx + Math.cos(w.angle) * w.currentR;
                    const y = this.cy + Math.sin(w.angle) * w.currentR;
                    const size = this.isSmall ? 2 : 8;

                    this.ctx.beginPath();
                    this.ctx.arc(x, y, size, 0, Math.PI*2);
                    this.ctx.fill();
                });
                this.ctx.shadowBlur = 0;
            }
        }

        // --- INIT ---
        // 1. Small FAB Engine (Always running)
        const fabEngine = new BeeEngine('fabCanvas', true);

        // 2. Large Overlay Engine (Runs when opened)
        const mainEngine = new BeeEngine('mainCanvas', false);

        // --- CONTROLLER ---
        window.CoAI = {
            trigger: (mode, title, sub) => {
                const ol = document.getElementById('loader-overlay');
                const t = document.getElementById('overlayTitle');
                const s = document.getElementById('overlaySub');
                
                mainEngine.setMode(mode.toUpperCase());
                t.innerText = title;
                s.innerText = sub;
                t.style.color = mainEngine.color;
                
                ol.classList.add('active');
            },
            hide: () => {
                document.getElementById('loader-overlay').classList.remove('active');
                mainEngine.setMode('IDLE');
            }
        };

        // Handle Resize
        window.addEventListener('resize', () => {
            fabEngine.resize();
            mainEngine.resize();
        });

    </script>
