```sql
  ON tc.constraint_name = ccu.constraint_name
```

**Then for EACH foreign key, generate an orphan check:**

```sql
-- GENERATE orphan checks for every foreign key relationship
```

## 3D: INSERT Guards in Service Layer

Every service that creates records MUST use this pattern:

```typescript
// PATTERN: Safe insert with attribution
async function safeInsert(table: any, data: any, context: RequestContext) {
  // 1. Validate workspace_id exists
  if (!data.workspace_id && !context.workspaceId) {
    throw new Error(`Cannot insert into ${table.name}: no workspace_id`);
  }
  
  // 2. Check for duplicates BEFORE inserting (prevents the agent's duplicate problem)
  if (data.uniqueKey) {
    const existing = await db.select().from(table)
      .where(and(
        eq(table.workspace_id, data.workspace_id || context.workspaceId),
        eq(table[data.uniqueKey.column], data[data.uniqueKey.column])
      ))
      .limit(1);
    
    if (existing.length > 0) {
      // UPDATE existing instead of creating duplicate
      return db.update(table)
        .set({ ...data, updated_at: new Date() })
        .where(eq(table.id, existing[0].id))
        .returning();
    }
  }
  
  // 3. Insert with full attribution
  return db.insert(table).values({
    ...data,
    workspace_id: data.workspace_id || context.workspaceId,
    created_by: context.userId || null,
    created_by_type: context.type || 'system',
    created_at: new Date(),
    updated_at: new Date(),
  }).returning();
}
```

## 3E: Upsert Pattern for Idempotent Operations

The agent keeps creating duplicates because it uses INSERT when it should use UPSERT:

```sql
-- PATTERN: Upsert (insert or update if exists)
-- Use this for any table with a natural unique key

-- Example: org_subscriptions (one per workspace)
INSERT INTO org_subscriptions (workspace_id, tier_id, status, current_period_start, current_period_end)
VALUES ($1, $2, 'active', $3, $4)
ON CONFLICT (workspace_id) 
DO UPDATE SET 
  tier_id = EXCLUDED.tier_id,
  status = EXCLUDED.status,
  current_period_start = EXCLUDED.current_period_start,
  current_period_end = EXCLUDED.current_period_end,
  updated_at = NOW();

-- Example: credit_balances (one per workspace)
INSERT INTO credit_balances (workspace_id, subscription_credits, carryover_credits, purchased_credits)
VALUES ($1, $2, $3, $4)
ON CONFLICT (workspace_id)
DO UPDATE SET
  subscription_credits = EXCLUDED.subscription_credits,
  updated_at = NOW();
```

**RULE FOR THE AGENT: Before every INSERT, ask yourself: "Can this record already exist?" If yes, use UPSERT (ON CONFLICT DO UPDATE), NOT raw INSERT.**

```typescript
// server/crons/databaseHealthCheck.ts

// Runs daily at 3 AM
async function databaseHealthCheck() {
  const issues = [];
  
  // Check for NULL workspace_ids across all tables
  const nullWorkspaceResults = await checkNullWorkspaceIds();
  if (nullWorkspaceResults.length > 0) {
    issues.push(`NULL workspace_id found in: ${nullWorkspaceResults.map(r => r.table_name).join(', ')}`);
  }
  
  // Check for orphaned foreign keys
  const orphanResults = await checkOrphanedForeignKeys();
  if (orphanResults.length > 0) {
    issues.push(`Orphaned references found: ${orphanResults.map(r => r.relationship).join(', ')}`);
  }
  
  // Check for duplicate billing records
  const dupeResults = await checkBillingDuplicates();
  if (dupeResults.length > 0) {
    issues.push(`Duplicate billing records: ${JSON.stringify(dupeResults)}`);
  }
  
  // Check billing chain integrity
  const billingGaps = await checkBillingChain();
  if (billingGaps.length > 0) {
    issues.push(`Billing chain gaps: ${billingGaps.map(g => g.issue).join(', ')}`);
  }
  
  if (issues.length > 0) {
    // Send alert notification to org owner / platform admin
    await notificationService.sendSystemAlert({
      title: 'Database Health Check — Issues Found',
      body: issues.join('\n'),
      severity: 'high',
    });
    
    // Log to audit trail
    await auditLog.log({
      workspaceId: 'system',
      actorType: 'cron',
      action: 'health_check.issues_found',
      entityType: 'system',
      changeType: 'action',
      metadata: { issues },
    });
  }
}
```
