/**
 * Calculate distance between two GPS coordinates using Haversine formula
 * Returns distance in meters
 */
export function calculateDistance(
  point1: { lat: number; lng: number },
  point2: { lat: number; lng: number }
): number {
  const R = 6371e3; // Earth's radius in meters
  const φ1 = (point1.lat * Math.PI) / 180;
  const φ2 = (point2.lat * Math.PI) / 180;
  const Δφ = ((point2.lat - point1.lat) * Math.PI) / 180;
  const Δλ = ((point2.lng - point1.lng) * Math.PI) / 180;

  const a =
    Math.sin(Δφ / 2) * Math.sin(Δφ / 2) +
    Math.cos(φ1) * Math.cos(φ2) * Math.sin(Δλ / 2) * Math.sin(Δλ / 2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));

  return R * c; // Distance in meters
}

const GEOFENCE_RADIUS = 100; // 100 meters = 328 feet

async function validateClockIn(
  employeeId: string,
  latitude: number,
  longitude: number
) {
  // 1. Get employee's current shift
  const currentShift = await getCurrentShift(employeeId);
  
  if (!currentShift) {
    return {
      allowed: false,
      reason: 'No shift scheduled for this time',
      code: 'NO_SHIFT_SCHEDULED'
    };
  }

  // 2. Get the site location for this shift
  const site = await getSiteLocation(currentShift.siteId);
  
  if (!site.latitude || !site.longitude) {
    // Site doesn't have GPS coordinates - allow clock in but log warning
    await logWarning({
      type: 'SITE_MISSING_GPS',
      siteId: site.id,
      siteName: site.name
    });
    return { allowed: true, warning: 'Site GPS not configured' };
  }

  // 3. Calculate distance
  const distanceInMeters = calculateDistance(
    { lat: latitude, lng: longitude },
    { lat: site.latitude, lng: site.longitude }
  );

  // 4. Check geofence
  if (distanceInMeters > GEOFENCE_RADIUS) {
    // VIOLATION - Employee is not at correct location
    
    // Log GPS violation
    await db.gpsViolations.create({
      employeeId,
      shiftId: currentShift.id,
      siteId: site.id,
      attemptedLocation: { lat: latitude, lng: longitude },
      correctLocation: { lat: site.latitude, lng: site.longitude },
      distanceMeters: Math.round(distanceInMeters),
      timestamp: new Date(),
      blocked: true
    });

    // Emit event to Trinity
    await platformEventBus.emit('gps.violation', {
      employeeId,
      employeeName: currentShift.employeeName,
      siteName: site.name,
      distanceMeters: Math.round(distanceInMeters),
      timestamp: new Date()
    });

    // Send immediate SMS to manager via Twilio
    const manager = await getShiftManager(currentShift.id);
    if (manager && manager.phone) {
      await smsService.send({
        to: manager.phone,
        message: `🚨 GPS Alert: ${currentShift.employeeName} attempted to clock in ${Math.round(distanceInMeters)}m away from ${site.name}. Clock-in blocked.`
      });
    }

    // Update employee reliability score (penalty)
    await updateEmployeeScore(employeeId, {
      gpsViolation: -5, // -5 points per violation
      lastViolation: new Date()
    });

    return {
      allowed: false,
      reason: `You must be within ${GEOFENCE_RADIUS}m of ${site.name} to clock in. You are ${Math.round(distanceInMeters)}m away.`,
      code: 'GPS_GEOFENCE_VIOLATION',
      distanceMeters: Math.round(distanceInMeters)
    };
  }

  // 5. Valid - within geofence
  return {
    allowed: true,
    location: { lat: latitude, lng: longitude },
    siteVerified: site.name,
    distanceMeters: Math.round(distanceInMeters)
  };
}

import { getQuickBooksClient } from './quickbooksClient';
import { gemini } from '../ai-brain/providers/geminiClient';

export async function syncInvoiceToQuickBooks(invoiceId: string) {
  const invoice = await getInvoice(invoiceId);
  const workspace = await getWorkspace(invoice.workspaceId);

  try {
    // 1. Get QuickBooks client
    const qbClient = await getQuickBooksClient(workspace.id);

    // 2. Map CoAIleague invoice to QuickBooks format
    const qbInvoice = {
      CustomerRef: {
        value: invoice.client.qbCustomerId // From migration mapping
      },
      Line: invoice.lineItems.map(item => ({
        Amount: item.amount,
        DetailType: 'SalesItemLineDetail',
        SalesItemLineDetail: {
          ItemRef: {
            value: workspace.qbSecurityServiceItemId // Security services revenue item
          },
          Qty: item.hours,
          UnitPrice: item.rate,
          Description: `${item.date} - ${item.employeeName} at ${item.siteName} (${item.hours} hrs @ $${item.rate}/hr)`
        }
      })),
      DueDate: invoice.dueDate,
      TxnDate: invoice.invoiceDate,
      DocNumber: invoice.invoiceNumber
    };

    // 3. Create invoice in QuickBooks
    const qbResponse = await qbClient.createInvoice(qbInvoice);

    // 4. Store QB invoice ID
    await updateInvoice(invoiceId, {
      qbInvoiceId: qbResponse.Invoice.Id,
      qbSyncStatus: 'synced',
      qbSyncedAt: new Date()
    });

    // 5. Trinity logs success
    await trinity.log({
      action: 'quickbooks.invoice_synced',
      invoiceId,
      qbInvoiceId: qbResponse.Invoice.Id,
      amount: invoice.total,
      client: invoice.client.name,
      success: true
    });

    return { success: true, qbInvoiceId: qbResponse.Invoice.Id };

  } catch (error) {
    // Gemini analyzes the error
    const errorAnalysis = await analyzeQuickBooksError(error, invoice);

    if (errorAnalysis.action === 'RETRY') {
      // Auto-retry with modifications
      return await retryInvoiceSync(invoiceId, errorAnalysis.modifications);
    } else {
      // Escalate to admin
      await escalateToAdmin({
        type: 'quickbooks_sync_failed',
        invoiceId,
        error: error.message,
        geminiAnalysis: errorAnalysis,
        recommendation: errorAnalysis.recommendation
      });

      await updateInvoice(invoiceId, {
        qbSyncStatus: 'failed',
        qbSyncError: error.message,
        qbSyncFailedAt: new Date()
      });

      return { success: false, error: error.message };
    }
  }
}

async function analyzeQuickBooksError(error: any, invoice: any) {
  const model = gemini.getGenerativeModel({ model: 'gemini-2.5-pro' });

  const prompt = `You are a QuickBooks integration specialist analyzing a sync error.

ERROR DETAILS:
${JSON.stringify(error, null, 2)}

INVOICE DATA:
Customer: ${invoice.client.name} (QB ID: ${invoice.client.qbCustomerId})
Amount: $${invoice.total}
Line Items: ${invoice.lineItems.length}
Invoice Number: ${invoice.invoiceNumber}

ANALYZE:
1. Root cause of error
2. Recommended action: RETRY | FIX_DATA | ESCALATE
3. If RETRY: What modifications to make?
4. If FIX_DATA: What data needs correction?
5. If ESCALATE: Why human intervention needed?

Respond in JSON format:
{
  "rootCause": "string",
  "action": "RETRY" | "FIX_DATA" | "ESCALATE",
  "modifications": {},
  "recommendation": "string"
}`;

  const result = await model.generateContent(prompt);
  const analysis = JSON.parse(result.response.text());

  // LLM Judge validates retry decision
  if (analysis.action === 'RETRY') {
    const judgeApproval = await llmJudge.evaluate({
      action: 'quickbooks.retry_invoice_sync',
      context: analysis,
      riskLevel: 'medium'
    });

    if (!judgeApproval.approved) {
      analysis.action = 'ESCALATE';
      analysis.recommendation = judgeApproval.reason;
    }
  }

  return analysis;
}
