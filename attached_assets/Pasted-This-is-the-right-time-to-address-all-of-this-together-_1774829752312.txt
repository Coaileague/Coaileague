```sql
DECLARE
  slug TEXT;
BEGIN
  -- Generate URL-safe slug from company name
  slug := LOWER(REGEXP_REPLACE(NEW.company_name, '[^A-Za-z0-9]', '', 'g'));
  slug := LEFT(slug, 20);
  
  -- Ensure uniqueness
  WHILE EXISTS (SELECT 1 FROM workspaces WHERE email_slug = slug) LOOP

  NEW.email_slug := slug;

  NEW.support_email := 'support.' || slug || '@mail.coaileague.com';
  NEW.calloffs_email := 'calloffs.' || slug || '@mail.coaileague.com';
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
```

**Trinity Email Intelligence Layer:**

Build `server/services/trinityEmailProcessor.ts`:

```typescript
// Processes all inbound emails routed to any Trinity-managed address
// Routes to correct handler based on TO address

export class TrinityEmailProcessor {
  
  async processInbound(emailData: {
    to: string;
    from: string;
    subject: string;
    body: string;
    htmlBody?: string;
    attachments?: any[];
    messageId: string;
    inReplyTo?: string;
    threadId?: string;
  }) {
    // 1. Identify workspace from TO address
    const workspace = await this.resolveWorkspaceFromEmail(emailData.to);
    if (!workspace) {
      // Unknown address — log and ignore or bounce
      logger.warn(`Inbound email to unknown address: ${emailData.to}`);
      return;
    }
    
    // 2. Identify sender
    const sender = await this.identifySender(emailData.from, workspace.id);
    
    // 3. Store email in database
    const storedEmail = await this.storeInboundEmail(emailData, workspace.id, sender);
    
    // 4. Route to correct Trinity handler based on TO address pattern
    const addressType = this.getAddressType(emailData.to, workspace);
    
    switch(addressType) {
      case 'careers':
        await this.handleCareersEmail(storedEmail, workspace, sender);
        break;
      case 'verify':
        await this.handleVerificationEmail(storedEmail, workspace, sender);
        break;
      case 'support':
        await this.handleSupportEmail(storedEmail, workspace, sender);
        break;
      case 'calloffs':
        await this.handleCalloffEmail(storedEmail, workspace, sender);
        break;
      case 'trinity':
        await this.handleTrinityDirectEmail(storedEmail, workspace, sender);
        break;
      case 'main':
        await this.handleMainInboxEmail(storedEmail, workspace, sender);
        break;
    }
  }
  
  // Trinity replies maintaining thread context
  async replyToEmail(params: {
    originalMessageId: string;
    toAddress: string;
    fromAddress: string; // Trinity's workspace email
    subject: string;
    body: string;
    workspaceId: string;
    templateId?: string;
  }) {
    // Send via Resend maintaining thread
    await resend.emails.send({
      from: params.fromAddress,
      to: params.toAddress,
      subject: params.subject.startsWith('Re:') ? params.subject : `Re: ${params.subject}`,
      html: params.body,
      headers: {
        'In-Reply-To': params.originalMessageId,
        'References': params.originalMessageId
      }
    });
    
    // Log in NDS
    await notificationDeliveryService.trackEmailSent({
      workspaceId: params.workspaceId,
      to: params.toAddress,
      from: params.fromAddress,
      subject: params.subject,
      sentBy: 'trinity-system-actor-000000000000'
    });
  }
  
  getAddressType(toAddress: string, workspace: any): string {
    if (toAddress === workspace.careers_email) return 'careers';
    if (toAddress === workspace.verify_email) return 'verify';
    if (toAddress === workspace.support_email) return 'support';
    if (toAddress === workspace.calloffs_email) return 'calloffs';
    if (toAddress === workspace.trinity_email) return 'trinity';
    return 'main';
  }
  
  async resolveWorkspaceFromEmail(toAddress: string) {
    return await db.query(`
      SELECT * FROM workspaces 
      WHERE trinity_email = $1 
        OR careers_email = $1 
        OR verify_email = $1 
        OR support_email = $1 
        OR calloffs_email = $1
      LIMIT 1
    `, [toAddress]).then(r => r.rows[0]);
  }
}
```

```typescript
async function checkConsentBeforeSend(userId: string, channel: string): Promise<boolean> {
  // Critical operational messages bypass consent check
  // (panic alerts, security incidents, payment failures)
  const EXEMPT_TYPES = ['panic_alert', 'duress_emergency', 'payment_failed', 'security_breach'];
  
  const consent = await db.query(`
    SELECT trinity_voice_calls, trinity_sms, trinity_email, all_opted_out
    FROM consent_preferences WHERE user_id = $1
  `, [userId]);
  
  if (!consent.rows[0]) return false; // No consent record = no consent
  if (consent.rows[0].all_opted_out) return false;
  
  switch(channel) {
    case 'sms': return consent.rows[0].trinity_sms;
    case 'voice': return consent.rows[0].trinity_voice_calls;
    case 'email': return consent.rows[0].trinity_email;
    default: return true; // In-app notifications always allowed
  }
}
```

```typescript
// server/services/interviewChatOrchestrator.ts

export class InterviewChatOrchestrator {
  
  // Trinity sends first message when candidate joins
  async startInterview(chatroomId: string) {
    const room = await this.getChatroom(chatroomId);
    const candidate = await this.getCandidate(room.candidate_id);
    const questions = await this.getQuestionsForPosition(room.workspace_id, candidate.position_applied);
    
    // Update chatroom with questions
    await db.query(
      'UPDATE interview_chatrooms SET questions_asked = $1, status = $2, started_at = NOW() WHERE id = $3',
      [JSON.stringify(questions), 'active', chatroomId]
    );
    
    // Trinity sends welcome
    await this.sendTrinityMessage(chatroomId, `
      Hello ${candidate.full_name}! 
      
      I'm Trinity, the AI assistant for ${room.workspace_name}. 
      I'll be conducting your initial interview today for the ${candidate.position_applied} position.
      
      This interview should take about 10-15 minutes. I'll ask you several questions and 
      your responses will be shared with our hiring team for review.
      
      Please answer as naturally and completely as you can. There are no trick questions.
      
      Let's begin. ${questions[0].question_text}
    `);
    
    // Notify human co-pilot if assigned
    if (room.human_copilot_user_id) {
      await this.sendCopilotWhisper(chatroomId, `
        Interview started. Candidate: ${candidate.full_name}
        Position: ${candidate.position_applied}
        ${questions.length} questions queued.
        Initial score: Pending first response.
      `);
    }
  }
  
  // Process candidate response
  async processCandidateResponse(chatroomId: string, responseText: string) {
    const room = await this.getChatroom(chatroomId);
    const currentQuestion = room.questions_asked[room.current_question_index];
    
    // Score the response using AI
    const score = await this.scorResponse(responseText, currentQuestion);
    
    // Store response with score
    const responses = [...room.responses_received, {
      questionId: currentQuestion.id,
      questionText: currentQuestion.question_text,
      response: responseText,
      score: score.score,
      flags: score.flags,
      scoredAt: new Date().toISOString()
    }];
    
    // Send co-pilot whisper with scoring
    if (room.human_copilot_user_id) {
      await this.sendCopilotWhisper(chatroomId, `
        ⚡ Trinity Scoring: ${score.score}/10
        Flags: ${score.flags.join(', ') || 'None'}
        Assessment: ${score.assessment}
        Suggested follow-up: "${score.suggestedFollowUp}"
      `);
    }
    
    // Move to next question or close
    const nextIndex = room.current_question_index + 1;
    
    if (nextIndex >= room.questions_asked.length) {
      // Interview complete
      await this.completeInterview(chatroomId, responses);
    } else {
      // Ask next question (adaptive based on score)
      const nextQuestion = this.getAdaptiveNextQuestion(
        room.questions_asked, nextIndex, score
      );
      
      await db.query(`
        UPDATE interview_chatrooms 
        SET current_question_index = $1, responses_received = $2
        WHERE id = $3
      `, [nextIndex, JSON.stringify(responses), chatroomId]);
      
      await this.sendTrinityMessage(chatroomId, nextQuestion.question_text);
    }
  }
  
  // Score response using AI
  async scorResponse(response: string, question: any): Promise<ScoringResult> {
    const prompt = `
      Interview question: "${question.question_text}"
      Ideal response criteria: ${JSON.stringify(question.scoring_criteria)}
      Candidate response: "${response}"
      
      Score this response 1-10 and identify any flags.
      Return JSON: { score: number, flags: string[], assessment: string, suggestedFollowUp: string }
    `;
    
    const result = await openai.chat.completions.create({
      model: 'gpt-4',
      messages: [{ role: 'user', content: prompt }],
      response_format: { type: 'json_object' }
    });
    
    return JSON.parse(result.choices[0].message.content);
  }
  
  // Complete interview and generate scorecard
  async completeInterview(chatroomId: string, responses: any[]) {
    const room = await this.getChatroom(chatroomId);
    const candidate = await this.getCandidate(room.candidate_id);
    
    // Calculate overall score
    const scores = responses.map(r => r.score);
    const overallScore = Math.round(scores.reduce((a, b) => a + b, 0) / scores.length * 10);
    
    // Generate recommendation
    const recommendation = 
      overallScore >= 80 ? 'strong_hire' :
      overallScore >= 65 ? 'hire' :
      overallScore >= 50 ? 'maybe' : 'no_hire';
    
    // Generate scorecard PDF
    const pdfUrl = await this.generateScorecardPdf(room, responses, overallScore, recommendation);
    
    // Update chatroom
    await db.query(`
      UPDATE interview_chatrooms SET
        status = 'completed',
        overall_score = $1,
        trinity_recommendation = $2,
        pdf_scorecard_url = $3,
        pdf_generated_at = NOW(),
        completed_at = NOW(),
        responses_received = $4
      WHERE id = $5
    `, [overallScore, recommendation, pdfUrl, JSON.stringify(responses), chatroomId]);
    
    // Trinity closing message to candidate
    await this.sendTrinityMessage(chatroomId, `
      Thank you ${candidate.full_name} for your time today! 
      
      Your interview is now complete. Our hiring team will review your responses 
      and be in touch within 2-3 business days.
      
      We appreciate your interest in joining ${room.workspace_name}!
    `);
    
    // Notify org with scorecard
    await notificationDeliveryService.send({
      workspaceId: room.workspace_id,
      type: 'interview_complete',
      data: {
        candidateName: candidate.full_name,
        position: candidate.position_applied,
        score: overallScore,
        recommendation,
        scorecardUrl: pdfUrl
      },
      targetRoles: ['org_owner', 'org_admin']
    });
    
    // Advance candidate stage
    await db.query(
      'UPDATE interview_candidates SET stage = $1 WHERE id = $2',
      ['awaiting_decision', room.candidate_id]
    );
  }
}
```

```typescript
// server/middleware/requireLegalAcceptance.ts
export async function requireLegalAcceptance(req: Request, res: Response, next: NextFunction) {
  if (!req.user) return next(); // Auth middleware handles this
  
  // Skip for legal routes themselves
  const exemptPaths = ['/api/legal', '/api/auth', '/health', '/status'];
  if (exemptPaths.some(p => req.path.startsWith(p))) return next();
  
  // Check if user has accepted current agreements
  const pending = await db.query(`
    SELECT la.id FROM legal_agreements la
    WHERE la.is_current = true
    AND la.agreement_type = ANY($1::text[])
    AND NOT EXISTS (
      SELECT 1 FROM user_legal_acceptances ula
      WHERE ula.user_id = $2
      AND ula.agreement_type = la.agreement_type
      AND ula.version_accepted = la.version
      AND ula.revoked_at IS NULL
    )
    LIMIT 1
  `, [getRequiredAgreementsForRole(req.user.role), req.user.id]);
  
  if (pending.rows.length > 0) {
    return res.status(403).json({ 
      error: 'LEGAL_ACCEPTANCE_REQUIRED',
      message: 'Please accept the required agreements to continue'
    });
  }
  
  next();
}
```
