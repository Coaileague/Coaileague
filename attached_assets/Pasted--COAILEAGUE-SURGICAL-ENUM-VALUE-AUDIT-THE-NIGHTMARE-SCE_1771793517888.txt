## 3.1 actor_type Enum Mapping

```typescript
// CREATE THIS FILE: server/constants/actorTypes.ts

/**
 * ACTOR TYPE ENUM VALUES
 * 
 * The database enum 'actor_type' ONLY accepts these values.
 * Any other value will cause an insert to FAIL.
 */

export const ACTOR_TYPES = {
  END_USER: 'END_USER',
  SUPPORT_STAFF: 'SUPPORT_STAFF', 
  AI_AGENT: 'AI_AGENT',
  SYSTEM: 'SYSTEM',
} as const;

export type ActorType = typeof ACTOR_TYPES[keyof typeof ACTOR_TYPES];

/**
 * Map common code values to valid enum values
 */
export function normalizeActorType(value: string): ActorType {
  const mapping: Record<string, ActorType> = {
    // Lowercase variations
    'system': ACTOR_TYPES.SYSTEM,
    'user': ACTOR_TYPES.END_USER,
    'end_user': ACTOR_TYPES.END_USER,
    'enduser': ACTOR_TYPES.END_USER,
    'admin': ACTOR_TYPES.SUPPORT_STAFF,
    'support': ACTOR_TYPES.SUPPORT_STAFF,
    'support_staff': ACTOR_TYPES.SUPPORT_STAFF,
    'staff': ACTOR_TYPES.SUPPORT_STAFF,
    'ai': ACTOR_TYPES.AI_AGENT,
    'ai_agent': ACTOR_TYPES.AI_AGENT,
    'bot': ACTOR_TYPES.AI_AGENT,
    'webhook': ACTOR_TYPES.SYSTEM,
    'automation': ACTOR_TYPES.SYSTEM,
    'cron': ACTOR_TYPES.SYSTEM,
    
    // Already correct (uppercase)
    'END_USER': ACTOR_TYPES.END_USER,
    'SUPPORT_STAFF': ACTOR_TYPES.SUPPORT_STAFF,
    'AI_AGENT': ACTOR_TYPES.AI_AGENT,
    'SYSTEM': ACTOR_TYPES.SYSTEM,
  };
  
  const normalized = mapping[value] || mapping[value.toLowerCase()];
  
  if (!normalized) {
    console.warn(`Unknown actor type: ${value}, defaulting to SYSTEM`);
    return ACTOR_TYPES.SYSTEM;
  }
  
  return normalized;
}

/**
 * Validate that a value is a valid actor type
 */
export function isValidActorType(value: string): value is ActorType {
  return Object.values(ACTOR_TYPES).includes(value as ActorType);
}
```

## 5.2 Option B: Add Database Trigger (Safety Net)

```sql
-- Create a trigger that normalizes actor_type values before insert
CREATE OR REPLACE FUNCTION normalize_actor_type()
RETURNS TRIGGER AS $$
BEGIN
  -- Normalize common lowercase values to uppercase
  NEW.actor_type = CASE UPPER(NEW.actor_type)
    WHEN 'SYSTEM' THEN 'SYSTEM'
    WHEN 'USER' THEN 'END_USER'
    WHEN 'END_USER' THEN 'END_USER'
    WHEN 'ADMIN' THEN 'SUPPORT_STAFF'
    WHEN 'SUPPORT_STAFF' THEN 'SUPPORT_STAFF'
    WHEN 'STAFF' THEN 'SUPPORT_STAFF'
    WHEN 'AI' THEN 'AI_AGENT'
    WHEN 'AI_AGENT' THEN 'AI_AGENT'
    WHEN 'BOT' THEN 'AI_AGENT'
    WHEN 'WEBHOOK' THEN 'SYSTEM'
    WHEN 'AUTOMATION' THEN 'SYSTEM'
    ELSE 'SYSTEM'  -- Default fallback
  END;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Apply to credit_transactions
CREATE TRIGGER normalize_credit_transactions_actor_type
  BEFORE INSERT OR UPDATE ON credit_transactions
  FOR EACH ROW
  EXECUTE FUNCTION normalize_actor_type();

-- Apply to write_ahead_log
CREATE TRIGGER normalize_wal_actor_type
  BEFORE INSERT OR UPDATE ON write_ahead_log
  FOR EACH ROW
  EXECUTE FUNCTION normalize_actor_type();
```
