// 🚀 TRINITY HOLISTIC GROWTH ENGINE
// "The code that connects the dots"

class HolisticGrowthEngine {
  constructor(core) {
    this.core = core;
  }

  // The Weekly "State of the Union" Scan
  async analyzeBusinessHealth(orgId) {
    // 1. FETCH THE SILOED DATA
    const goals = await this.core.db.getGoals(orgId);          // "Grow 20%"
    const income = await this.core.finance.getIncome(orgId);   // $50k
    const spending = await this.core.finance.getSpending(orgId); // $42k
    const schedule = await this.core.scheduler.getFutureShifts(orgId); // 500 hours

    // 2. SYNTHESIZE (The Logic)
    const currentMargin = (income - spending) / income;
    const laborPercent = spending.payroll / income;

    // 3. GENERATE STRATEGY
    const strategies = [];

    // Check: Is Manpower Killing Growth?
    if (laborPercent > 0.40) { // If labor is >40% of income
      strategies.push({
        type: "MANPOWER_OPTIMIZATION",
        insight: `Labor costs are ${laborPercent * 100}% of income. Too high for growth.`,
        action: "Trigger 'ScheduleMaster' to optimize for budget cap.",
        savings_est: (spending.payroll * 0.10) // Est 10% savings
      });
    }

    // Check: Is Cashflow Ready for Investment?
    if (goals.type === 'EXPANSION' && currentMargin > 0.25) {
      strategies.push({
        type: "GROWTH_SIGNAL",
        insight: "Margins are healthy (25%+). You have capital to deploy.",
        action: "Analyze 'Market Watchtower' for ad spend opportunities."
      });
    }

    return strategies;
  }
}
